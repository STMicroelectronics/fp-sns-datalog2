70void ExtiDefISR(uint16_t exti_pin)
71{
72 EXTIPin2CallbckMap xMap = EXTI_GET_P2F_MAP();
73 for (int i = 0; xMap[i].pfCallback != NULL; i++)
74 {
75 if (__HAL_GPIO_EXTI_GET_IT(xMap[i].nPin) && exti_pin == xMap[i].nPin)
76 {
77 /* EXTI line interrupt detected */
78 __HAL_GPIO_EXTI_CLEAR_IT(xMap[i].nPin);
79 xMap[i].pfCallback(xMap[i].nPin);
80 }
81 }
82}
