112EPowerMode AppPowerModeHelper_vtblComputeNewPowerMode(IAppPowerModeHelper *_this, const SysEvent event)
113{
114 assert_param(_this != NULL);
115 AppPowerModeHelper_t *p_obj = (AppPowerModeHelper_t *)_this;
116
117 EPowerMode power_mode = p_obj->status.active_power_mode;
118
119 switch (event.xEvent.nSource)
120 {
121
122 case SYS_PM_EVT_SRC_DATALOG:
123 if (power_mode == E_POWER_MODE_SLEEP_1)
124 {
125 power_mode = E_POWER_MODE_STATE1;
126 }
127 else if (power_mode == E_POWER_MODE_STATE1)
128 {
129 power_mode = E_POWER_MODE_SENSORS_ACTIVE;
130 }
131 else if (power_mode == E_POWER_MODE_SENSORS_ACTIVE)
132 {
133 power_mode = E_POWER_MODE_STATE1;
134 }
135 break;
136
137 case SYS_PM_EVT_SRC_PB:
138 if (power_mode == E_POWER_MODE_STATE1)
139 {
140 power_mode = E_POWER_MODE_SENSORS_ACTIVE;
141 }
142 else if (power_mode == E_POWER_MODE_SENSORS_ACTIVE)
143 {
144 power_mode = E_POWER_MODE_STATE1;
145 }
146 else if (power_mode == E_POWER_MODE_SLEEP_1)
147 {
148 power_mode = E_POWER_MODE_STATE1;
149 }
150 break;
151
152 case SYS_PM_EVT_SRC_LP_TIMER:
153 if (power_mode == E_POWER_MODE_STATE1)
154 {
155 power_mode = E_POWER_MODE_SLEEP_1;
156 }
157 break;
158
159 default:
160
161 SYS_DEBUGF(SYS_DBG_LEVEL_VERBOSE, ("PMH: wrong SysEvent.\r\n"));
162
163 sys_error_handler();
164 break;
165 }
166
167#ifdef SYS_DEBUG
168 SYS_DEBUGF(SYS_DBG_LEVEL_VERBOSE, ("PMH: new PM:%u-%u.\r\n", p_obj->status.active_power_mode, power_mode));
169#endif
170
171 return power_mode;
172}

174boolean_t AppPowerModeHelper_vtblCheckPowerModeTransaction(IAppPowerModeHelper *_this,
175 const EPowerMode active_power_mode, const EPowerMode new_power_mode)
176{
177 UNUSED(_this);
178 boolean_t res = FALSE;
179
180 switch (active_power_mode)
181 {
182 case E_POWER_MODE_STATE1:
183 if ((new_power_mode == E_POWER_MODE_SENSORS_ACTIVE) || (new_power_mode == E_POWER_MODE_SLEEP_1))
184 {
185 res = TRUE;
186 }
187 break;
188 case E_POWER_MODE_SENSORS_ACTIVE:
189 if (new_power_mode == E_POWER_MODE_STATE1)
190 {
191 res = TRUE;
192 }
193 break;
194 case E_POWER_MODE_SLEEP_1:
195 if (new_power_mode == E_POWER_MODE_STATE1)
196 {
197 res = TRUE;
198 }
199 break;
200 default:
201 res = FALSE;
202 }
203
204 if (res == FALSE)
205 {
206
207 SYS_DEBUGF(SYS_DBG_LEVEL_VERBOSE, ("PMH: ERR PM transaction %u -> %u\r\n", (uint8_t)active_power_mode,
208 (uint8_t)new_power_mode));
209
210 sys_error_handler();
211 }
212
213 SYS_DEBUGF(SYS_DBG_LEVEL_VERBOSE, ("PMH: PM transaction %u -> %u\r\n", (uint8_t)active_power_mode,
214 (uint8_t)new_power_mode));
215
216 return res;

219sys_error_code_t AppPowerModeHelper_vtblDidEnterPowerMode(IAppPowerModeHelper *_this, EPowerMode power_mode)
220{
221 assert_param(_this != NULL);
222 sys_error_code_t res = SYS_NO_ERROR_CODE;
223 AppPowerModeHelper_t *p_obj = (AppPowerModeHelper_t *)_this;
224
225 p_obj->status.active_power_mode = power_mode;
226
227 switch (power_mode)
228 {
229 case E_POWER_MODE_SLEEP_1:
230
231 /* before put the MCU in STOP check if there are event pending in the system queue*/
232
233 SYS_DEBUGF(SYS_DBG_LEVEL_VERBOSE, ("PMH: try SLEEP_1\r\n"));
234
235 /* disable the IRQ*/
236 __asm volatile("cpsid i");
237
238 /* reset the WWDG*/
239 SysResetAEDCounter();
240
241 if (!SysEventsPending())
242 {
243 HAL_SuspendTick();
244 /* there are no other message waiting so I can put the MCU in stop
245 Enable Power Control clock*/
246 __HAL_RCC_PWR_CLK_ENABLE();
247
248 /* Enter Stop Mode*/
249
250 /* Disable all used wakeup sources: WKUP pin*/
251 HAL_PWR_DisableWakeUpPin(PWR_WAKEUP_PIN2);
252
253 SystemClock_Backup();
254 HAL_PWR_EnterSTOPMode(PWR_LOWPOWERREGULATOR_ON, PWR_STOPENTRY_WFI);
255
256 /* The MCU has exited the STOP mode
257 reset the WWDG*/
258 SysResetAEDCounter();
259
260 /* Configures system clock after wake-up from STOP*/
261 SystemClock_Restore();
262 HAL_ResumeTick();
263 }
264
265 /* enable the IRQ*/
266 __asm volatile("cpsie i");
267 break;
268
269 case E_POWER_MODE_STATE1:
270
271 SYS_DEBUGF(SYS_DBG_LEVEL_VERBOSE, ("PMH: STATE1\r\n"));
272
273#if defined(SYS_DEBUG)
274 SysDebugLogFreeHeapSize();
275#endif
276 break;
277
278
279 case E_POWER_MODE_SENSORS_ACTIVE:
280
281 SYS_DEBUGF(SYS_DBG_LEVEL_VERBOSE, ("PMH: E_POWER_MODE_SENSORS_ACTIVE\r\n"));
282
283#if defined(SYS_DEBUG)
284 SysDebugLogFreeHeapSize();
285#endif
286 break;
287
288 default:
289 sys_error_handler();
290 break;
291 }
292
293 return res;
294}
