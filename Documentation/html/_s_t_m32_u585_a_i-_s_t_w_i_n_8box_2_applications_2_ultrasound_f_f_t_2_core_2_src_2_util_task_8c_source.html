264sys_error_code_t UtilTask_vtblDoEnterPowerMode(AManagedTask *_this, const EPowerMode active_power_mode,
265 const EPowerMode new_power_mode)
266{
267 assert_param(_this != NULL);
268 sys_error_code_t res = SYS_NO_ERROR_CODE;
269 UtilTask_t *p_obj = (UtilTask_t *) _this;
270
271 if (new_power_mode == E_POWER_MODE_STATE1)
272 {
273 /* turn on the USER LED */
274 HAL_GPIO_WritePin(LED1_GPIO_Port, LED1_Pin, GPIO_PIN_SET);
275
276 struct utilMessage_t msg =
277 {
278 .msgId = APP_MESSAGE_ID_UTIL,
279 .nCmdID = UTIL_CMD_ID_START_LP_TIMER
280 };
281
282 if (TX_SUCCESS != tx_queue_front_send(&p_obj->in_queue, &msg, AMT_MS_TO_TICKS(150)))
283 {
284 res = SYS_TASK_QUEUE_FULL_ERROR_CODE;
285 }
286 }
287 else if (new_power_mode == E_POWER_MODE_SENSORS_ACTIVE || (new_power_mode == E_POWER_MODE_STARTING))
288 {
289 if (TX_SUCCESS != tx_timer_deactivate(&p_obj->auto_low_power_timer))
290 {
291 SYS_SET_SERVICE_LEVEL_ERROR_CODE(SYS_UTIL_TASK_LP_TIMER_ERROR_CODE);
292 res = SYS_UTIL_TASK_LP_TIMER_ERROR_CODE;
293 }
294 }
295 else if (new_power_mode == E_POWER_MODE_SLEEP_1)
296 {
297 /* turn off the USER LED */
298 HAL_GPIO_WritePin(LED1_GPIO_Port, LED1_Pin, GPIO_PIN_RESET);
299
300 if (TX_SUCCESS != tx_timer_deactivate(&p_obj->auto_low_power_timer))
301 {
302 SYS_SET_SERVICE_LEVEL_ERROR_CODE(SYS_UTIL_TASK_LP_TIMER_ERROR_CODE);
303 res = SYS_UTIL_TASK_LP_TIMER_ERROR_CODE;
304 }
305 }
306
307 SYS_DEBUGF(SYS_DBG_LEVEL_VERBOSE, ("UTIL: -> %d\r\n", (uint8_t)new_power_mode));
308
309 return res;
310}

499 t_start = HAL_GetTick();
