150int32_t HCI_TL_SPI_Receive(uint8_t *buffer, uint16_t size)
151{
152 uint16_t byte_count;
153 uint8_t len = 0;
154 uint8_t char_ff = 0xff;
155 volatile uint8_t read_char;
156
157 uint8_t header_master[HEADER_SIZE] = {0x0b, 0x00, 0x00, 0x00, 0x00};
158 uint8_t header_slave[HEADER_SIZE];
159
160 HCI_TL_SPI_Disable_IRQ();
161
162 /* CS reset */
163 HAL_GPIO_WritePin(BLE_SPI_CS_GPIO_Port, BLE_SPI_CS_Pin, GPIO_PIN_RESET);
164
165 /* Read the header */
166 HAL_SPI_TransmitReceive(&hspi3, header_master, header_slave, HEADER_SIZE, 1000);
167
168 /* device is ready */
169 byte_count = (header_slave[4] << 8) | header_slave[3];
170
171 if (byte_count > 0)
172 {
173 /* avoid to read more data that size of the buffer */
174 if (byte_count > size)
175 {
176 byte_count = size;
177 }
178
179 for (len = 0; len < byte_count; len++)
180 {
181 HAL_SPI_TransmitReceive(&hspi3, &char_ff, (uint8_t *)&read_char, 1, 1000);
182 buffer[len] = read_char;
183 }
184 }
185
191 uint32_t tickstart = HAL_GetTick();
192 while ((HAL_GetTick() - tickstart) < TIMEOUT_IRQ_HIGH)
193 {
194 if (HAL_GPIO_ReadPin(BLE_INT_GPIO_Port, BLE_INT_Pin) == GPIO_PIN_RESET)
195 {
196 break;
197 }
198 }
199
200 HCI_TL_SPI_Enable_IRQ();
201
202 /* Release CS line */
203 HAL_GPIO_WritePin(BLE_SPI_CS_GPIO_Port, BLE_SPI_CS_Pin, GPIO_PIN_SET);
204
205 return len;
206}

215int32_t HCI_TL_SPI_Send(uint8_t *buffer, uint16_t size)
216{
217 int32_t result;
218 uint16_t rx_bytes;
219
220 uint8_t header_master[HEADER_SIZE] = {0x0a, 0x00, 0x00, 0x00, 0x00};
221 uint8_t header_slave[HEADER_SIZE];
222
223 static uint8_t read_char_buf[MAX_BUFFER_SIZE];
224 uint32_t tickstart = HAL_GetTick();
225
226 HCI_TL_SPI_Disable_IRQ();
227
228 do
229 {
230 uint32_t tickstart_data_available = HAL_GetTick();
231
232 result = 0;
233
234 /* CS reset */
235 HAL_GPIO_WritePin(BLE_SPI_CS_GPIO_Port, BLE_SPI_CS_Pin, GPIO_PIN_RESET);
236
237 /*
238 * Wait until BlueNRG-2 is ready.
239 * When ready it will raise the IRQ pin.
240 */
241 while (!IsDataAvailable())
242 {
243 if ((HAL_GetTick() - tickstart_data_available) > TIMEOUT_DURATION)
244 {
245 result = -3;
246 break;
247 }
248 }
249 if (result == -3)
250 {
251 /* The break causes the exiting from the "while", so the CS line must be released */
252 HAL_GPIO_WritePin(BLE_SPI_CS_GPIO_Port, BLE_SPI_CS_Pin, GPIO_PIN_SET);
253 break;
254 }
255
256
257 /* Read header */
258 HAL_SPI_TransmitReceive(&hspi3, header_master, header_slave, HEADER_SIZE, 1000);
259
260 rx_bytes = (((uint16_t)header_slave[2]) << 8) | ((uint16_t)header_slave[1]);
261
262 if (rx_bytes >= size)
263 {
264 /* Buffer is big enough */
265 HAL_SPI_TransmitReceive(&hspi3, buffer, read_char_buf, size, 1000);
266 }
267 else
268 {
269 /* Buffer is too small */
270 result = -2;
271 }
272
273 /* Release CS line */
274 HAL_GPIO_WritePin(BLE_SPI_CS_GPIO_Port, BLE_SPI_CS_Pin, GPIO_PIN_SET);
275
276 if ((HAL_GetTick() - tickstart) > TIMEOUT_DURATION)
277 {
278 result = -3;
279 break;
280 }
281
282 } while (result < 0);
283
289 tickstart = HAL_GetTick();
290 while ((HAL_GetTick() - tickstart) < TIMEOUT_IRQ_HIGH)
291 {
292 if (HAL_GPIO_ReadPin(BLE_INT_GPIO_Port, BLE_INT_Pin) == GPIO_PIN_RESET)
293 {
294 break;
295 }
296 }
297
298 HCI_TL_SPI_Enable_IRQ();
299
300 return result;
301}
