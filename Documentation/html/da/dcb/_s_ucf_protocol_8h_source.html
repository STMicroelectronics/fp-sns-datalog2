45/*
46 * UCF extended format supports the following commands:
47 * - MEMS_UCF_OP_READ: read the register at the location specified by the
48 * "address" field ("data" field is ignored
49 * - MEMS_UCF_OP_WRITE: write the value specified by the "data" field at the
50 * location specified by the "address" field
51 * - MEMS_UCF_OP_DELAY: wait the number of milliseconds specified by the "data"
52 * field ("address" field is ignored)
53 * - MEMS_UCF_OP_POLL_SET: poll the register at the location specified by the
54 * "address" field until all the bits identified by the mask specified by the
55 * "data" field are set to 1
56 * - MEMS_UCF_OP_POLL_RESET: poll the register at the location specified by the
57 * "address" field until all the bits identified by the mask specified by the
58 * "data" field are reset to 0
59 */
60

62#define MEMS_UCF_OP_WRITE 1
63#define MEMS_UCF_OP_DELAY 2
64#define MEMS_UCF_OP_POLL_SET 3
65#define MEMS_UCF_OP_POLL_RESET 4
