129sys_error_code_t I2CMasterDriver_vtblInit(IDriver *_this, void *p_params)
130{
131 assert_param(_this != NULL);
132 assert_param(p_params != NULL);
133 sys_error_code_t res = SYS_NO_ERROR_CODE;
134 UINT nRes = TX_SUCCESS;
135 I2CMasterDriver_t *p_obj = (I2CMasterDriver_t *) _this;
136 p_obj->mx_handle.p_mx_i2c_cfg = ((I2CMasterDriverParams_t *) p_params)->p_mx_i2c_cfg;
137 I2C_HandleTypeDef *p_i2c = p_obj->mx_handle.p_mx_i2c_cfg->p_i2c_handle;
138
139 p_obj->mx_handle.p_mx_i2c_cfg->p_mx_dma_init_f();
140 p_obj->mx_handle.p_mx_i2c_cfg->p_mx_init_f();
141
142 /* Register I2C DMA complete Callback*/
143 if (HAL_OK != HAL_I2C_RegisterCallback(p_i2c, HAL_I2C_MEM_RX_COMPLETE_CB_ID, I2CMasterDrvMemTxRxCpltCallback))
144 {
145 SYS_SET_LOW_LEVEL_ERROR_CODE(SYS_UNDEFINED_ERROR_CODE);
146 res = SYS_UNDEFINED_ERROR_CODE;
147 }
148 else if (HAL_OK != HAL_I2C_RegisterCallback(p_i2c, HAL_I2C_MEM_TX_COMPLETE_CB_ID, I2CMasterDrvMemTxRxCpltCallback))
149 {
150 SYS_SET_LOW_LEVEL_ERROR_CODE(SYS_UNDEFINED_ERROR_CODE);
151 res = SYS_UNDEFINED_ERROR_CODE;
152 }
153 else if (HAL_OK != HAL_I2C_RegisterCallback(p_i2c, HAL_I2C_ERROR_CB_ID, I2CMasterDrvErrorCallback))
154 {
155 SYS_SET_LOW_LEVEL_ERROR_CODE(SYS_UNDEFINED_ERROR_CODE);
156 res = SYS_UNDEFINED_ERROR_CODE;
157 }
158 else
159 {
160 if (!HWDriverMap_IsInitialized(&sTheClass.ip_drv_map))
161 {
162 (void) HWDriverMap_Init(&sTheClass.ip_drv_map, sTheClass.ip_drv_map_elements, I2CDRV_CFG_HARDWARE_PERIPHERALS_COUNT);
163 }
164
165 /* Add the driver to the map.
166 * Use the peripheral address as unique key for the map. */
167 HWDriverMapElement_t *p_element = NULL;
168 uint32_t key = (uint32_t) p_obj->mx_handle.p_mx_i2c_cfg->p_i2c_handle->Instance;
169 p_element = HWDriverMap_AddElement(&sTheClass.ip_drv_map, key, _this);
170
171 if (p_element == NULL)
172 {
173 SYS_SET_LOW_LEVEL_ERROR_CODE(SYS_INVALID_PARAMETER_ERROR_CODE);
174 res = SYS_INVALID_PARAMETER_ERROR_CODE;
175 }
176 else
177 {
178 nRes = tx_semaphore_create(&p_obj->sync_obj, "I2CDrv", 0);
179 if (nRes != TX_SUCCESS)
180 {
181 SYS_SET_LOW_LEVEL_ERROR_CODE(SYS_OUT_OF_MEMORY_ERROR_CODE);
182 res = SYS_OUT_OF_MEMORY_ERROR_CODE;
183
184 (void) HWDriverMap_RemoveElement(&sTheClass.ip_drv_map, key);
185 }
186 }
187 }
188
189 SYS_DEBUGF(SYS_DBG_LEVEL_VERBOSE, ("I2CMasterDriver: initialization done: %d.\r\n", res));
190
191 return res;
192}
