61BleCharTypeDef *BLE_InitInertialService(uint8_t AccEnable, uint8_t GyroEnable, uint8_t MagEnabled)
62{
63 /* Data structure pointer for BLE service */
64 BleCharTypeDef *BleCharPointer = NULL;
65 InertialCharSize = 2;
66
67 /* Init data structure for identify environmental info services enabled */
68 InertialFeaturesEnabled.AccIsEnable = AccEnable;
69 InertialFeaturesEnabled.GyroIsEnable = GyroEnable;
70 InertialFeaturesEnabled.MagIsEnabled = MagEnabled;
71
72 if ((AccEnable == 1U) ||
73 (GyroEnable == 1U) ||
74 (MagEnabled == 1U))
75 {
76 /* Init data structure pointer for inertial info service */
77 BleCharPointer = &BleCharInertial;
78 memset(BleCharPointer, 0, sizeof(BleCharTypeDef));
79 BleCharPointer->AttrMod_Request_CB = AttrMod_Request_Inertial;
80 COPY_INERTIAL_CHAR_UUID((BleCharPointer->uuid));
81
82 /* Enables BLE accelerometer feature */
83 if (AccEnable == 1U)
84 {
85 BleCharPointer->uuid[14] |= 0x80U;
86 InertialCharSize += 3U * 2U;
87#if (BLE_DEBUG_LEVEL>1)
88 BLE_MANAGER_PRINTF("\t--> Accelerometer feature enabled\r\n");
89#endif
90 }
91
92 /* Enables BLE gyroscope feature */
93 if (GyroEnable == 1U)
94 {
95 BleCharPointer->uuid[14] |= 0x40U;
96 InertialCharSize += 3U * 2U;
97#if (BLE_DEBUG_LEVEL>1)
98 BLE_MANAGER_PRINTF("\t--> Gyroscope feature enabled\r\n");
99#endif
100 }
101
102 /* Enables BLE magnetometer feature */
103 if (MagEnabled == 1U)
104 {
105 BleCharPointer->uuid[14] |= 0x20U;
106 InertialCharSize += 3U * 2U;
107#if (BLE_DEBUG_LEVEL>1)
108 BLE_MANAGER_PRINTF("\t--> Magnetometer feature enabled\r\n");
109#endif
110 }
111
112 BleCharPointer->Char_UUID_Type = UUID_TYPE_128;
113 BleCharPointer->Char_Value_Length = InertialCharSize;
114 BleCharPointer->Char_Properties = CHAR_PROP_NOTIFY;
115 BleCharPointer->Security_Permissions = ATTR_PERMISSION_NONE;
116 BleCharPointer->GATT_Evt_Mask = GATT_NOTIFY_READ_REQ_AND_WAIT_FOR_APPL_RESP;
117 BleCharPointer->Enc_Key_Size = 16;
118 BleCharPointer->Is_Variable = 0;
119
120 BLE_MANAGER_PRINTF("BLE Inertial features ok\r\n");
121 }
122 else
123 {
124 BLE_MANAGER_PRINTF(" ERROR: None inertial features is enabled\r\n");
125 }
126
127 return BleCharPointer;
128}

165tBleStatus BLE_AccGyroMagUpdate(BLE_MANAGER_INERTIAL_Axes_t *Acc,
166 BLE_MANAGER_INERTIAL_Axes_t *Gyro,
167 BLE_MANAGER_INERTIAL_Axes_t *Mag)
168{
169 tBleStatus ret;
170 uint8_t BuffPos;
171
172 uint8_t buff[2 + (3 * 2)/*Acc*/ + (3 * 2)/*Gyro*/ + (3 * 2)/*Mag*/];
173
174 /* Time Stamp */
175 STORE_LE_16(buff, (HAL_GetTick() >> 3));
176 BuffPos = 2;
177
178 if (InertialFeaturesEnabled.AccIsEnable == 1U)
179 {
180 STORE_LE_16(buff + BuffPos, ((uint16_t)Acc->x));
181 BuffPos += 2U;
182 STORE_LE_16(buff + BuffPos, ((uint16_t)Acc->y));
183 BuffPos += 2U;
184 STORE_LE_16(buff + BuffPos, ((uint16_t)Acc->z));
185 BuffPos += 2U;
186 }
187
188 if (InertialFeaturesEnabled.GyroIsEnable == 1U)
189 {
190 Gyro->x /= 100;
191 Gyro->y /= 100;
192 Gyro->z /= 100;
193
194 STORE_LE_16(buff + BuffPos, ((uint16_t)Gyro->x));
195 BuffPos += 2U;
196 STORE_LE_16(buff + BuffPos, ((uint16_t)Gyro->y));
197 BuffPos += 2U;
198 STORE_LE_16(buff + BuffPos, ((uint16_t)Gyro->z));
199 BuffPos += 2U;
200 }
201
202 if (InertialFeaturesEnabled.MagIsEnabled == 1U)
203 {
204 STORE_LE_16(buff + BuffPos, ((uint16_t)Mag->x));
205 BuffPos += 2U;
206 STORE_LE_16(buff + BuffPos, ((uint16_t)Mag->y));
207 BuffPos += 2U;
208 STORE_LE_16(buff + BuffPos, ((uint16_t)Mag->z));
209 BuffPos += 2U;
210 }
211
212 ret = ACI_GATT_UPDATE_CHAR_VALUE(&BleCharInertial, 0, InertialCharSize, buff);
213
214 if (ret != (tBleStatus)BLE_STATUS_SUCCESS)
215 {
216 if (ret != (tBleStatus)BLE_STATUS_INSUFFICIENT_RESOURCES)
217 {
218 if (BLE_StdErr_Service == BLE_SERV_ENABLE)
219 {
220 BytesToWrite = (uint8_t)sprintf((char *)BufferToWrite, "Error Updating Acc/Gyro/Mag Char\n");
221 Stderr_Update(BufferToWrite, BytesToWrite);
222 }
223 else
224 {
225 BLE_MANAGER_PRINTF("Error: Updating Acc/Gyro/Mag Char ret=%x\r\n", ret);
226 }
227 }
228 else
229 {
230 BLE_MANAGER_PRINTF("Error: Updating Acc/Gyro/Mag Char ret=%x\r\n", ret);
231 }
232 }
233
234 return ret;
235}
