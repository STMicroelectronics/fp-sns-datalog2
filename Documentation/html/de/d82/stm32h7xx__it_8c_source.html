349 {
350 if (__HAL_GPIO_EXTI_GET_IT(xMap[i].nPin) && (xMap[i].nPin == exti_pin))

352 /* EXTI line interrupt detected */
353 __HAL_GPIO_EXTI_CLEAR_IT(xMap[i].nPin);
354 xMap[i].pfCallback(xMap[i].nPin);
