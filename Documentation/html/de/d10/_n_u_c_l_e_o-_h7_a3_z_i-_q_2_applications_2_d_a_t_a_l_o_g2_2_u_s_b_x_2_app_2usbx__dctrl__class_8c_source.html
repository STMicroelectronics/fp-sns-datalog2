110int8_t datalog_class_control(void *_this, uint8_t isHostToDevice, uint8_t cmd, uint16_t wValue, uint16_t wIndex,
111 uint8_t *pbuf, uint16_t length)
112{
113 assert_param(_this != NULL);
114
115 static uint32_t USB_packet_size = 0;
116 static uint32_t counter = 0;
117 static char *serialized_cmd = 0;
118 static char *p = 0;
119 usbx_dctrl_class_t *obj = (usbx_dctrl_class_t *) _this;
120
121 if (isHostToDevice)
122 {
123 switch (obj->state)
124 {
125 case USBD_WCID_WAITING_FOR_SIZE :
126
127 if (cmd != CMD_SIZE_SET) /* discard if it isn't a size set command */
128 {
129 return -1;
130 }
131
132 SysFree(serialized_cmd);
133
134 if (length == 2)
135 {
136 USB_packet_size = *(uint16_t *) pbuf;
137 }
138 else
139 {
140 USB_packet_size = *(uint32_t *) pbuf;
141 }
142
143 /* adding one character to include '\0'*/
144 USB_packet_size++;
145
146 serialized_cmd = SysAlloc(USB_packet_size); /* Allocate the buffer to receive next command */
147 if (serialized_cmd == NULL)
148 {
149 /*Mem alloc error*/
150 SYS_SET_SERVICE_LEVEL_ERROR_CODE(SYS_TASK_HEAP_OUT_OF_MEMORY_ERROR_CODE);
151 return -1;
152 }
153 else
154 {
155 /*Mem alloc ok*/
156 }
157
158 p = serialized_cmd;
159 obj->state = USBD_WCID_WAITING_FOR_DATA;
160 counter = USB_packet_size;
161
162 break;
163 case USBD_WCID_WAITING_FOR_DATA :
164 {
165 uint32_t i = 0;
166
167 if (cmd != CMD_DATA_SET)
168 {
169 return -1; /* error */
170 }
171
172 for (i = 0; i < length; i++)
173 {
174 *p++ = pbuf[i];
175 counter--;
176 }
177
178 if (counter-1 == 0) /* The complete message has been received */
179 {
180 *p = '\0';
181 IParseCommand(obj->cmd_parser, serialized_cmd, sObj.comm_interface_id);
182 }
183 break;
184 }
185 }
186 }
187 else /* Device to host */
188 {
189 switch (obj->state)
190 {
191 case USBD_WCID_WAITING_FOR_SIZE_REQUEST : /* Host needs size */
192 {
193 if (cmd != CMD_SIZE_GET)
194 {
195 return -1; /* error*/
196 }
197
198 SysFree(serialized_cmd);
200 char *response_name;
201
202 ISerializeResponse(obj->cmd_parser, &response_name, &serialized_cmd, &USB_packet_size, 0);
203
204 *(uint32_t *) pbuf = USB_packet_size;
205 p = serialized_cmd;
206 obj->state = USBD_WCID_WAITING_FOR_DATA_REQUEST;
207 counter = USB_packet_size;
208
209 break;
210 }
211 case USBD_WCID_WAITING_FOR_DATA_REQUEST :
212 {
213 uint32_t i = 0;
214
215 if (cmd != CMD_DATA_GET)
216 {
217 return -1; /* error*/
218 }
219
220 for (i = 0; i < length; i++)
221 {
222 pbuf[i] = *p++;
223 counter--;
224 }
225
226 if (counter == 0) /* The complete message has been received */
227 {
228 SysFree(serialized_cmd);
229 serialized_cmd = NULL;
230 obj->state = USBD_WCID_WAITING_FOR_SIZE;
231 }
232 break;
233 }
234 }
235 }
236
237 return (0);
238}

242sys_error_code_t usbx_dctrl_vtblStream_init(IStream_t *_this, uint8_t comm_interface_id, void *param)
243{
244 assert_param(_this != NULL);
245 sys_error_code_t res = SYS_NO_ERROR_CODE;
246 usbx_dctrl_class_t *obj = (usbx_dctrl_class_t *) _this;
247
248 /* Device framework FS length*/
249 ULONG device_framework_fs_length;
250 /* Device String framework length*/
251 ULONG string_framework_length;
252 /* Device language id framework length*/
253 ULONG language_id_framework_length;
254 /* Device Framework Full Speed */
255 UCHAR *device_framework_full_speed;
256 /* String Framework*/
257 UCHAR *string_framework;
258 /* Language_Id_Framework*/
259 UCHAR *language_id_framework;
260 /* Status Tx */
261 UINT tx_status = UX_SUCCESS;
262
263 UCHAR *pointer = obj->memory_pointer;
264 obj->is_class_initialized_by_the_host = false;
265
266 obj->comm_interface_id = comm_interface_id;
267
268 if (tx_byte_pool_create(&obj->ux_device_app_byte_pool, "Ux App memory pool", obj->ux_device_byte_pool_buffer,
269 UX_DEVICE_APP_MEM_POOL_SIZE) != TX_SUCCESS)
270 {
271 res = SYS_TASK_HEAP_OUT_OF_MEMORY_ERROR_CODE;
272 SYS_SET_SERVICE_LEVEL_ERROR_CODE(res);
273 return res;
274 }
275
276 /* Allocate USBX_MEMORY_SIZE. */
277 tx_status = tx_byte_allocate(&obj->ux_device_app_byte_pool, (VOID **) &pointer, USBX_MEMORY_SIZE, TX_NO_WAIT);
278
279 /* Check memory allocation */
280 if (UX_SUCCESS != tx_status)
281 {
282 res = SYS_TASK_HEAP_OUT_OF_MEMORY_ERROR_CODE;
283 SYS_SET_SERVICE_LEVEL_ERROR_CODE(res);
284 return res;
285 }
286
287 /* Initialize USBX Memory */
288 ux_system_initialize(pointer, USBX_MEMORY_SIZE, UX_NULL, 0);
289
290 /* Get_Device_Framework_Full_Speed and get the length */
291 device_framework_full_speed = USBD_Get_Device_Framework_Speed(&device_framework_fs_length);
292
293 /* Get_String_Framework and get the length */
294 string_framework = USBD_Get_StringDescription_Framework(&string_framework_length);
295
296 if (string_framework == NULL)
297 {
298 res = SYS_TASK_INVALID_PARAM_ERROR_CODE;
299 SYS_SET_SERVICE_LEVEL_ERROR_CODE(res);
300 return res;
301 }
302
303 /* Get_Language_Id_Framework and get the length */
304 language_id_framework = USBD_Get_Language_Id_Framework(&language_id_framework_length);
305
306#if (USBX_MAJOR_VERSION == 6) && (USBX_MINOR_VERSION >= 2)
307 tx_status = ux_device_stack_initialize(NULL, 0, device_framework_full_speed, device_framework_fs_length, string_framework, string_framework_length,
308 language_id_framework, language_id_framework_length, USBD_ChangeFunction);
309#else
310 tx_status = ux_device_stack_initialize(NULL, 0, device_framework_full_speed, device_framework_fs_length, string_framework, string_framework_length,
311 language_id_framework, language_id_framework_length, UX_NULL);
312#endif
313 /* Check the device stack class status */
314 if (tx_status != UX_SUCCESS)
315 {
316 res = SYS_INIT_TASK_FAILURE_ERROR_CODE;
317 SYS_SET_SERVICE_LEVEL_ERROR_CODE(res);
318 return res;
319 }
320
321 IDrvInit(obj->m_pxDriver, (void *) obj->mx_drv_cfg);
322 PCDDrvSetFIFO((PCDDriver_t *) obj->m_pxDriver, PCD_FIFO_SIZE, PCD_FIFO_RX_EP, PCD_FIFO_CTRL_EP, SS_N_IN_ENDPOINTS);
323 PCDDrvSetExtDCD((PCDDriver_t *) obj->m_pxDriver, _ux_dcd_stm32_initialize);
324
325 UX_SLAVE_CLASS_SENSOR_STREAMING_PARAMETER *ss_class_parameter = _ux_utility_memory_allocate(UX_NO_ALIGN, UX_REGULAR_MEMORY,
326 sizeof(UX_SLAVE_CLASS_SENSOR_STREAMING_PARAMETER));
327
329 ss_class_parameter->ux_slave_class_sensor_streaming_instance_activate = datalog_class_active;
330 ss_class_parameter->ux_slave_class_sensor_streaming_instance_deactivate = datalog_class_deactive;
331 ss_class_parameter->ux_slave_class_sensor_streaming_instance_control = datalog_class_control;
332 ss_class_parameter->app_obj_ctrl = (void *) obj;
333
334 /* registers a slave class to the slave stack. */
335 tx_status = ux_device_class_sensor_streaming_stack_class_register(ss_class_parameter);
336
337 return res;
338}
