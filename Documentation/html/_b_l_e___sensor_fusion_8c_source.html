48{
49 /* Data structure pointer for BLE service */
50 BleCharTypeDef *BleCharPointer;
51
52 uint8_t DataLenght = 2U + (6U * NumberQuaternionsToSend);
53
54 if ((NumberQuaternionsToSend > 0U) && (NumberQuaternionsToSend <= 3U))

56 /* Init data structure pointer for Sensor Fusion info service */
57 BleCharPointer = &BleCharSensorFusion;

60 COPY_SENSOR_FUSION_CHAR_UUID((BleCharPointer->uuid));

73 BLE_MANAGER_PRINTF("BLE Sensor Fusion features init error: Number Quaternions To Send not correct\r\n");

98tBleStatus BLE_SensorFusionUpdate(BLE_MOTION_SENSOR_Axes_t *data, uint8_t NumberQuaternionsToSend)
99{
100 tBleStatus ret;
101 uint8_t dimByte;
102
103 uint8_t buff[2U + (6U * 3U)];
104
105 STORE_LE_16(buff, (HAL_GetTick() >> 3));
106
107 switch (NumberQuaternionsToSend)
108 {
109 case 1:
110 STORE_LE_16(buff + 2, data[0].x);
111 STORE_LE_16(buff + 4, data[0].y);
112 STORE_LE_16(buff + 6, data[0].z);
113 break;
114 case 2:
115 STORE_LE_16(buff + 2, data[0].x);
116 STORE_LE_16(buff + 4, data[0].y);

118 STORE_LE_16(buff + 8, data[1].x);
119 STORE_LE_16(buff + 10, data[1].y);
120 STORE_LE_16(buff + 12, data[1].z);
121 break;
122 case 3:
123 STORE_LE_16(buff + 2, data[0].x);
124 STORE_LE_16(buff + 4, data[0].y);
125 STORE_LE_16(buff + 6, data[0].z);
126 STORE_LE_16(buff + 8, data[1].x);
127 STORE_LE_16(buff + 10, data[1].y);
128 STORE_LE_16(buff + 12, data[1].z);
129 STORE_LE_16(buff + 14, data[2].x);
130 STORE_LE_16(buff + 16, data[2].y);
131 STORE_LE_16(buff + 18, data[2].z);
132 break;
133 }
134
135 dimByte = 2U + (6U * NumberQuaternionsToSend);
136 ret = ACI_GATT_UPDATE_CHAR_VALUE(&BleCharSensorFusion, 0, dimByte, buff);

139 {
140 if (BLE_StdErr_Service == BLE_SERV_ENABLE)
141 {
142 BytesToWrite = (uint8_t)sprintf((char *)BufferToWrite, "Error Updating Sensor Fusion Char\n");
143 Stderr_Update(BufferToWrite, BytesToWrite);

145 else

147 BLE_MANAGER_PRINTF("Error Updating Sensor Fusion Char\r\n");
148 }
149 }
150 return ret;
151}
