35sys_error_code_t UCFP_LoadCompressedUcf(SUcfProtocol_t *_this, const char *p_ucf, uint32_t size)
36{
37 assert_param(_this != NULL);
38 assert_param(p_ucf != NULL);
39 assert_param(size > 0U);
40 sys_error_code_t res = SYS_NO_ERROR_CODE;
41
42 uint32_t ucf_lines = size / COMPRESSED_UCF_LINE_WIDTH;
43 uint32_t i;
44 char ucf_reg[3];
45 char ucf_data[4];
46 uint8_t reg;
47 uint8_t data;
48
49 /* Add string termination */
50 ucf_reg[2] = '\0';
51 ucf_data[2] = '\0';
52 ucf_data[3] = '\0';
53
54 for (i = 0; i < ucf_lines; i++)
55 {
56 if (*p_ucf == 'W' || *p_ucf == 'w')
57 {
58 /* Wait command */
59 p_ucf++;
60 ucf_data[0] = *(p_ucf++);
61 ucf_data[1] = *(p_ucf++);
62 ucf_data[2] = *(p_ucf++);
63
64 uint16_t ms = (uint16_t) strtol(ucf_data, NULL, 16);
65
66 tx_thread_sleep(ms);
67
68 ucf_data[2] = '\0'; /* put back termination character for write commands */
69 }
70 else
71 {
72 /* Write command */
73 ucf_reg[0] = *(p_ucf++);
74 ucf_reg[1] = *(p_ucf++);
75 reg = (uint8_t) strtol(ucf_reg, NULL, 16);
76
77 ucf_data[0] = *(p_ucf++);
78 ucf_data[1] = *(p_ucf++);
79 data = (uint8_t) strtol(ucf_data, NULL, 16);
80
81 res = ISensorWriteReg(_this->sensor_ll, reg, &data, 1);
82 if (SYS_IS_ERROR_CODE(res))
83 {
84 break;
85 }
86 }
87 }
88
89 /* Sync sensor internal model with the sensor registers written by the ucf */
90 if (ISensorSyncModel(_this->sensor_ll) != SYS_NO_ERROR_CODE)
91 {
92 res = SYS_BASE_ERROR_CODE;
93 }
94

96}

108sys_error_code_t UCFP_LoadUcfHeader(SUcfProtocol_t *_this, const ucf_line_ext_t *p_ucf, uint32_t size)
109{
110 assert_param(_this != NULL);
111 assert_param(p_ucf != NULL);
112 assert_param(size > 0U);
113 sys_error_code_t res = SYS_NO_ERROR_CODE;
114
115 uint32_t ucf_lines = size / sizeof(ucf_line_ext_t);
116 uint32_t i;
117
118 for (i = 0; i < ucf_lines; i++)
119 {
120 switch (p_ucf[i].op)
121 {
122 case MEMS_UCF_OP_WRITE:
123 res = ISensorWriteReg(_this->sensor_ll, p_ucf[i].address, &p_ucf[i].data, 1);
124 if (SYS_IS_ERROR_CODE(res))
125 {
126 break;
127 }
128 break;
129 case MEMS_UCF_OP_DELAY:
130 tx_thread_sleep(p_ucf[i].data);
131 break;
132 case MEMS_UCF_OP_READ:
133 case MEMS_UCF_OP_POLL_SET:
134 case MEMS_UCF_OP_POLL_RESET:
135 res = SYS_NOT_IMPLEMENTED_ERROR_CODE;
136 break;
137 default:
138 res = SYS_INVALID_PARAMETER_ERROR_CODE;
139 break;
140 }
141 }
142
143 /* Sync sensor internal model with the sensor registers written by the ucf */
144 if (ISensorSyncModel(_this->sensor_ll) != SYS_NO_ERROR_CODE)
145 {
146 res = SYS_BASE_ERROR_CODE;
147 }
148
149 return res;
150}

161 p_ch = strtok((char *) p_ucf, " -,_\r\n");

166 res = SYS_OUT_OF_MEMORY_ERROR_CODE;

168 }

171 p_ch = strtok(NULL, " -,_\r\n");
172 p_compressed_ucf[i++] = p_ch[0];
173 p_compressed_ucf[i++] = p_ch[1];
174 p_ch = strtok(NULL, " -,_\r\n");
175 p_compressed_ucf[i++] = p_ch[0];
176 p_compressed_ucf[i++] = p_ch[1];
177 }
178 else if (strncmp(p_ch, "WAIT", 2) == 0)
179 {
180 p_ch = strtok(NULL, " -,_\r\n");
181 uint16_t number = atol(p_ch);
182 char tmp[10];
183 sprintf(tmp, "W%03d", number); /* Example --> W005 */
184 p_compressed_ucf[i++] = tmp[0];
185 p_compressed_ucf[i++] = tmp[1];
186 p_compressed_ucf[i++] = tmp[2];
187 p_compressed_ucf[i++] = tmp[3];
188 }
189 p_ch = strtok(NULL, " -,_\r\n");
190 }
191 p_compressed_ucf[i] = 0;
192 *compressed_ucf_size_actual = i;
193
194 return res;
195}

197sys_error_code_t UCFP_GetUcf(const char *p_compressed_ucf, uint32_t compressed_ucf_size, char *p_ucf, uint32_t ucf_size,
198 uint32_t *ucf_size_actual)
199{
200 uint32_t i;
201 uint32_t out_size = 0;
202 sys_error_code_t res = SYS_NO_ERROR_CODE;
203
204 for (i = 0; i < compressed_ucf_size / 4; i++)
205 {
206 if (out_size >= ucf_size)
207 {
208 res = SYS_OUT_OF_MEMORY_ERROR_CODE;
209 return res;
210 }
211 if (p_compressed_ucf[4U * i] != 'W')
212 {
213 *p_ucf++ = 'A';
214 *p_ucf++ = 'c';
215 *p_ucf++ = ' ';
216 *p_ucf++ = p_compressed_ucf[4U * i];
217 *p_ucf++ = p_compressed_ucf[4U * i + 1U];
218 *p_ucf++ = ' ';
219 *p_ucf++ = p_compressed_ucf[4U * i + 2U];
220 *p_ucf++ = p_compressed_ucf[4U * i + 3U];
221 *p_ucf++ = '\n';
222 out_size += 9;
223 }
224 else
225 {
226 *p_ucf++ = 'W';
227 *p_ucf++ = 'A';
228 *p_ucf++ = 'I';
229 *p_ucf++ = 'T';
230 *p_ucf++ = ' ';
231 /* copy the first 2 numbers only if they are != '0' */
232 if (p_compressed_ucf[4 * i + 1] != '0')
233 {
234 *p_ucf++ = p_compressed_ucf[4 * i + 1];
235 out_size++;
236 }
237 if (p_compressed_ucf[4 * i + 2] != '0')
238 {
239 *p_ucf++ = p_compressed_ucf[4 * i + 2];
240 out_size++;
241 }
242 *p_ucf++ = p_compressed_ucf[4 * i + 3];
243 *p_ucf++ = '\n';
244 out_size += 7;
245 }
246 }
247 *ucf_size_actual = out_size;
248
249 return res;
250}
