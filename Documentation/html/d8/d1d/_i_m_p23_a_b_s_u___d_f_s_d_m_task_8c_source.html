348sys_error_code_t IMP23ABSUTask_vtblHardwareInit(AManagedTask *_this, void *pParams)
349{
350 assert_param(_this != NULL);
351 sys_error_code_t res = SYS_NO_ERROR_CODE;
352 IMP23ABSUTask *p_obj = (IMP23ABSUTask *) _this;
353
354 p_obj->p_driver = AnalogMicDriverAlloc();
355 if (p_obj == NULL)
356 {
357 res = SYS_OUT_OF_MEMORY_ERROR_CODE;
358 SYS_SET_LOW_LEVEL_ERROR_CODE(SYS_OUT_OF_MEMORY_ERROR_CODE);
359 }
360 else
361 {
362 AnalogMicDriverParams_t cfg_params =
363 {
364 .p_mx_adc_cfg = (void *) p_obj->p_mx_adc_cfg,
365 .p_mx_dfsdm_cfg = (void *) p_obj->p_mx_dfsdm_cfg
366 };
367
368 res = IDrvInit(p_obj->p_driver, &cfg_params);
369 if (!SYS_IS_ERROR_CODE(res))
370 {
371 DFSDMDriverFilterRegisterCallback((DFSDMDriver_t *) p_obj->p_driver, HAL_DFSDM_FILTER_REGCONV_HALFCOMPLETE_CB_ID,
372 DFSDM_Filter_1_HalfComplete_Callback);
373 DFSDMDriverFilterRegisterCallback((DFSDMDriver_t *) p_obj->p_driver, HAL_DFSDM_FILTER_REGCONV_COMPLETE_CB_ID,
374 DFSDM_Filter_1_Complete_Callback);
375 }
376
377 if (!MTMap_IsInitialized(&sTheClass.task_map))
378 {
379 (void) MTMap_Init(&sTheClass.task_map, sTheClass.task_map_elements, IMP23ABSU_TASK_CFG_MAX_INSTANCES_COUNT);
380 }
381
382 /* Add the managed task to the map.*/
383 /* Use the PIN as unique key for the map. */
384 MTMapElement_t *p_element = NULL;
385// uint32_t key = (uint32_t) p_obj->pIRQConfig->pin;
386 uint32_t key = (uint32_t)((DFSDMDriver_t *) p_obj->p_driver)->mx_handle.p_mx_dfsdm_cfg->p_dfsdm_filter;
387 p_element = MTMap_AddElement(&sTheClass.task_map, key, _this);
388 if (p_element == NULL)
389 {
390 SYS_SET_LOW_LEVEL_ERROR_CODE(SYS_INVALID_PARAMETER_ERROR_CODE);
391 res = SYS_INVALID_PARAMETER_ERROR_CODE;
392 return res;
393 }
394 }
395
396 return res;
397}

399sys_error_code_t IMP23ABSUTask_vtblOnCreateTask(AManagedTask *_this, tx_entry_function_t *pTaskCode, CHAR **pName,
400 VOID **pvStackStart,
401 ULONG *pStackDepth, UINT *pPriority, UINT *pPreemptThreshold, ULONG *pTimeSlice, ULONG *pAutoStart,
402 ULONG *pParams)
403{
404 assert_param(_this != NULL);
405 sys_error_code_t res = SYS_NO_ERROR_CODE;
406 IMP23ABSUTask *p_obj = (IMP23ABSUTask *) _this;
407
408 /* Create task specific sw resources. */
409
410 uint32_t item_size = (uint32_t) IMP23ABSU_TASK_CFG_IN_QUEUE_ITEM_SIZE;
411 VOID *p_queue_items_buff = SysAlloc(IMP23ABSU_TASK_CFG_IN_QUEUE_LENGTH * item_size);
412 if (p_queue_items_buff == NULL)
413 {
414 res = SYS_TASK_HEAP_OUT_OF_MEMORY_ERROR_CODE;
415 SYS_SET_SERVICE_LEVEL_ERROR_CODE(res);
416 return res;
417 }
418
419 if (TX_SUCCESS != tx_queue_create(&p_obj->in_queue, "IMP23ABSU_Q", item_size / 4u, p_queue_items_buff,
420 IMP23ABSU_TASK_CFG_IN_QUEUE_LENGTH * item_size))
421 {
422 res = SYS_TASK_HEAP_OUT_OF_MEMORY_ERROR_CODE;
423 SYS_SET_SERVICE_LEVEL_ERROR_CODE(res);
424 return res;
425 }
426
427 /* Initialize the EventSrc interface */
428 p_obj->p_event_src = DataEventSrcAlloc();
429 if (p_obj->p_event_src == NULL)
430 {
431 SYS_SET_SERVICE_LEVEL_ERROR_CODE(SYS_OUT_OF_MEMORY_ERROR_CODE);
432 res = SYS_OUT_OF_MEMORY_ERROR_CODE;
433 return res;
434 }
435 IEventSrcInit(p_obj->p_event_src);
436
437 memset(p_obj->p_dma_data_buff, 0, sizeof(p_obj->p_dma_data_buff));
438 memset(p_obj->p_sensor_data_buff, 0, sizeof(p_obj->p_sensor_data_buff));
439 p_obj->mic_id = 0;
440 p_obj->prev_timestamp = 0.0f;
441 p_obj->half = 0;
442 p_obj->old_in = 0;
443 p_obj->old_out = 0;
444 _this->m_pfPMState2FuncMap = sTheClass.p_pm_state2func_map;
445
446
447 *pTaskCode = AMTExRun;
448 *pName = "IMP23ABSU";
449 *pvStackStart = NULL; // allocate the task stack in the system memory pool.
450 *pStackDepth = IMP23ABSU_TASK_CFG_STACK_DEPTH;
451 *pParams = (ULONG) _this;
452 *pPriority = IMP23ABSU_TASK_CFG_PRIORITY;
453 *pPreemptThreshold = IMP23ABSU_TASK_CFG_PRIORITY;
454 *pTimeSlice = TX_NO_TIME_SLICE;
455 *pAutoStart = TX_AUTO_START;
456
457 res = IMP23ABSUTaskSensorInitTaskParams(p_obj);
458 if (SYS_IS_ERROR_CODE(res))
459 {
460 res = SYS_TASK_HEAP_OUT_OF_MEMORY_ERROR_CODE;
461 SYS_SET_SERVICE_LEVEL_ERROR_CODE(res);
462 return res;
463 }
464
465 res = IMP23ABSUTaskSensorRegister(p_obj);
466 if (SYS_IS_ERROR_CODE(res))
467 {
468 SYS_DEBUGF(SYS_DBG_LEVEL_VERBOSE, ("IMP23ABSU: unable to register with DB\r\n"));
469 sys_error_handler();
470 }
471
472 return res;
473}

475sys_error_code_t IMP23ABSUTask_vtblDoEnterPowerMode(AManagedTask *_this, const EPowerMode ActivePowerMode,
476 const EPowerMode NewPowerMode)
477{
478 assert_param(_this != NULL);
479 sys_error_code_t res = SYS_NO_ERROR_CODE;
480 IMP23ABSUTask *p_obj = (IMP23ABSUTask *) _this;
481
482 if (IMP23ABSUTaskSensorIsActive(p_obj))
483 {
484 IDrvDoEnterPowerMode(p_obj->p_driver, ActivePowerMode, NewPowerMode);
485 }
486
487 if (NewPowerMode == E_POWER_MODE_SENSORS_ACTIVE)
488 {
489 if (IMP23ABSUTaskSensorIsActive(p_obj))
490 {
491 SMMessage xReport =
492 {
493 .sensorMessage.messageId = SM_MESSAGE_ID_SENSOR_CMD,
494 .sensorMessage.nCmdID = SENSOR_CMD_ID_INIT
495 };
496
497 if (tx_queue_send(&p_obj->in_queue, &xReport, AMT_MS_TO_TICKS(50)) != TX_SUCCESS)
498 {
499 res = SYS_SENSOR_TASK_MSG_LOST_ERROR_CODE;
500 SYS_SET_SERVICE_LEVEL_ERROR_CODE(SYS_SENSOR_TASK_MSG_LOST_ERROR_CODE);
501 }
502
503 // reset the variables for the actual ODR computation.
504 p_obj->prev_timestamp = 0.0f;
505 }
506
507 SYS_DEBUGF(SYS_DBG_LEVEL_VERBOSE, ("IMP23ABSU: -> SENSORS_ACTIVE\r\n"));
508 }
509 else if (NewPowerMode == E_POWER_MODE_STATE1)
510 {
511 if (ActivePowerMode == E_POWER_MODE_SENSORS_ACTIVE)
512 {
513 /* Empty the task queue and disable INT or timer */
514 tx_queue_flush(&p_obj->in_queue);
515 }
516
517 SYS_DEBUGF(SYS_DBG_LEVEL_VERBOSE, ("IMP23ABSU: -> STATE1\r\n"));
518 }
519 else if (NewPowerMode == E_POWER_MODE_SLEEP_1)
520 {
521 /* the MCU is going in stop so I put the sensor in low power (this code is executed from the INIT task).*/
522
523 SYS_DEBUGF(SYS_DBG_LEVEL_VERBOSE, ("IMP23ABSU: -> SLEEP_1\r\n"));
524 }
525 return res;
526}

555sys_error_code_t IMP23ABSUTask_vtblForceExecuteStep(AManagedTaskEx *_this, EPowerMode ActivePowerMode)
556{
557 assert_param(_this != NULL);
558 sys_error_code_t res = SYS_NO_ERROR_CODE;
559 IMP23ABSUTask *p_obj = (IMP23ABSUTask *) _this;
560
561 SMMessage xReport =
562 {
563 .internalMessageFE.messageId = SM_MESSAGE_ID_FORCE_STEP,
564 .internalMessageFE.nData = 0
565 };
566
567 if ((ActivePowerMode == E_POWER_MODE_STATE1) || (ActivePowerMode == E_POWER_MODE_SENSORS_ACTIVE))
568 {
569 if (AMTExIsTaskInactive(_this))
570 {
571 res = IMP23ABSUTaskPostReportToFront(p_obj, (SMMessage *) &xReport);
572 }
573 else
574 {
575 // do nothing and wait for the step to complete.
576 // _this->m_xStatus.nDelayPowerModeSwitch = 0;
577 }
578 }
579 else
580 {
581 UINT state;
582 if (TX_SUCCESS == tx_thread_info_get(&_this->m_xTaskHandle, TX_NULL, &state, TX_NULL, TX_NULL, TX_NULL, TX_NULL,
583 TX_NULL,
584 TX_NULL))
585 {
586 if (state == TX_SUSPENDED)
587 {
588 tx_thread_resume(&_this->m_xTaskHandle);
589 }
590 }
591 }
592 return res;
