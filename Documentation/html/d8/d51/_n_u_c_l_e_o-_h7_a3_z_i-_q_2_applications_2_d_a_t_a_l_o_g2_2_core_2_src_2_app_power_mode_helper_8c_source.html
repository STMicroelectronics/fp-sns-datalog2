112EPowerMode AppPowerModeHelper_vtblComputeNewPowerMode(IAppPowerModeHelper *_this, const SysEvent event)
113{
114 assert_param(_this != NULL);
115 AppPowerModeHelper_t *p_obj = (AppPowerModeHelper_t *)_this;
116
117 EPowerMode power_mode = p_obj->status.active_power_mode;
118
119 switch (event.xEvent.nSource)
120 {
121 case SYS_PM_EVT_SRC_DATALOG:
122 if (power_mode == E_POWER_MODE_SLEEP_1)
123 {
124 power_mode = E_POWER_MODE_STATE1;
125 }
126 else if (power_mode == E_POWER_MODE_STATE1)
127 {
128 power_mode = E_POWER_MODE_SENSORS_ACTIVE;
129 }
130 else if (power_mode == E_POWER_MODE_SENSORS_ACTIVE)
131 {
132 power_mode = E_POWER_MODE_STATE1;
133 }
134 break;
135
136 case SYS_PM_EVT_SRC_PB:
137 if (power_mode == E_POWER_MODE_STATE1)
138 {
139 power_mode = E_POWER_MODE_SENSORS_ACTIVE;
140 }
141 else if (power_mode == E_POWER_MODE_SENSORS_ACTIVE)
142 {
143 power_mode = E_POWER_MODE_STATE1;
144 }
145 else if (power_mode == E_POWER_MODE_SLEEP_1)
146 {
147 power_mode = E_POWER_MODE_STATE1;
148 }
149 break;
150
151 case SYS_PM_EVT_SRC_LP_TIMER:
152 if (power_mode == E_POWER_MODE_STATE1)
153 {
154 power_mode = E_POWER_MODE_SLEEP_1;
155 }
156 break;
157
158 default:

160 SYS_DEBUGF(SYS_DBG_LEVEL_VERBOSE, ("PMH: wrong SysEvent.\r\n"));
161
162 sys_error_handler();

164 }
165
166#ifdef SYS_DEBUG
167 SYS_DEBUGF(SYS_DBG_LEVEL_VERBOSE, ("PMH: new PM:%u-%u.\r\n", p_obj->status.active_power_mode, power_mode));

173boolean_t AppPowerModeHelper_vtblCheckPowerModeTransaction(IAppPowerModeHelper *_this,
174 const EPowerMode active_power_mode, const EPowerMode new_power_mode)
175{
176 UNUSED(_this);
177 boolean_t res = FALSE;
178
179 switch (active_power_mode)
180 {
181 case E_POWER_MODE_STATE1:
182 if ((new_power_mode == E_POWER_MODE_SENSORS_ACTIVE) || (new_power_mode == E_POWER_MODE_SLEEP_1))
183 {
184 res = TRUE;
185 }
186 break;
187 case E_POWER_MODE_SENSORS_ACTIVE:
188 if (new_power_mode == E_POWER_MODE_STATE1)
189 {
190 res = TRUE;
191 }
192 break;
193 case E_POWER_MODE_SLEEP_1:
194 if (new_power_mode == E_POWER_MODE_STATE1)
195 {
196 res = TRUE;
197 }
198 break;
199 default:
200 res = FALSE;
201 }
202
203 if (res == FALSE)
204 {
205
206 SYS_DEBUGF(SYS_DBG_LEVEL_VERBOSE, ("PMH: ERR PM transaction %u -> %u\r\n", (uint8_t)active_power_mode,
207 (uint8_t)new_power_mode));
208
209 sys_error_handler();
210 }
211
212 SYS_DEBUGF(SYS_DBG_LEVEL_VERBOSE, ("PMH: PM transaction %u -> %u\r\n", (uint8_t)active_power_mode,
213 (uint8_t)new_power_mode));
214
215 return res;

218sys_error_code_t AppPowerModeHelper_vtblDidEnterPowerMode(IAppPowerModeHelper *_this, EPowerMode power_mode)
219{
220 assert_param(_this != NULL);
221 sys_error_code_t res = SYS_NO_ERROR_CODE;
222 AppPowerModeHelper_t *p_obj = (AppPowerModeHelper_t *)_this;
223
224 p_obj->status.active_power_mode = power_mode;
225
226 switch (power_mode)
227 {
228 case E_POWER_MODE_SLEEP_1:
229
230 /* before put the MCU in STOP check if there are event pending in the system queue*/
231
232 SYS_DEBUGF(SYS_DBG_LEVEL_VERBOSE, ("PMH: try SLEEP_1\r\n"));
233
234 /* disable the IRQ*/
235 __asm volatile("cpsid i");
236
237 /* reset the WWDG*/
238 SysResetAEDCounter();
239
240 if (!SysEventsPending())
241 {
242 HAL_SuspendTick();
243 /* there are no other message waiting so I can put the MCU in stop
244 Enable Power Control clock*/
245// __HAL_RCC_PWR_CLK_ENABLE(); //TODO: STF - check
246
247 /* Enter Stop Mode*/
248
249 /* Disable all used wakeup sources: WKUP pin*/
250 HAL_PWR_DisableWakeUpPin(PWR_WAKEUP_PIN2);
251
252 SystemClock_Backup();
253 HAL_PWR_EnterSTOPMode(PWR_LOWPOWERREGULATOR_ON, PWR_STOPENTRY_WFI);
254
255 /* The MCU has exited the STOP mode
256 reset the WWDG*/
257 SysResetAEDCounter();
258
259 /* Configures system clock after wake-up from STOP*/
260 SystemClock_Restore();
261 HAL_ResumeTick();
262 }
263
264 /* enable the IRQ*/
265 __asm volatile("cpsie i");
266 break;
267
268 case E_POWER_MODE_STATE1:
269
270 SYS_DEBUGF(SYS_DBG_LEVEL_VERBOSE, ("PMH: STATE1\r\n"));
271
272#if defined(SYS_DEBUG)
273 SysDebugLogFreeHeapSize();
274#endif
275 break;
276
277
278 case E_POWER_MODE_SENSORS_ACTIVE:
279
280 SYS_DEBUGF(SYS_DBG_LEVEL_VERBOSE, ("PMH: E_POWER_MODE_SENSORS_ACTIVE\r\n"));
281
282#if defined(SYS_DEBUG)
283 SysDebugLogFreeHeapSize();
284#endif
285 break;
286
287 default:
288 sys_error_handler();
289 break;
290 }
291
292 return res;
293}
