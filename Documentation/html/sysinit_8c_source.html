197sys_error_code_t SysInit(boolean_t bEnableBootIF)
198{
199 /* First step: initialize the minimum set of resources before passing the control
200 to the RTOS.*/
201
202 sys_error_code_t xRes = SYS_NO_ERROR_CODE;
203
204 /* Reset of all peripherals, Initializes the Flash interface and the Systick.*/
205 if (HAL_OK != HAL_Init())
206 {
207 sys_error_handler();
208 }
209
210 /* Configure the system clock.*/
211 SystemClock_Config();
212 SystemPower_Config();
213
214#if( configAPPLICATION_ALLOCATED_HEAP == 1 )
215 // initialize the FreeRTOS heap.
216 memset(ucHeap, 0, configTOTAL_HEAP_SIZE);
217#endif
218
219#if INIT_TASK_CFG_ENABLE_BOOT_IF == 1
220
221 if (bEnableBootIF)
222 {
223 s_xTheSystem.m_pxAppBootIF = SysGetBootIF();
224 if (s_xTheSystem.m_pxAppBootIF != NULL)
225 {
226 /* initialize the BootIF*/
227 xRes = IBootInit(s_xTheSystem.m_pxAppBootIF);
228 // check the trigger condition
229 if (!IBootCheckDFUTrigger(s_xTheSystem.m_pxAppBootIF))
230 {
231 /* prepare to jump to the main application*/
232 uint32_t nAppAddress = IBootGetAppAdderss(s_xTheSystem.m_pxAppBootIF);
233 xRes = IBootOnJampToApp(s_xTheSystem.m_pxAppBootIF, nAppAddress);
234 if (!SYS_IS_ERROR_CODE(xRes) &&
235 (((*(__IO uint32_t *)nAppAddress) & 0x2FFE0000) == 0x20000000))
236 {
237 typedef void (*pFunction)(void);
238 volatile uint32_t JumpAddress = *(__IO uint32_t *)(nAppAddress + 4);
239 volatile pFunction JumpToApplication = (pFunction) JumpAddress;
240 /* initialize user application's Stack Pointer */
241 __set_MSP(*(__IO uint32_t *) nAppAddress);
242 /* jump to the user application*/
243 JumpToApplication();
244 }
245 }
246 }
247 }
248
249#endif /* INIT_TASK_CFG_ENABLE_BOOT_IF */
250
251#ifdef SYS_DEBUG
252 if (SysDebugInit() != 0)
253 {
254 sys_error_handler();
255 }
256#ifdef DEBUG
257 HAL_DBGMCU_EnableDBGStopMode();
258 SYS_DBGMCU_FREEZE_WWDG();
259#endif /* DEBUG */
260#endif /* SYS_DEBUG */
261
262 /* Clear the global error.*/
263 SYS_CLEAR_ERROR();
264
265 /* Create the INIT task to complete the system initialization
266 after RTOS is started.*/
267
268#ifdef SYS_TP_RTOS_FREERTOS
269 if (xTaskCreate(InitTaskRun, "INIT", INIT_TASK_CFG_STACK_SIZE, NULL, INIT_TASK_CFG_PRIORITY,
270 &s_xTheSystem.m_xInitTask) != pdPASS)
271 {
272 xRes = SYS_OUT_OF_MEMORY_ERROR_CODE;
273 SYS_SET_SERVICE_LEVEL_ERROR_CODE(xRes);
274 }
275
276 // ThreadX use a different approach. After the scheduler is started, it will call the following function
277 // that the application code must overwrite:
278 // void tx_application_define(void *first_unused_memory)
279#endif
280
281 return xRes;
282}

284sys_error_code_t SysPostEvent(SysEvent xEvent)
285{
286 sys_error_code_t xRes = SYS_NO_ERROR_CODE;
287 UINT xResult;
288
289 if (SYS_IS_ERROR_EVENT(xEvent))
290 {
291 /* notify the error delegate to allow a first response to critical errors.*/
292 xRes = IAEDOnNewErrEvent(s_xTheSystem.m_pxAppErrorDelegate, xEvent);
293 }
294
295 ULONG wait_option = TX_NO_WAIT;
296
297 if (!SYS_IS_CALLED_FROM_ISR())
298 {
299 /* check if we are in the system timer thread*/
300 TX_THREAD *p_current_thread = tx_thread_identify();
301 if (p_current_thread != &_tx_timer_thread)
302 {
303 wait_option = SYS_MS_TO_TICKS(INIT_TASK_CFG_POST_EVENT_TIMEOUT_MS);
304 }
305 }
306
307 xResult = tx_queue_send(&s_xTheSystem.m_xSysQueue, &xEvent, wait_option);
308
309 if (xResult != TX_SUCCESS)
310 {
311 SYS_SET_SERVICE_LEVEL_ERROR_CODE(SYS_INIT_TASK_POWER_MODE_NOT_ENABLE_ERROR_CODE);
312 xRes = SYS_INIT_TASK_POWER_MODE_NOT_ENABLE_ERROR_CODE;
313 }
314
315 return xRes;
316}

425static void InitTaskRun(ULONG thread_input)
426{
427 sys_error_code_t xRes = SYS_NO_ERROR_CODE;
428 UINT nRtosRes = TX_SUCCESS;
429 UNUSED(thread_input);
430
431 SYS_DEBUGF(SYS_DBG_LEVEL_VERBOSE, ("System Initialization\r\n"));
432
433#if defined(DEBUG) && !defined(SYS_TP_MCU_STM32U5)
434 tx_thread_stack_error_notify(SysThreadxStackErrorHandler);
435#endif
436
437
438 /* vTaskSuspendAll();
439 to suspend all tasks, they are created with auto_start set to 0.
440 At the end of the system initialization all tasks with auto_start set to 1 are resumed. */
441
442 // allocate the system memory pool
443 CHAR* pool_name = "SYS_MEM_POOL";
444 if (TX_SUCCESS != tx_byte_pool_create(&s_xTheSystem.m_xSysMemPool, pool_name, s_xTheSystem.m_pnHeap,
445 INIT_TASK_CFG_HEAP_SIZE))
446 {
447 // if the memory pool allocation fails then the execution is blocked
448 sys_error_handler();
449 }
450
451 CHAR *pcMemory = NULL;
452 // Create the queue for the system messages.
453 if (TX_SUCCESS != tx_byte_allocate(&s_xTheSystem.m_xSysMemPool, (VOID **)&pcMemory,
454 INIT_TASK_CFG_QUEUE_ITEM_SIZE * INIT_TASK_CFG_QUEUE_LENGTH, TX_NO_WAIT))
455 {
456 sys_error_handler();
457 }
458 CHAR* queue_name = "SYS_Q";
459 (void)tx_queue_create(&s_xTheSystem.m_xSysQueue, queue_name, INIT_TASK_CFG_QUEUE_ITEM_SIZE / sizeof(uint32_t), pcMemory,
460 INIT_TASK_CFG_QUEUE_ITEM_SIZE * INIT_TASK_CFG_QUEUE_LENGTH);
461
462 /* Check if the system has resumed from WWDG reset*/
463 if (__HAL_RCC_GET_FLAG(SYS_RCC_FLAG_WWDGRST) != RESET)
464 {
465 __NOP();
466 SYS_DEBUGF(SYS_DBG_LEVEL_WARNING, ("INIT: start after WWDG reset!\r\n"));
467 }
468#ifndef SYS_TP_MCU_STM32H7
469 /* Check if the system has resumed from the Option Byte loading occurred*/
470 if (__HAL_RCC_GET_FLAG(RCC_FLAG_OBLRST) != RESET)
471 {
472 (void)HAL_FLASH_OB_Lock();
473 (void)HAL_FLASH_Lock();
474
475 SYS_DEBUGF(SYS_DBG_LEVEL_WARNING, ("INIT: start after OB reset!\r\n"));
476 }
477#endif
478
479 /* check the reset flags*/
480 SYS_DEBUGF(SYS_DBG_LEVEL_VERBOSE, ("INIT: reset flags: 0x%x\r\n", READ_BIT(RCC->CSR, 0xFF000000U)));
481
482 /* Clear reset flags in any case*/
483 __HAL_RCC_CLEAR_RESET_FLAGS();
484
485#if (SYS_TS_CFG_ENABLE_SERVICE == 1)
486 /* Initialize the System Timestamp service*/
487 xRes = SysTsInit(&s_xTheSystem.m_xTimestampSrv, SYS_TS_CFG_TSDRIVER_PARAMS);
488 if (SYS_IS_ERROR_CODE(xRes))
489 {
490 __NOP();
491 SYS_DEBUGF(SYS_DBG_LEVEL_WARNING, ("INIT: error during timestamp srv initialization.\r\n"));
492 }
493#endif
494
495 /* Get the default application error manager delegate*/
496 s_xTheSystem.m_pxAppErrorDelegate = SysGetErrorDelegate();
497 /* initialize the application error manager delegate*/
498 xRes = IAEDInit(s_xTheSystem.m_pxAppErrorDelegate, NULL);
499 if (SYS_IS_ERROR_CODE(xRes))
500 {
501 sys_error_handler();
502 }
503
504 /* Get the default Power Mode Helper object*/
505 s_xTheSystem.m_pxAppPowerModeHelper = SysGetPowerModeHelper();
506 /* Initialize the Power Mode Helper object*/
507 xRes = IapmhInit(s_xTheSystem.m_pxAppPowerModeHelper);
508 if (SYS_IS_ERROR_CODE(xRes))
509 {
510 sys_error_handler();
511 }
512
513 /* Allocate the global application context*/
514 ApplicationContext xContext;
515 /* Initialize the context*/
516 xRes = ACInit(&xContext);
517
518 UNUSED(xRes); /* at the moment ACInit() return always SYS_NO_ERROR_CODE */
519
520 xRes = SysLoadApplicationContext(&xContext);
521 if (xRes != SYS_NO_ERROR_CODE)
522 {
523 /* it seems that there is no application to run!!!*/
524 SYS_DEBUGF(SYS_DBG_LEVEL_WARNING, ("INIT: no application tasks loaded!\r\n"));
525 sys_error_handler();
526 }
527
528 SYS_DEBUGF(SYS_DBG_LEVEL_VERBOSE, ("INIT: added %i managed tasks.\r\n", ACGetTaskCount(&xContext)));
529
530 /* Initialize the hardware resource for all tasks*/
531 SYS_DEBUGF(SYS_DBG_LEVEL_VERBOSE, ("INIT: task hardware initialization.\r\n"));
532
533 AManagedTask *pxTask = ACGetFirstTask(&xContext);
534 while ((pxTask != NULL) && !SYS_IS_ERROR_CODE(xRes))
535 {
536 xRes = AMTHardwareInit(pxTask, NULL);
537 if (SYS_IS_ERROR_CODE(xRes))
538 {
539 SYS_SET_SERVICE_LEVEL_ERROR_CODE(SYS_INIT_TASK_FAILURE_ERROR_CODE);
540 SYS_DEBUGF(SYS_DBG_LEVEL_SEVERE, ("\r\nINIT: system failure.\r\n"));
541 }
542 else
543 {
544 pxTask = ACGetNextTask(&xContext, pxTask);
545
546 SYS_DEBUGF(SYS_DBG_LEVEL_VERBOSE, ("...\r\n"));
547 }
548 }
549
550 SYS_DEBUGF(SYS_DBG_LEVEL_VERBOSE, ("DONE.\r\n"));
551
552 /* Create the application tasks*/
553 tx_entry_function_t pvTaskCode;
554 CHAR *pcName;
555 VOID *pvStackStart;
556 ULONG nStackSize;
557 UINT nPriority;
558 UINT nPreemptThreshold;
559 ULONG nTimeSlice;
560 ULONG nAutoStart;
561 ULONG nParams;
562
563 pxTask = ACGetFirstTask(&xContext);
564 while ((pxTask != NULL) && !SYS_IS_ERROR_CODE(xRes))
565 {
566 xRes = AMTOnCreateTask(pxTask, &pvTaskCode, &pcName, &pvStackStart, &nStackSize, &nPriority, &nPreemptThreshold, &nTimeSlice, &nAutoStart, &nParams);
567 if (SYS_IS_ERROR_CODE(xRes))
568 {
569 SYS_SET_SERVICE_LEVEL_ERROR_CODE(SYS_INIT_TASK_FAILURE_ERROR_CODE);
570 SYS_DEBUGF(SYS_DBG_LEVEL_SEVERE, ("INIT: system failure.\r\n"));
571 }
572 else
573 {
574 if (pvStackStart == NULL)
575 {
576 // allocate the task stack in the system memory pool
577 nRtosRes = tx_byte_allocate(&s_xTheSystem.m_xSysMemPool, &pvStackStart, nStackSize, TX_NO_WAIT);
578 }
579 if (nRtosRes == TX_SUCCESS)
580 {
581 if (nAutoStart == TX_AUTO_START)
582 {
583 pxTask->m_xStatus.nAutoStart = 1;
584 }
585 nRtosRes = tx_thread_create(&pxTask->m_xTaskHandle, pcName, pvTaskCode, nParams, pvStackStart, nStackSize,
586 nPriority, nPreemptThreshold, nTimeSlice, TX_DONT_START);
587 }
588 if (nRtosRes != TX_SUCCESS)
589 {
590 SYS_SET_SERVICE_LEVEL_ERROR_CODE(SYS_INIT_TASK_FAILURE_ERROR_CODE);
591 SYS_DEBUGF(SYS_DBG_LEVEL_SEVERE, ("INIT: unable to create task %s.\r\n", pcName));
592 }
593 }
594 pxTask = ACGetNextTask(&xContext, pxTask);
595 }
596
597 (void)SysOnStartApplication(&xContext);
598 (void)IAEDOnStartApplication(s_xTheSystem.m_pxAppErrorDelegate, &xContext);
599
600 SYS_DEBUGF(SYS_DBG_LEVEL_SL, ("INIT: system initialized.\r\n"));
601
602#if defined(DEBUG) || defined(SYS_DEBUG)
603 if (SYS_DBG_LEVEL_SL >= g_sys_dbg_min_level)
604 {
605 APIVersion xVersion = SysGetAPIVersion();
606 ULONG nFreeHeapSize = 0;
607 tx_byte_pool_info_get(&s_xTheSystem.m_xSysMemPool, TX_NULL, &nFreeHeapSize, TX_NULL, TX_NULL, TX_NULL, TX_NULL);
608 SYS_DEBUGF(SYS_DBG_LEVEL_SL, ("INIT: eLooM version v%i.%i.%i\r\n", xVersion.m_nMajor, xVersion.m_nMinor,
609 xVersion.m_nPatch));
610 SYS_DEBUGF(SYS_DBG_LEVEL_SL, ("INIT: free heap = %i.\r\n", nFreeHeapSize));
611 SYS_DEBUGF(SYS_DBG_LEVEL_SL, ("INIT: SystemCoreClock = %iHz.\r\n", SystemCoreClock));
612 }
613#endif
614
615 /*Resume all tasks created with auto_start set to 1.*/
616 pxTask = ACGetFirstTask(&xContext);
617 while ((pxTask != NULL) && (!SYS_IS_ERROR_CODE(xRes)))
618 {
619 if (pxTask->m_xStatus.nAutoStart == 1U)
620 {
621 (void)tx_thread_resume(&pxTask->m_xTaskHandle);
622 }
623 pxTask = ACGetNextTask(&xContext, pxTask);
624 }
625
626 // After the system initialization the INIT task is used to implement some system call
627 // because it is the owner of the Application Context.
628 // At the moment this is an initial implementation of a system level Power Management:
629 // wait for a system level power mode request
630 SysEvent xEvent;
631 for (;;)
632 {
633 if (TX_SUCCESS == tx_queue_receive(&s_xTheSystem.m_xSysQueue, &xEvent, TX_WAIT_FOREVER))
634 {
635 EPowerMode eActivePowerMode = IapmhGetActivePowerMode(s_xTheSystem.m_pxAppPowerModeHelper);
636 /* check if it is a system error event*/
637 if (SYS_IS_ERROR_EVENT(xEvent))
638 {
639 (void)IAEDProcessEvent(s_xTheSystem.m_pxAppErrorDelegate, &xContext, xEvent);
640 // check if the system is in low power mode and it was waked up by a strange IRQ.
641 if (IapmhIsLowPowerMode(s_xTheSystem.m_pxAppPowerModeHelper, eActivePowerMode))
642 {
643 // if the system was wake up due to an error event, then wait the error is recovered before put the MCU in STOP
644 if (!IAEDIsLastErrorPending(s_xTheSystem.m_pxAppErrorDelegate))
645 {
646 // then put the system again in low power mode.
647 (void)IapmhDidEnterPowerMode(s_xTheSystem.m_pxAppPowerModeHelper, eActivePowerMode);
648 }
649 }
650 }
651 else
652 {
653 /* it is a power mode event*/
654 EPowerMode ePowerMode = IapmhComputeNewPowerMode(s_xTheSystem.m_pxAppPowerModeHelper, xEvent);
655 if (ePowerMode != eActivePowerMode)
656 {
657 (void)IapmhCheckPowerModeTransaction(s_xTheSystem.m_pxAppPowerModeHelper, eActivePowerMode, ePowerMode);
658
659 SYS_DEBUGF(SYS_DBG_LEVEL_VERBOSE, ("INIT: evt:src=%x evt:param=%x\r\n", xEvent.xEvent.nSource, xEvent.xEvent.nParam));
660
661 /* first inform the AmanagedTaskEx that a transaction in the power mode state machine
662 is going to begin.*/
663 pxTask = ACGetFirstTask(&xContext);
664 for (; pxTask != NULL; pxTask = ACGetNextTask(&xContext, pxTask))
665 {
666 if (INIT_IS_KIND_OF_AMTEX(pxTask))
667 {
668 xRes = AMTExOnEnterPowerMode((AManagedTaskEx *)pxTask, eActivePowerMode, ePowerMode);
669 if (SYS_IS_ERROR_CODE(xRes))
670 {
671 sys_error_handler();
672 }
673 }
674 }
675
676 uint16_t nTaskToDoPMSwitch = ACGetTaskCount(&xContext);
677 /* then we do the power mode transaction for the task belonging to CLASS_0*/
678 nTaskToDoPMSwitch -= InitTaskDoEnterPowerModeForPMClass(&xContext, E_PM_CLASS_0, eActivePowerMode, ePowerMode);
679 if (nTaskToDoPMSwitch > 0U)
680 {
681 /* then we do the power mode transaction for the task belonging to CLASS_1*/
682 nTaskToDoPMSwitch -= InitTaskDoEnterPowerModeForPMClass(&xContext, E_PM_CLASS_1, eActivePowerMode, ePowerMode);
683 }
684 if (nTaskToDoPMSwitch > 0U)
685 {
686 /* then we do the power mode transaction for the task belonging to CLASS_2*/
687 nTaskToDoPMSwitch -= InitTaskDoEnterPowerModeForPMClass(&xContext, E_PM_CLASS_2, eActivePowerMode, ePowerMode);
688 }
689
690 /* Enter the specified power mode*/
691 (void)IapmhDidEnterPowerMode(s_xTheSystem.m_pxAppPowerModeHelper, ePowerMode);
692
693#if defined(DEBUG) || defined(SYS_DEBUG)
694 if (SYS_DBG_LEVEL_SL >= g_sys_dbg_min_level)
695 {
696 ULONG nFreeHeapSize = 0;
697 tx_byte_pool_info_get(&s_xTheSystem.m_xSysMemPool, TX_NULL, &nFreeHeapSize, TX_NULL, TX_NULL, TX_NULL, TX_NULL);
698 SYS_DEBUGF(SYS_DBG_LEVEL_SL, ("INIT: free heap = %i.\r\n", nFreeHeapSize));
699 }
700#endif
701
702 pxTask = ACGetFirstTask(&xContext);
703 for (; pxTask != NULL; pxTask = ACGetNextTask(&xContext, pxTask))
704 {
705 pxTask->m_xStatus.nPowerModeSwitchDone = 0;
706 pxTask->m_xStatus.nPowerModeSwitchPending = 0;
707 (void)tx_thread_resume(&pxTask->m_xTaskHandle);
708 }
709 }
710 else
711 {
712 /* check if the system is in a low power mode and it was waked up by a strange IRQ.*/
713 if (IapmhIsLowPowerMode(s_xTheSystem.m_pxAppPowerModeHelper, eActivePowerMode))
714 {
715 /* then put the system again in low power mode.*/
716 (void)IapmhDidEnterPowerMode(s_xTheSystem.m_pxAppPowerModeHelper, ePowerMode);
717 }
718 }
719 }
720 }
721 }

724static uint16_t InitTaskDoEnterPowerModeForPMClass(ApplicationContext *pxContext, EPMClass ePowerModeClass,
725 const EPowerMode eActivePowerMode, const EPowerMode eNewPowerMode)
726{
727 /* Forward the request to all managed tasks*/
728 AManagedTask *pTask;
729 EPMClass eTaskPMClass;
730 boolean_t bDelayPowerModeSwitch;
731 uint16_t nTaskCount = 0;
732
733 do
734 {
735 bDelayPowerModeSwitch = FALSE;
736 pTask = ACGetFirstTask(pxContext);
737 for (; pTask != NULL; pTask = ACGetNextTask(pxContext, pTask))
738 {
739
740 /* check if the task is a AMTEx and, in case, if its power mode class is equal to ePowerModeClass*/
741 eTaskPMClass = INIT_IS_KIND_OF_AMTEX(pTask) ? AMTExGetPMClass((AManagedTaskEx *)pTask) : E_PM_CLASS_0;
742 if (eTaskPMClass == ePowerModeClass)
743 {
744 /* notify the task that the power mode is changing,
745 so the task will suspend.*/
746 pTask->m_xStatus.nPowerModeSwitchPending = 1;
747 if (pTask->m_xStatus.nPowerModeSwitchDone == 0U)
748 {
749 if ((pTask->m_xStatus.nDelayPowerModeSwitch == 0U))
750 {
751
752 (void)AMTDoEnterPowerMode(pTask, eActivePowerMode, eNewPowerMode);
753 pTask->m_xStatus.nPowerModeSwitchDone = 1;
754 pTask->m_xStatus.nIsTaskStillRunning = 1;
755 nTaskCount++;
756 }
757 else
758 {
759 /* check if it is an Extended Managed Task*/
760 if (pTask->m_xStatus.nReserved == 1U)
761 {
762 // force the step execution to prepare the task for the power mode switch.
763 (void)AMTExForceExecuteStep((AManagedTaskEx *)pTask, eActivePowerMode);
764 }
765 bDelayPowerModeSwitch = TRUE;
766 }
767 }
768 }
769 }
770
771 if (bDelayPowerModeSwitch == TRUE)
772 {
773 tx_thread_sleep(SYS_MS_TO_TICKS(INIT_TASK_CFG_PM_SWITCH_DELAY_MS));
774 }
775
776 } while (bDelayPowerModeSwitch == TRUE);
777
778 return nTaskCount;
779}
