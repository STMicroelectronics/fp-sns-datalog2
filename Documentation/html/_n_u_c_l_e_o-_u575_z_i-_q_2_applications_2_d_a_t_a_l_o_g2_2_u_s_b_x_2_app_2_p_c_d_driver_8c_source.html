45 uint16_t ctrl_fifo_size, uint8_t n_in_ep)
46{
47 assert_param(_this != NULL);
48
49 if (!_this->is_initialized)
50 {
51 return SYS_BASE_ERROR_CODE;
52 }
53
54 uint16_t remaining_size = total_fifo_size;
55
56 /* OUT EP: not used for this class */
57 HAL_PCD_SetRxFiFo(_this->handle.p_mx_pcd_cfg->p_pcd, rx_fifo_size);
58 remaining_size -= rx_fifo_size;
59
60 /* Control EP: used for PnPL commands */
61 HAL_PCDEx_SetTxFiFo(_this->handle.p_mx_pcd_cfg->p_pcd, 0, ctrl_fifo_size);
62 remaining_size -= ctrl_fifo_size;
63
64 /* IN EP: used for raw data transmission */
65 for (int i = 0; i < n_in_ep; i++)
66 {
67 HAL_PCDEx_SetTxFiFo(_this->handle.p_mx_pcd_cfg->p_pcd, i + 1, remaining_size / n_in_ep);
68 }
69
70 return SYS_NO_ERROR_CODE;
71}
