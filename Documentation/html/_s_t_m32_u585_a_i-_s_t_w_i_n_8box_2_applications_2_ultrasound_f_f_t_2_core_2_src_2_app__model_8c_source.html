607/* USER Code : --> compute stream ids */
608static uint8_t __stream_control(ILog_Controller_t *ifn, bool status)
609{
610 int8_t i;
611 if (status) //set stream ids
612 {
613 int8_t j, st_i = 0;
614 uint16_t proposed_fifoWM = 1;
615 //sort stream id by bandwidth
616 for (i = 0; i < SENSOR_NUMBER; i++)
617 {
618 if (app_model.s_models[i] != NULL)
619 {
620 if (app_model.s_models[i]->sensor_status.is_active == true)
621 {
622 if (app_model.s_models[i]->sensor_status.isensor_class == ISENSOR_CLASS_MEMS)
623 {
624 app_model.s_models[i]->stream_params.bandwidth = app_model.s_models[i]->sensor_status.type.mems.odr * SMGetnBytesPerSample(i);
625 }
626 else if (app_model.s_models[i]->sensor_status.isensor_class == ISENSOR_CLASS_AUDIO)
627 {
628 app_model.s_models[i]->stream_params.bandwidth = app_model.s_models[i]->sensor_status.type.audio.frequency * SMGetnBytesPerSample(i);
629 }
630 else
631 {
632 /* TODO: add support for other ISENSOR_CLASS */
633 }
634
635 if (app_model.log_controller_model.interface == LOG_CTRL_MODE_SD)
636 {
637 /* 330ms of sensor data. Access to SD is optimized when buffer dimension is multiple of 512 */
638 app_model.s_models[i]->stream_params.sd_dps = (uint32_t)(app_model.s_models[i]->stream_params.bandwidth * 0.33f);
639 app_model.s_models[i]->stream_params.sd_dps = app_model.s_models[i]->stream_params.sd_dps - (app_model.s_models[i]->stream_params.sd_dps % 512) + 512;
640
641 /*********** IS IT STILL multiple of 512? **************/
642 if (app_model.s_models[i]->stream_params.sd_dps > app_model.s_models[i]->stream_params.bandwidth)
643 {
644 app_model.s_models[i]->stream_params.sd_dps = (uint32_t)app_model.s_models[i]->stream_params.bandwidth + 8;
645 }
646
647 proposed_fifoWM = app_model.s_models[i]->stream_params.sd_dps / SMGetnBytesPerSample(i) / 2;
648 if (proposed_fifoWM == 0)
649 {
650 proposed_fifoWM = 1;
651 }
652 sys_error_code_t ret = SMSensorSetFifoWM(i, proposed_fifoWM);
653 if (ret != SYS_NO_ERROR_CODE)
654 {
655
656 }
657 }
658 else if (app_model.log_controller_model.interface == LOG_CTRL_MODE_USB)
659 {
660 /* in case of slow sensor send 1 sample for each usb packet */
661 float low_odr = 0;
662 if (app_model.s_models[i]->sensor_status.isensor_class == ISENSOR_CLASS_MEMS)
663 {
664 low_odr = app_model.s_models[i]->sensor_status.type.mems.odr;
665 }
666 else if (app_model.s_models[i]->sensor_status.isensor_class == ISENSOR_CLASS_AUDIO)
667 {
668 low_odr = app_model.s_models[i]->sensor_status.type.audio.frequency;
669 }
670 else
671 {
672 /* TODO: add support for other ISENSOR_CLASS */
673 }
674
675 if (low_odr <= 20)
676 {
677 /* When there's a timestamp, more then one packet will be sent */
678 app_model.s_models[i]->stream_params.usb_dps = SMGetnBytesPerSample(i) + 8; /* 8 = timestamp dimension in bytes */
679 proposed_fifoWM = 1;
680 sys_error_code_t ret = SMSensorSetFifoWM(i, proposed_fifoWM);
681 if (ret != SYS_NO_ERROR_CODE)
682 {
683
684 }
685 }
686 else
687 {
688 /* 50ms of sensor data; when there's a timestamp packets will be sent fastly */
