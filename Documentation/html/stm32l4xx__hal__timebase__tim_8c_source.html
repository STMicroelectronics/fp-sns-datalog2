42HAL_StatusTypeDef HAL_InitTick(uint32_t TickPriority)
43{
44 RCC_ClkInitTypeDef clkconfig;
45 uint32_t uwTimclock, uwAPB1Prescaler;
46
47 uint32_t uwPrescalerValue;
48 uint32_t pFLatency;
49 HAL_StatusTypeDef status = HAL_OK;
50
51 /* Enable TIM6 clock */
52 __HAL_RCC_TIM6_CLK_ENABLE();
53
54 /* Get clock configuration */
55 HAL_RCC_GetClockConfig(&clkconfig, &pFLatency);
56
57 /* Get APB1 prescaler */
58 uwAPB1Prescaler = clkconfig.APB1CLKDivider;
59 /* Compute TIM6 clock */
60 if (uwAPB1Prescaler == RCC_HCLK_DIV1)
61 {
62 uwTimclock = HAL_RCC_GetPCLK1Freq();
63 }
64 else
65 {
66 uwTimclock = 2UL * HAL_RCC_GetPCLK1Freq();
67 }
68
69 /* Compute the prescaler value to have TIM6 counter clock equal to 1MHz */
70 uwPrescalerValue = (uint32_t) ((uwTimclock / 1000000U) - 1U);
71
72 /* Initialize TIM6 */
73 htim6.Instance = TIM6;
74
75 /* Initialize TIMx peripheral as follow:
76
77 + Period = [(TIM6CLK/1000) - 1]. to have a (1/1000) s time base.
78 + Prescaler = (uwTimclock/1000000 - 1) to have a 1MHz counter clock.
79 + ClockDivision = 0
80 + Counter direction = Up
81 */
82 htim6.Init.Period = (1000000U / 1000U) - 1U;
83 htim6.Init.Prescaler = uwPrescalerValue;
84 htim6.Init.ClockDivision = 0;
85 htim6.Init.CounterMode = TIM_COUNTERMODE_UP;
86 htim6.Init.AutoReloadPreload = TIM_AUTORELOAD_PRELOAD_DISABLE;
87
88 status = HAL_TIM_Base_Init(&htim6);
89 if (status == HAL_OK)
90 {
91 /* Start the TIM time Base generation in interrupt mode */
92 status = HAL_TIM_Base_Start_IT(&htim6);
93 if (status == HAL_OK)
94 {
95 /* Enable the TIM6 global Interrupt */
96 HAL_NVIC_EnableIRQ(TIM6_DAC_IRQn);
97 /* Configure the SysTick IRQ priority */
98 if (TickPriority < (1UL << __NVIC_PRIO_BITS))
99 {
100 /* Configure the TIM IRQ priority */
101 HAL_NVIC_SetPriority(TIM6_DAC_IRQn, TickPriority, 0U);
102 uwTickPrio = TickPriority;
103 }
104 else
105 {
106 status = HAL_ERROR;
107 }
108 }
109 }
110
111 HAL_TIM_RegisterCallback(&htim6, HAL_TIM_PERIOD_ELAPSED_CB_ID, TimeBase_TIM_PeriodElapsedCallback);
112
113 /* Return function status */
114 return status;
115}
