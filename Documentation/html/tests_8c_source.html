203 TEST(!json_object_dothas_value_of_type(root_object, "object.nested array", JSONObject));

208 TEST(json_object_dothas_value_of_type(root_object, "object.nested number", JSONNumber));
209 TEST(!json_object_dothas_value_of_type(root_object, "_object.nested whatever", JSONNumber));

829static char * read_file(const char * file_path) {
830 FILE *fp = NULL;
831 size_t size_to_read = 0;
832 size_t size_read = 0;
833 long pos;
834 char *file_contents;
835 fp = fopen(file_path, "r");
836 if (!fp) {
837 assert(0);
838 return NULL;
839 }
840 fseek(fp, 0L, SEEK_END);
841 pos = ftell(fp);
842 if (pos < 0) {
843 fclose(fp);
844 assert(0);
845 return NULL;
846 }
847 size_to_read = pos;
848 rewind(fp);
849 file_contents = (char*)malloc(sizeof(char) * (size_to_read + 1));
850 if (!file_contents) {
851 fclose(fp);
852 assert(0);
853 return NULL;
854 }
855 size_read = fread(file_contents, 1, size_to_read, fp);
856 if (size_read == 0 || ferror(fp)) {
857 fclose(fp);
858 free(file_contents);
859 assert(0);
860 return NULL;
861 }
862 fclose(fp);
863 file_contents[size_read] = '\0';
864 return file_contents;
865}
