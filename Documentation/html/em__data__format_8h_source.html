482static inline
483sys_error_code_t EMD_2dGetValueAt(const EMData_t *p_data, void *p_val, uint16_t d1_idx, uint16_t d2_idx)
484{
485 assert_param(p_data != NULL);
486 assert_param(p_data->dimensions == 2);
487 assert_param(d1_idx < p_data->shapes[0]);
488 assert_param(d2_idx < p_data->shapes[1]);
489
490 uint32_t val_pos = (p_data->shapes[1] * d1_idx) + d2_idx;
491 uintptr_t val_addr = ((uintptr_t)p_data->p_payload) + (val_pos * EMD_GetElementSize(p_data));
492 memcpy(p_val, (void *)val_addr, p_data->element_size);
493
494 return SYS_NO_ERROR_CODE;
495}
