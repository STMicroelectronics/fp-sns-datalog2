118sys_error_code_t SysLoadApplicationContext(ApplicationContext *pAppContext)
119{
120 assert_param(pAppContext);
121 sys_error_code_t res = SYS_NO_ERROR_CODE;
122 boolean_t ext_iis330is = FALSE;
123 hwd_st25dv_version st25dv_version;
124
125 /* Workaround to set malloc/free function even if BLE Init fails */
126 json_set_allocation_functions(SysAlloc, SysFree);
127
128 /* Check availability of external sensors */
129 ext_iis330is = HardwareDetection_Check_Ext_ISM330IS();
130
131 /* Check NFC chip version */
132 st25dv_version = HardwareDetection_Check_ST25DV();
133 if (st25dv_version == ST25DV04KC)
134 {
135 BOARD_ID = BOARD_ID_PROB;
136 FW_ID = USB_FW_ID_DATALOG2_PROB;
137 }
138
139 /* Allocate the task objects */
140 sUtilObj = UtilTaskAlloc(&MX_GPIO_SW1InitParams, &MX_GPIO_LED1InitParams, &MX_GPIO_LED2InitParams, &MX_GPIO_LED3InitParams);
141 sDatalogAppObj = DatalogAppTaskAlloc();
142 sI2C1BusObj = I2CBusTaskAlloc(&MX_I2C1InitParams);
143 sSPI2BusObj = SPIBusTaskAlloc(&MX_SPI2InitParams);
144 sSPI3BusObj = SPIBusTaskAlloc(&MX_SPI3InitParams);
145 sLIS2DU12Obj = LIS2DU12TaskAlloc(&MX_GPIO_ACC_INT2InitParams, &MX_GPIO_SPI_SEN_CS_AInitParams);
146 sLIS2MDLObj = LIS2MDLTaskAlloc(NULL, NULL);
147 sLPS22DFObj = LPS22DFTaskAlloc(NULL, NULL);
148 sMP23DB01HPObj = MP23DB01HPTaskAlloc(&MX_ADF1InitParams);
149 sSTTS22HObj = STTS22HTaskAlloc(NULL, NULL, 0x71);
150
151 /* Use the external ISM330IS with ISPU or the onboard LSM6DSV16X with MLC */
152 if (ext_iis330is)
153 {
154 sISM330ISObj = ISM330ISTaskAlloc(&MX_GPIO_INT2_EXTERNALInitParams, &MX_GPIO_INT1_EXTERNALInitParams, &MX_GPIO_CS_EXTERNALInitParams);
155 }
156 else
157 {
158 sLSM6DSV16XObj = LSM6DSV16XTaskAlloc(&MX_GPIO_IMU_INT1InitParams, NULL, &MX_GPIO_SPI_SEN_CS_GInitParams);
159 }
160
161 /* Add the task object to the context. */
162 res = ACAddTask(pAppContext, (AManagedTask *) sUtilObj);
163 res = ACAddTask(pAppContext, (AManagedTask *) sDatalogAppObj);
164 res = ACAddTask(pAppContext, (AManagedTask *) sI2C1BusObj);
165 res = ACAddTask(pAppContext, (AManagedTask *) sSPI2BusObj);
166 res = ACAddTask(pAppContext, (AManagedTask *) sSPI3BusObj);
167 res = ACAddTask(pAppContext, (AManagedTask *) sLIS2DU12Obj);
168 res = ACAddTask(pAppContext, (AManagedTask *) sLIS2MDLObj);
169 res = ACAddTask(pAppContext, (AManagedTask *) sLPS22DFObj);
170 res = ACAddTask(pAppContext, (AManagedTask *) sMP23DB01HPObj);
171 res = ACAddTask(pAppContext, (AManagedTask *) sSTTS22HObj);
172
173 /* Use the external ISM330IS with ISPU or the onboard LSM6DSV16X with MLC */
174 if (ext_iis330is)
175 {
176 res = ACAddTask(pAppContext, (AManagedTask *) sISM330ISObj);
177 }
178 else
179 {
180 res = ACAddTask(pAppContext, (AManagedTask *) sLSM6DSV16XObj);
181 }
182
183 pLIS2DU12_ACC_PnPLObj = Lis2du12_Acc_PnPLAlloc();
184 pLIS2MDL_MAG_PnPLObj = Lis2mdl_Mag_PnPLAlloc();
185 pLPS22DF_PRESS_PnPLObj = Lps22df_Press_PnPLAlloc();
186 pMP23DB01HP_MIC_PnPLObj = Mp23db01hp_Mic_PnPLAlloc();
187 pSTTS22H_TEMP_PnPLObj = Stts22h_Temp_PnPLAlloc();
188
189 /* Use the external ISM330IS with ISPU or the onboard LSM6DSV16X with MLC */
190 if (sISM330ISObj)
191 {
192 pISM330IS_ACC_PnPLObj = Ism330is_Acc_PnPLAlloc();
193 pISM330IS_GYRO_PnPLObj = Ism330is_Gyro_PnPLAlloc();
194 pISM330IS_ISPU_PnPLObj = Ism330is_Ispu_PnPLAlloc();
195 }
196 if(sLSM6DSV16XObj)
197 {
198 pLSM6DSV16X_ACC_PnPLObj = Lsm6dsv16x_Acc_PnPLAlloc();
199 pLSM6DSV16X_GYRO_PnPLObj = Lsm6dsv16x_Gyro_PnPLAlloc();
200 pLSM6DSV16X_MLC_PnPLObj = Lsm6dsv16x_Mlc_PnPLAlloc();
201 }
202
203 pDeviceInfoPnPLObj = Deviceinformation_PnPLAlloc();
204 pAcquisitionInfoPnPLObj = Acquisition_Info_PnPLAlloc();
205 pTagsInfoPnPLObj = Tags_Info_PnPLAlloc();
206 pFirmwareInfoPnPLObj = Firmware_Info_PnPLAlloc();
207 pLogControllerPnPLObj = Log_Controller_PnPLAlloc();
208 pAutomodePnPLObj = Automode_PnPLAlloc();
209
210 PnPLSetBOARDID(BOARD_ID);
211 PnPLSetFWID(FW_ID);
212
213 return res;
214}

216sys_error_code_t SysOnStartApplication(ApplicationContext *pAppContext)
217{
218 UNUSED(pAppContext);
219
220 /* connect the sensor task to the bus. */
221 I2CBusTaskConnectDevice((I2CBusTask *) sI2C1BusObj, (I2CBusIF *)LIS2MDLTaskGetSensorIF((LIS2MDLTask *) sLIS2MDLObj));
222 I2CBusTaskConnectDevice((I2CBusTask *) sI2C1BusObj, (I2CBusIF *)LPS22DFTaskGetSensorIF((LPS22DFTask *) sLPS22DFObj));
223 I2CBusTaskConnectDevice((I2CBusTask *) sI2C1BusObj, (I2CBusIF *)STTS22HTaskGetSensorIF((STTS22HTask *) sSTTS22HObj));
224 SPIBusTaskConnectDevice((SPIBusTask *) sSPI2BusObj, (SPIBusIF *)LIS2DU12TaskGetSensorIF((LIS2DU12Task *) sLIS2DU12Obj));
225
226 /* Use the external ISM330IS with ISPU or the onboard LSM6DSV16X with MLC */
227 if(sISM330ISObj)
228 {
229 SPIBusTaskConnectDevice((SPIBusTask *) sSPI3BusObj, (SPIBusIF *)ISM330ISTaskGetSensorIF((ISM330ISTask *) sISM330ISObj));
230 }
231 if(sLSM6DSV16XObj)
232 {
233 SPIBusTaskConnectDevice((SPIBusTask *) sSPI2BusObj, (SPIBusIF *)LSM6DSV16XTaskGetSensorIF((LSM6DSV16XTask *) sLSM6DSV16XObj));
234 }
235
236 /************ Connect the Sensor events to the DatalogAppTask ************/
237 IEventListener *DatalogAppListener = DatalogAppTask_GetEventListenerIF((DatalogAppTask *) sDatalogAppObj);
238 IEventSrcAddEventListener(LIS2DU12TaskGetEventSrcIF((LIS2DU12Task *) sLIS2DU12Obj), DatalogAppListener);
239 IEventSrcAddEventListener(LIS2MDLTaskGetMagEventSrcIF((LIS2MDLTask *) sLIS2MDLObj), DatalogAppListener);
240 IEventSrcAddEventListener(LPS22DFTaskGetPressEventSrcIF((LPS22DFTask *) sLPS22DFObj), DatalogAppListener);
241 IEventSrcAddEventListener(MP23DB01HPTaskGetEventSrcIF((MP23DB01HPTask *) sMP23DB01HPObj), DatalogAppListener);
242 IEventSrcAddEventListener(STTS22HTaskGetTempEventSrcIF((STTS22HTask *) sSTTS22HObj), DatalogAppListener);
243
244 /* Use the external ISM330IS with ISPU or the onboard LSM6DSV16X with MLC */
245 if (sISM330ISObj)
246 {
247 IEventSrcAddEventListener(ISM330ISTaskGetAccEventSrcIF((ISM330ISTask *) sISM330ISObj), DatalogAppListener);
248 IEventSrcAddEventListener(ISM330ISTaskGetGyroEventSrcIF((ISM330ISTask *) sISM330ISObj), DatalogAppListener);
249 IEventSrcAddEventListener(ISM330ISTaskGetMlcEventSrcIF((ISM330ISTask *) sISM330ISObj), DatalogAppListener);
250 }
251 if(sLSM6DSV16XObj)
252 {
253 IEventSrcAddEventListener(LSM6DSV16XTaskGetAccEventSrcIF((LSM6DSV16XTask *) sLSM6DSV16XObj), DatalogAppListener);
254 IEventSrcAddEventListener(LSM6DSV16XTaskGetGyroEventSrcIF((LSM6DSV16XTask *) sLSM6DSV16XObj), DatalogAppListener);
255 IEventSrcAddEventListener(LSM6DSV16XTaskGetMlcEventSrcIF((LSM6DSV16XTask *) sLSM6DSV16XObj), DatalogAppListener);
256 }
257
258 //Sensor PnPL Components
259 Lis2du12_Acc_PnPLInit(pLIS2DU12_ACC_PnPLObj);
260 Lis2mdl_Mag_PnPLInit(pLIS2MDL_MAG_PnPLObj);
261 Lps22df_Press_PnPLInit(pLPS22DF_PRESS_PnPLObj);
262 Mp23db01hp_Mic_PnPLInit(pMP23DB01HP_MIC_PnPLObj);
263 Stts22h_Temp_PnPLInit(pSTTS22H_TEMP_PnPLObj);
264
265 /* Use the external ISM330IS with ISPU or the onboard LSM6DSV16X with MLC */
266 if (sISM330ISObj)
267 {
268 Ism330is_Acc_PnPLInit(pISM330IS_ACC_PnPLObj);
269 Ism330is_Gyro_PnPLInit(pISM330IS_GYRO_PnPLObj);
270 Ism330is_Ispu_PnPLInit(pISM330IS_ISPU_PnPLObj, DatalogAppTask_GetIIspuControllerIF((DatalogAppTask *) sDatalogAppObj, (AManagedTask *) sISM330ISObj));
271 }
272 if(sLSM6DSV16XObj)
273 {
274 Lsm6dsv16x_Acc_PnPLInit(pLSM6DSV16X_ACC_PnPLObj);
275 Lsm6dsv16x_Gyro_PnPLInit(pLSM6DSV16X_GYRO_PnPLObj);
276 Lsm6dsv16x_Mlc_PnPLInit(pLSM6DSV16X_MLC_PnPLObj, DatalogAppTask_GetIMLCControllerIF((DatalogAppTask *) sDatalogAppObj, (AManagedTask *) sLSM6DSV16XObj));
277 }
278
279 //Other PnPL Components
280 Deviceinformation_PnPLInit(pDeviceInfoPnPLObj);
281 Firmware_Info_PnPLInit(pFirmwareInfoPnPLObj);
282 Acquisition_Info_PnPLInit(pAcquisitionInfoPnPLObj);
283 Tags_Info_PnPLInit(pTagsInfoPnPLObj);
284 Log_Controller_PnPLInit(pLogControllerPnPLObj, DatalogAppTask_GetILogControllerIF((DatalogAppTask *) sDatalogAppObj));
285 Automode_PnPLInit(pAutomodePnPLObj);
286
287 return SYS_NO_ERROR_CODE;
288}
