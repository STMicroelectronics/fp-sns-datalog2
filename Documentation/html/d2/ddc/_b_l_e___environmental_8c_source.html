75BleCharTypeDef *BLE_InitEnvService(uint8_t PressEnable, uint8_t HumEnable, uint8_t NumTempEnabled)
76{
77 /* Data structure pointer for BLE service */
78 BleCharTypeDef *BleCharPointer = NULL;
79 EnvironmentalCharSize = 2;
80
81 /* Init data structure for identify environmental info services enabled */
82 EnvFeaturesEnabled.PressureIsEnable = PressEnable;
83 EnvFeaturesEnabled.HumidityIsEnable = HumEnable;
84 EnvFeaturesEnabled.NumberTemperaturesEnabled = NumTempEnabled;
85
86 if ((PressEnable == 1U) ||
87 (HumEnable == 1U) ||
88 (NumTempEnabled > 0U))
89 {
90 /* Init data structure pointer for environmental info service */
91 BleCharPointer = &BleCharEnv;
92 memset(BleCharPointer, 0, sizeof(BleCharTypeDef));
93 BleCharPointer->AttrMod_Request_CB = AttrMod_Request_Env;
94 BleCharPointer->Read_Request_CB = Read_Request_Env;
95 COPY_ENVIRONMENTAL_CHAR_UUID((BleCharPointer->uuid));
96
97 /* Enables BLE Pressure feature */
98 if (PressEnable == 1U)
99 {
100 BleCharPointer->uuid[14] |= 0x10U;
101 EnvironmentalCharSize += 4U;
102#if (BLE_DEBUG_LEVEL>1)
103 BLE_MANAGER_PRINTF("\t--> Pressure feature enabled\r\n");
104#endif /* BLE_DEBUG_LEVEL>1 */
105 }
106
107 /* Enables BLE Humidity feature */
108 if (HumEnable == 1U)
109 {
110 BleCharPointer->uuid[14] |= 0x08U;
111 EnvironmentalCharSize += 2U;
112#if (BLE_DEBUG_LEVEL>1)
113 BLE_MANAGER_PRINTF("\t--> Humidity feature enabled\r\n");
114#endif /* BLE_DEBUG_LEVEL>1 */
115 }
116
117 if (NumTempEnabled == 1U)
118 {
119 /* Enables only one BLE temperature feature */
120 BleCharPointer->uuid[14] |= 0x04U;
121 EnvironmentalCharSize += 2U;
122#if (BLE_DEBUG_LEVEL>1)
123 BLE_MANAGER_PRINTF("\t--> Only one temperature feature enabled\r\n");
124#endif /* BLE_DEBUG_LEVEL>1 */
125 }
126 else if (NumTempEnabled == 2U)
127 {
128 /* Enables two BLE temperatures feature */
129 BleCharPointer->uuid[14] |= 0x05U;
130 EnvironmentalCharSize += 4U;
131#if (BLE_DEBUG_LEVEL>1)
132 BLE_MANAGER_PRINTF("\t--> Two temperatures features enabled\r\n");
133#endif /* BLE_DEBUG_LEVEL>1 */
134 }
135
136 BleCharPointer->Char_UUID_Type = UUID_TYPE_128;
137 BleCharPointer->Char_Value_Length = EnvironmentalCharSize;
138 BleCharPointer->Char_Properties = ((uint8_t)(CHAR_PROP_NOTIFY)) | ((uint8_t)(CHAR_PROP_READ));
139 BleCharPointer->Security_Permissions = ATTR_PERMISSION_NONE;
140 BleCharPointer->GATT_Evt_Mask = GATT_NOTIFY_READ_REQ_AND_WAIT_FOR_APPL_RESP;
141 BleCharPointer->Enc_Key_Size = 16;
142 BleCharPointer->Is_Variable = 0;
143
144 if (CustomReadRequestEnv == NULL)
145 {
146 BLE_MANAGER_PRINTF("Warning: Read request environmental function not defined\r\n");
147 }
148
149 BLE_MANAGER_PRINTF("BLE Environmental features ok\r\n");
150 }
151 else
152 {
153 BLE_MANAGER_PRINTF(" ERROR: None environmental features is enabled\r\n");
154 }
155
156 return BleCharPointer;
157}

201tBleStatus BLE_EnvironmentalUpdate(int32_t Press, uint16_t Hum, int16_t Temp1, int16_t Temp2)
202{
203 tBleStatus ret;
204 uint8_t BuffPos;
205
206 uint8_t buff[2 + 4/*Press*/ + 2/*Hum*/ + 2/*Temp1*/ + 2/*Temp2*/];
207
208 /* Time Stamp */
209 STORE_LE_16(buff, (HAL_GetTick() >> 3));
210 BuffPos = 2;
211
212 if (EnvFeaturesEnabled.PressureIsEnable == 1U)
213 {
214 STORE_LE_32((buff + BuffPos), ((uint32_t)Press));
215 BuffPos += 4U;
216 }
217
218 if (EnvFeaturesEnabled.HumidityIsEnable == 1U)
219 {
220 STORE_LE_16((buff + BuffPos), Hum);
221 BuffPos += 2U;
222 }
223
224 if (EnvFeaturesEnabled.NumberTemperaturesEnabled >= 1U)
225 {
226 STORE_LE_16((buff + BuffPos), ((uint16_t)Temp1));
227 BuffPos += 2U;
228 }
229
230 if (EnvFeaturesEnabled.NumberTemperaturesEnabled == 2U)
231 {
232 STORE_LE_16((buff + BuffPos), ((uint16_t)Temp2));
233 BuffPos += 2U;
234 }
235
236 ret = ACI_GATT_UPDATE_CHAR_VALUE(&BleCharEnv, 0, EnvironmentalCharSize, buff);
237
238 if (ret != (tBleStatus)BLE_STATUS_SUCCESS)
239 {
240 if (BLE_StdErr_Service == BLE_SERV_ENABLE)
241 {
242 BytesToWrite = (uint8_t)sprintf((char *)BufferToWrite, "Error Updating Environmental Char\n");
243 Stderr_Update(BufferToWrite, BytesToWrite);
244 }
245 else
246 {
247 BLE_MANAGER_PRINTF("Error: Updating Environmental Char\r\n");
248 }
249 }
250 return ret;
251}
