365void ExtiDefISR(uint16_t exti_pin)
366{
367 EXTIPin2CallbckMap xMap = EXTI_GET_P2F_MAP();
368 for (int i = 0; xMap[i].pfCallback != NULL; i++)
369 {
370 if (__HAL_GPIO_EXTI_GET_IT(xMap[i].nPin) && (xMap[i].nPin == exti_pin))
371 {
372 /* EXTI line interrupt detected */
373 __HAL_GPIO_EXTI_CLEAR_IT(xMap[i].nPin);
374 xMap[i].pfCallback(xMap[i].nPin);
375 }
376 }
