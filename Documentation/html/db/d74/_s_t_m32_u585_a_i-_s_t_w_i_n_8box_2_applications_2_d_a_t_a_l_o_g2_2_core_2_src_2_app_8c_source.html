142 uint8_t stts22h_address;
143 boolean_t ext_iis3dwb = FALSE;
144 boolean_t ext_iis330is = FALSE;
145 boolean_t ext_stts22h = FALSE;

148 json_set_allocation_functions(SysAlloc, SysFree);

150 /* Check availability of external sensors */
151 ext_iis3dwb = HardwareDetection_Check_Ext_IIS3DWB();

153 ext_stts22h = HardwareDetection_Check_Ext_STTS22H(&stts22h_address);

155 /* Check NFC chip version */
156 st25dv_version = HardwareDetection_Check_ST25DV();
157 if (st25dv_version == ST25DV64KC)
158 {
159 BOARD_ID = BOARD_ID_BOXB;
160 FW_ID = USB_FW_ID_DATALOG2_BOXB;

163 /************ Allocate task objects ************/
164 sUtilObj = UtilTaskAlloc(&MX_TIM4InitParams, &MX_GPIO_PA8InitParams, &MX_GPIO_PA0InitParams, &MX_GPIO_PD0InitParams, &MX_TIM5InitParams, &MX_ADC4InitParams,
165 &MX_GPIO_UBInitParams, &MX_GPIO_LED1InitParams, &MX_GPIO_LED2InitParams);
166 sDatalogAppObj = DatalogAppTaskAlloc();
167 sI2C2BusObj = I2CBusTaskAlloc(&MX_I2C2InitParams);

169 sIIS2DLPCObj = IIS2DLPCTaskAlloc(&MX_GPIO_INT2_DLPCInitParams, &MX_GPIO_CS_DLPCInitParams);

171 sIIS2MDCObj = IIS2MDCTaskAlloc(&MX_GPIO_INT_MAGInitParams, NULL);
172 sIIS3DWBObj = IIS3DWBTaskAlloc(&MX_GPIO_INT1_DWBInitParams, &MX_GPIO_CS_DWBInitParams);
173
174 if (ext_iis3dwb)
175 {
176 /* Use the external IIS3DWB and onboard ISM330DHCX */
177 sIIS3DWBExtObj = IIS3DWBTaskAllocSetName(&MX_GPIO_INT1_EXTERNAL_DWBInitParams, &MX_GPIO_CS_EXTERNALInitParams, "iis3dwb_ext");
178 sISM330DHCXObj = ISM330DHCXTaskAlloc(&MX_GPIO_INT1_DHCXInitParams, &MX_GPIO_INT2_DHCXInitParams, &MX_GPIO_CS_DHCXInitParams);
179 }
180 else if (ext_iis330is)
181 {
182 /* Use the onboard IIS3DWB and the external ISM330IS */
183 sISM330ISObj = ISM330ISTaskAlloc(&MX_GPIO_INT2_EXInitParams, &MX_GPIO_INT1_EXTERNAL_ISPUInitParams, &MX_GPIO_CS_EXTERNALInitParams);
184 }
185 else
186 {
187 /* Use the onboard IIS3DWB and ISM330DHCX */
188 sISM330DHCXObj = ISM330DHCXTaskAlloc(&MX_GPIO_INT1_DHCXInitParams, &MX_GPIO_INT2_DHCXInitParams, &MX_GPIO_CS_DHCXInitParams);
189 }
190
191 /* Use the onboard STTS22H (address low) */
192 sSTTS22HObj = STTS22HTaskAlloc(NULL, NULL, STTS22H_I2C_ADD_L);
193 if (ext_stts22h)
194 {
195 /* Use the external STTS22H on I2C3 */
196 sSTTS22HExtObj = STTS22HTaskAllocSetName(NULL, NULL, stts22h_address, "stts22h_ext");
197 sI2C3BusObj = I2CBusTaskAlloc(&MX_I2C3InitParams);
198 }
199
200 sILPS22QSObj = ILPS22QSTaskAlloc(NULL, NULL);
201 sIMP23ABSUObj = IMP23ABSUTaskAlloc(&MX_MDF1InitParams, &MX_ADC1InitParams);
202 sIMP34DT05Obj = IMP34DT05TaskAlloc(&MX_ADF1InitParams);
203

210 res = ACAddTask(pAppContext, (AManagedTask *) sIIS2DLPCObj);
211 res = ACAddTask(pAppContext, (AManagedTask *) sIIS2ICLXObj);
212 res = ACAddTask(pAppContext, (AManagedTask *) sIIS2MDCObj);
213 res = ACAddTask(pAppContext, (AManagedTask *) sIIS3DWBObj);
214 res = ACAddTask(pAppContext, (AManagedTask *) sIIS3DWBExtObj);
215 res = ACAddTask(pAppContext, (AManagedTask *) sILPS22QSObj);
216 res = ACAddTask(pAppContext, (AManagedTask *) sIMP23ABSUObj);
217 res = ACAddTask(pAppContext, (AManagedTask *) sIMP34DT05Obj);
218 res = ACAddTask(pAppContext, (AManagedTask *) sSTTS22HObj);
219 res = ACAddTask(pAppContext, (AManagedTask *) sSTTS22HExtObj);
220 res = ACAddTask(pAppContext, (AManagedTask *) sISM330DHCXObj);
221 res = ACAddTask(pAppContext, (AManagedTask *) sISM330ISObj);

223 pIIS2DLPC_ACC_PnPLObj = Iis2dlpc_Acc_PnPLAlloc();
224 pIIS2ICLX_ACC_PnPLObj = Iis2iclx_Acc_PnPLAlloc();

234 pSTTS22H_Ext_TEMP_PnPLObj = Stts22h_Ext_Temp_PnPLAlloc();

239 pIIS3DWB_Ext_ACC_PnPLObj = Iis3dwb_Ext_Acc_PnPLAlloc();

241 if (sISM330DHCXObj)
