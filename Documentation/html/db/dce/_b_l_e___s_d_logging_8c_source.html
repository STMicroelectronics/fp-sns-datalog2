107tBleStatus BLE_SDLoggingUpdate(BLE_SDLog_Status_t Status, uint32_t FeatureMask, uint32_t TimeStep)
108{
109 tBleStatus ret;
110
111 uint8_t buff[2 + 9];
112
113 /* Time Stamp */
114 STORE_LE_16(buff, (HAL_GetTick() >> 3));
115 buff[2] = (uint8_t)Status;
116 STORE_LE_32(buff + 3, FeatureMask);
117 STORE_LE_32(buff + 7, TimeStep);
118
119 ret = ACI_GATT_UPDATE_CHAR_VALUE(&BleCharSDLog, 0, 2 + 9, buff);
120
121 if (ret != (tBleStatus)BLE_STATUS_SUCCESS)
122 {
123 if (BLE_StdErr_Service == BLE_SERV_ENABLE)
124 {
125 BytesToWrite = (uint8_t)sprintf((char *)BufferToWrite, "Error Updating SDLog Char\n");
126 Stderr_Update(BufferToWrite, BytesToWrite);
127 }
128 else
129 {
130 BLE_MANAGER_PRINTF("Error: Updating SDLog Char\r\n");
131 }
132 }
133 return ret;
134}
