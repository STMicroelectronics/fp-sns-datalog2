161sys_error_code_t ADPU2_vtblAttachToDataSource(IDPU2_t *_this, ISourceObservable *p_data_source, IDataBuilder_t *p_builder, IDB_BuildStrategy_e build_strategy)
162{
163 assert_param(_this != NULL);
164 assert_param(p_data_source != NULL);
165 assert_param(p_builder != NULL);
166 sys_error_code_t res = SYS_NO_ERROR_CODE;
167 ADPU2_t *p_obj = (ADPU2_t*)_this;
168
169 if (p_obj->is_chained_as_next)
170 {
171 /* a DPU can have n data sources or one DPU attached as input data source, but not to both.*/
172 res = SYS_INVALID_FUNC_CALL_ERROR_CODE;
173 SYS_SET_SERVICE_LEVEL_ERROR_CODE(SYS_INVALID_FUNC_CALL_ERROR_CODE);
174
175 SYS_DEBUGF(SYS_DBG_LEVEL_WARNING, ("ADPU2: error DPU chained, in AttachToDataSource!\r\n"));
176 }
177
178 if (!SYS_IS_ERROR_CODE(res))
179 {
180 AttachedSourceObservedItem_t *p_list_item = ASOListFindItem(&p_obj->attached_data_src_list, p_data_source);
181 if (p_list_item != NULL)
182 {
183 /* DPU is already attached to this data source*/
184 res = SYS_ADPU2_DATA_SRC_ALREADY_ATTACHED;
185 SYS_SET_SERVICE_LEVEL_ERROR_CODE(SYS_ADPU2_DATA_SRC_ALREADY_ATTACHED);
186 }
187 else
188 {
189 p_list_item = SysAlloc(sizeof(AttachedSourceObservedItem_t));
190 if (p_list_item != NULL)
191 {
192 p_list_item->p_data_source = p_data_source;
193 p_list_item->build_strategy = build_strategy;
194 p_list_item->p_builder = p_builder;
195 ASOListAddItem(&p_obj->attached_data_src_list, p_list_item);
196 }
197 else
198 {
199 res = SYS_OUT_OF_MEMORY_ERROR_CODE;
200 SYS_SET_SERVICE_LEVEL_ERROR_CODE(SYS_OUT_OF_MEMORY_ERROR_CODE);
201 }
202 }
203 }
204
205 if (!SYS_IS_ERROR_CODE(res))
206 {
207 /* register the DPU as a listener of the data source.*/
208 IEventSrc *p_event_src = ISourceGetEventSrcIF(p_data_source);
209 res = IEventSrcAddEventListener(p_event_src, ADPU2_GetEventListenerIF(p_obj));
210 if (SYS_IS_ERROR_CODE(res))
211 {
212 sys_error_handler();
213 }
214 }
215
216 return res;
217}

219sys_error_code_t ADPU2_vtblDetachFromDataSource(IDPU2_t *_this, ISourceObservable *p_data_source, IDataBuilder_t **p_data_builder)
220{
221 assert_param(_this != NULL);
222 assert_param(p_data_source != NULL);
223 sys_error_code_t res = SYS_NO_ERROR_CODE;
224 ADPU2_t *p_obj = (ADPU2_t*)_this;
225
226 /*check if the data source is attached to this DPU*/
227 AttachedSourceObservedItem_t *p_item = ASOListFindItem(&p_obj->attached_data_src_list, p_data_source);
228 if (p_item != NULL)
229 {
230 /* remove the item from the list.*/
231 ASOListRemoveItem(&p_obj->attached_data_src_list, p_item);
232
233 /* remove the DPU as listener from the data source.*/
234 IEventSrc *p_evenr_src = ISourceGetEventSrcIF(p_data_source);
235 res = IEventSrcRemoveEventListener(p_evenr_src, ADPU2_GetEventListenerIF(p_obj));
236 if(SYS_IS_ERROR_CODE(res))
237 {
238 sys_error_handler();
239 }
240
241 if (p_data_builder != NULL)
242 {
243 *p_data_builder = p_item->p_builder;
244 }
245
246 /*release the memory of the item*/
247 SysFree(p_item);
248 }
249 else
250 {
251 if (p_data_builder != NULL){
252 *p_data_builder = NULL;
253 }
254 }
255
256 return res;
257}

259sys_error_code_t ADPU2_vtblAttachToDPU(IDPU2_t *_this, IDPU2_t *p_next_dpu, IDataBuilder_t *p_builder, IDB_BuildStrategy_e build_strategy)
260{

273 }
274 else

276 p_obj->next_dpu.p_next = (ADPU2_t*)p_next_dpu;
277 p_obj->next_dpu.build_strategy = build_strategy;
278 p_obj->next_dpu.p_builder = p_builder;
279 p_obj->next_dpu.p_next->is_chained_as_next = true;

311sys_error_code_t ADPU2_vtblDispatchEvents(IDPU2_t *_this, DataEvent_t *p_evt)
312{
313 assert_param(_this != NULL);
314 assert_param(p_evt != NULL);
315 sys_error_code_t res = SYS_NO_ERROR_CODE;
316 ADPU2_t *p_obj = (ADPU2_t*)_this;
317
318 /* first notify the listeners.*/
319 res = IEventSrcSendEvent((IEventSrc *)&p_obj->data_evt_src_if, (IEvent *) p_evt, NULL);
320 if(SYS_IS_ERROR_CODE(res))
321 {
322 SYS_SET_SERVICE_LEVEL_ERROR_CODE(SYS_UNDEFINED_ERROR_CODE);
323
324 SYS_DEBUGF(SYS_DBG_LEVEL_WARNING, ("ADPU2: error during evt dispatching!\r\n"));
325 }
326 else
327 {
328 /* then propagate the data into DPU2 chain*/
329 if(p_obj->next_dpu.p_next != NULL)
330 {
331 res = ADPU2_OnNewInputDataFromDPU(p_obj->next_dpu.p_next, p_evt, p_obj);
332
333 if (res == SYS_IDB_DATA_READY_ERROR_CODE)
334 {
335 res = SYS_NO_ERROR_CODE;
336 }
337 }
338
339 if(SYS_IS_ERROR_CODE(res))
340 {
341 SYS_SET_SERVICE_LEVEL_ERROR_CODE(SYS_UNDEFINED_ERROR_CODE);
342
343 SYS_DEBUGF(SYS_DBG_LEVEL_WARNING, ("ADPU2: error during DPU chaining!\r\n"));
344 }
345 }
346
347 return res;
348}

401sys_error_code_t ADPU2_vtblOnNewDataReady(IEventListener *_this, const DataEvent_t *p_evt)
402{
403 assert_param(_this != NULL);
404 assert_param(p_evt != NULL);
405 sys_error_code_t res = SYS_NO_ERROR_CODE;
406 ADPU2_t* p_obj = (ADPU2_t*) ((uint32_t) _this - offsetof (ADPU2_t , data_evt_listener_if));
407 SYS_DECLARE_CS(cs);
408
409 if (p_obj->active)
410 {
411 /* search for the data source.*/
412 AttachedSourceObservedItem_t *p_aso_item = ASOListFindItemBySrcID(&p_obj->attached_data_src_list, p_evt->tag);
413
414 /* Check if we are start building a new data.*/
415 SYS_ENTER_CRITICAL(cs);
416 if (p_obj->cbh.p_producer_data_buff == NULL)
417 {
418 SYS_EXIT_CRITICAL(cs);
419 /* we start to build a new data.*/
420 res = ADPU2_PrepareToBuildNewData(p_obj, NULL);
421 if (SYS_IS_ERROR_CODE(res))
422 {
423 /* no free cb item or there is a data source without a data builder. Check the build strategy...*/
424 if (p_aso_item->build_strategy != E_IDB_SKIP_DATA)
425 {
426 /* unable to build the IN data.*/
427 SYS_DEBUGF(SYS_DBG_LEVEL_SEVERE, ("DPU: error: unable to build IN data: data lost.\r\n"));
428
429 sys_error_handler();
430 }
431 else
432 {
433 /* build_strategy == E_IDB_SKIP_DATA */
434 return SYS_NO_ERROR_CODE;
435 }
436 }
437 }
438 else
439 {
440 SYS_EXIT_CRITICAL(cs);
441 }
442
443 res = IDataBuilder_OnNewInData(p_aso_item->p_builder, &p_obj->in_data, p_evt->p_data, p_aso_item->build_strategy, ADPU2_DataBuffAlloc);
444
445 if (res == SYS_IDB_DATA_READY_ERROR_CODE)
446 {
447 if (!--p_obj->data_builder_to_complete)
448 {
449 /*a new data is ready*/
450 CB_SetItemReady(p_obj->cbh.p_cb, p_obj->cbh.p_producer_data_buff);
451 SYS_ENTER_CRITICAL(cs);
452 p_obj->cbh.p_producer_data_buff = NULL;
453 SYS_EXIT_CRITICAL(cs);
454 if(p_obj->notify_data_ready_f)
455 {
456 /* I do not process inline the new data, but I notify the app.
457 * It will be responsibility of the app to call ADPU2_ProcessAndDispatch or manually
458 * do the Process&Dispatch */
459 p_obj->notify_data_ready_f((IDPU2_t*)p_obj, p_obj->p_data_ready_callback_param);
460 }
461 else
462 {
463 res = ADPU2_ProcessAndDispatch(p_obj);
464 }
465
466 SYS_DEBUGF(SYS_DBG_LEVEL_ALL, ("ADPU2: new data ready\r\n"));
467 }
468 }
469 }
470
471 return res;
472}

478sys_error_code_t ADPU2_Init(ADPU2_t *_this, EMData_t in_data, EMData_t out_data)
479{
480 assert_param(_this);
481 ADPU2_t *p_obj = (ADPU2_t*)_this;
482 sys_error_code_t res = SYS_NO_ERROR_CODE;
483
484 /* The DPU vtbl has been initialized during the allocation of the derived DPU,
485 * so we have to initialize only the listener IF
486 */
487 p_obj->data_evt_listener_if.vptr = &sTheClass.if_data_evt_listener_vtbl;
488 IEventListenerSetOwner((IEventListener*)&p_obj->data_evt_listener_if, p_obj);
489
490 /* Initialize the data event source IF*/
491 (void)DataEventSrcAllocStatic(&p_obj->data_evt_src_if);
492 (void)IEventSrcInit((IEventSrc*)&p_obj->data_evt_src_if);
493
494 /* Initialize the list of attached data source*/
495 (void)ASOItemInit(&p_obj->attached_data_src_list);
496
497 p_obj->active = true;
498 p_obj->notify_data_ready_f = NULL;
499 p_obj->p_data_ready_callback_param = NULL;
500 p_obj->tag = 0U;
501 p_obj->data_builder_to_complete = 0;
502 p_obj->in_data = in_data;
503 p_obj->out_data = out_data;
504 p_obj->cbh.p_cb = NULL;
505 p_obj->cbh.p_producer_data_buff = NULL;
506 p_obj->next_dpu.p_next = NULL;
507 p_obj->next_dpu.p_builder = NULL;
508 p_obj->is_chained_as_next = false;
509
510 return res;
511}

543sys_error_code_t ADPU2_Reset(ADPU2_t *_this)
544{
545 assert_param(_this != NULL);
546 sys_error_code_t res = SYS_NO_ERROR_CODE;
547
548 /*check if the DPU has already a cb.*/
549 if (_this->cbh.p_cb != NULL)
550 {
551 /* reset the Circular Buffer handler.*/
552 (void)CB_Init(_this->cbh.p_cb, CB_GetItemsBuffer(_this->cbh.p_cb), CB_GetItemSize(_this->cbh.p_cb));
553 _this->cbh.p_producer_data_buff = NULL;
554 /*the above code line will trigger a data builders reset when the DPU receives new data,*/
555 /*so we do not need to reset the data builder here explicitly.*/
556 }
557
558 /*check if the DPU has another DPU chained as next.*/
559 if (_this->next_dpu.p_next != NULL)
560 {
561 /*reset the next DPU handler.*/
562 (void)IDataBuilder_Reset(_this->next_dpu.p_builder, _this);
563
564 //TODO: STF - shall I reset all the DPU in the chain ?
565// ADPU2_Reset(_this->next_dpu.p_next);
566 }
567
568 return res;
569}

591sys_error_code_t ADPU2_SetInDataBuffer(ADPU2_t *_this, uint8_t *p_buffer, uint32_t buffer_size)
592{
593 assert_param(_this != NULL);
594 assert_param((p_buffer != NULL) || (buffer_size == 0));
595 sys_error_code_t res = SYS_NO_ERROR_CODE;
596
597 /*check if the DPU has already a cb.*/
598 if (_this->cbh.p_cb != NULL)
599 {
600 /*release the cb*/
601 CB_Free(_this->cbh.p_cb);
602 _this->cbh.p_cb = NULL;
603 _this->cbh.p_producer_data_buff = NULL;
604 }
605
606 if (buffer_size > 0)
607 {
608 size_t payload_size = EMD_GetPayloadSize(&_this->in_data);
609 uint16_t cb_items = buffer_size / payload_size;
610 _this->cbh.p_cb = CB_Alloc(cb_items);
611 if (_this->cbh.p_cb != NULL)
612 {
613 (void)CB_Init(_this->cbh.p_cb, p_buffer, payload_size);
614 }
615 else
616 {
617 res = SYS_OUT_OF_MEMORY_ERROR_CODE;
618 SYS_SET_SERVICE_LEVEL_ERROR_CODE(res);
619 }
620 }
621
622 return res;
623}

653sys_error_code_t ADPU2_ProcessAndDispatch(ADPU2_t *_this)
654{
655 assert_param(_this != NULL);
656 sys_error_code_t res = SYS_NO_ERROR_CODE;
657
658 CBItem *p_ready_item;
659 CB_GetReadyItemFromTail(_this->cbh.p_cb, &p_ready_item);
660 if (p_ready_item == NULL)
661 {
662 res = SYS_ADPU2_NO_READY_ITEM_ERROR_CODE;
663 SYS_SET_SERVICE_LEVEL_ERROR_CODE(SYS_ADPU2_NO_READY_ITEM_ERROR_CODE);
664 return res;
665 }
666
667 EMData_t in_data = _this->in_data;
668 in_data.p_payload = CB_GetItemData(p_ready_item);
669 res = IDPU2_Process((IDPU2_t*)_this, in_data, _this->out_data);
670 CB_ReleaseItem(_this->cbh.p_cb, p_ready_item);
671 double timestamp = SysTsGetTimestampF(SysGetTimestampSrv());
672 if (!SYS_IS_ERROR_CODE(res))
673 {
674 DataEvent_t data_evt;
675 DataEventInit((IEvent*)&data_evt, (IEventSrc*)&_this->data_evt_src_if, &_this->out_data, timestamp, _this->tag);
676 res = IDPU2_DispatchEvents((IDPU2_t*)_this, &data_evt);
677 }
678
679 return res;
680}

704static AttachedSourceObservedItem_t *ASOListFindItemByDataBuilder(AttachedSourceObservedItem_t *p_list_head, IDataBuilder_t *p_data_builder)
705{
706 assert_param(p_data_builder != NULL);
707
708 AttachedSourceObservedItem_t *p_tmp = p_list_head->p_next;
709 bool found = false;
710 while (!found && (p_tmp != NULL))
711 {
712 if (p_tmp->p_builder == p_data_builder)
713 {
714 /* data source is already in the list of attached data source.*/
715 found = true;
716 }
717 else
718 {
719 /*move to the next list item.*/
720 p_tmp = p_tmp->p_next;
721 }
722 }
723
724 return p_tmp;
725}

727static AttachedSourceObservedItem_t *ASOListFindItemBySrcID(AttachedSourceObservedItem_t *p_list_head, uint16_t data_src_id)
728{
729 assert_param(p_list_head != NULL);
730
731 AttachedSourceObservedItem_t *p_tmp = p_list_head->p_next;
732 bool found = false;
733 while (!found && (p_tmp != NULL))
734 {
735 if (ISourceGetId(p_tmp->p_data_source) == data_src_id)
736 {
737 /* data source is already in the list of attached data source.*/
738 found = true;
739 }
740 else
741 {
742 /*move to the next list item.*/
743 p_tmp = p_tmp->p_next;

787static sys_error_code_t ADPU2_PrepareToBuildNewData(ADPU2_t *_this, AttachedSourceObservedItem_t *p_no_reset_item)
788{
789 assert_param(_this != NULL);
790 assert_param(_this->cbh.p_cb != NULL);
791 sys_error_code_t res = SYS_NO_ERROR_CODE;
792 _this->data_builder_to_complete = (p_no_reset_item == NULL ? 0U : 1U);
793
794 /*get a new empty buffer from the circular buffer.*/
795 res = (sys_error_code_t)CB_GetFreeItemFromHead(_this->cbh.p_cb, &_this->cbh.p_producer_data_buff);
796 if (!SYS_IS_ERROR_CODE(res))
797 {
798 _this->in_data.p_payload = (uint8_t*)CB_GetItemData(_this->cbh.p_producer_data_buff);
799
800 /* reset the data builder interface*/
801 AttachedSourceObservedItem_t *p_tmp = _this->attached_data_src_list.p_next;
802
803 while(p_tmp != NULL)
804 {
805 if (p_tmp->p_builder == NULL)
806 {
807 /* all source must have a data builder! */
808 sys_error_handler();
809 }
810 if (p_tmp != p_no_reset_item)
811 {
812 (void)IDataBuilder_Reset(p_tmp->p_builder, _this);
813 _this->data_builder_to_complete++;
814 }
815 p_tmp = p_tmp->p_next;
816 }
817 }
818
819 return res;
820}

822static uint8_t *ADPU2_DataBuffAlloc(IDataBuilder_t *_this, void *p_data_build_context)
823{
824 assert_param(_this != NULL);
825 assert_param(p_data_build_context != NULL);
826 ADPU2_t *p_obj = (ADPU2_t*)p_data_build_context;
827 uint8_t *p_buff = NULL;
828 sys_error_code_t res = SYS_NO_ERROR_CODE;
829 SYS_DECLARE_CS(cs);
830
831 /*a data builder needs a new buffer because it has completed its part of the data.*/
832
833 if (p_obj->data_builder_to_complete == 1U)
834 {
835 /* it is the last data builder so I can mark the data ready and allocate a new buffer.*/
836 SYS_DEBUGF(SYS_DBG_LEVEL_ALL, ("ADPU2: new data ready _\r\n"));
837
838 CB_SetItemReady(p_obj->cbh.p_cb, p_obj->cbh.p_producer_data_buff);
839 SYS_ENTER_CRITICAL(cs);
840 p_obj->cbh.p_producer_data_buff = NULL;
841 SYS_EXIT_CRITICAL(cs);
842 /* find the attached source observer list item*/
843 AttachedSourceObservedItem_t *p_aso_item = ASOListFindItemByDataBuilder(&p_obj->attached_data_src_list, _this);
844 res = ADPU2_PrepareToBuildNewData(p_obj, p_aso_item);
845 if (!SYS_IS_ERROR_CODE(res))
846 {
847 p_buff = p_obj->in_data.p_payload;
848 }
849
850 /* */
851 if(p_obj->notify_data_ready_f)
852 {
853 /* I do not process inline the new data, but I notify the app.
854 * It will be responsibility of the app to call ADPU2_ProcessAndDispatch or manually
855 * do the Process&Dispatch */
856 p_obj->notify_data_ready_f((IDPU2_t*)p_obj, p_obj->p_data_ready_callback_param);
857 }
858 else
859 {
860 res = ADPU2_ProcessAndDispatch(p_obj);
861 }
862 }
863
864 return p_buff;
865}

867static sys_error_code_t ADPU2_OnNewInputDataFromDPU(ADPU2_t *_this, DataEvent_t *p_evt, ADPU2_t *p_src_dpu)
868{
869 assert_param(_this != NULL);
870 assert_param(p_evt != NULL);
871 assert_param(p_src_dpu != NULL);
872 sys_error_code_t res = SYS_NO_ERROR_CODE;
873 SYS_DECLARE_CS(cs);
874
875 if (_this->active)
876 {
877 /* Check if we are start building a new data.*/
878 SYS_ENTER_CRITICAL(cs);
879 if (_this->cbh.p_producer_data_buff == NULL)
880 {
881 SYS_EXIT_CRITICAL(cs);
882 /*get a new empty buffer from the circular buffer.*/
883 res = (sys_error_code_t)CB_GetFreeItemFromHead(_this->cbh.p_cb, &_this->cbh.p_producer_data_buff);
884 if (!SYS_IS_ERROR_CODE(res))
885 {
886 _this->in_data.p_payload = (uint8_t*)CB_GetItemData(_this->cbh.p_producer_data_buff);
887
888 /* reset the data builder interface*/
889 (void)IDataBuilder_Reset(p_src_dpu->next_dpu.p_builder, _this);
890 }
891 if (_this->in_data.p_payload == NULL)
892 {
893 /* no free cb item or there is a data source without a data builder. Check the build strategy...*/
894 if (p_src_dpu->next_dpu.build_strategy != E_IDB_SKIP_DATA)
895 {
896 /* unable to build the IN data.*/
897 SYS_DEBUGF(SYS_DBG_LEVEL_SEVERE, ("DPU: error: unable to build IN data: data lost.\r\n"));
898
899 sys_error_handler();
900 }
901 else
902 {
903 /* build_strategy == E_IDB_SKIP_DATA */
904 return SYS_NO_ERROR_CODE;
905 }
906 }
907 }
908 else
909 {
910 SYS_EXIT_CRITICAL(cs);
911 }
912
913 res = IDataBuilder_OnNewInData(p_src_dpu->next_dpu.p_builder, &_this->in_data, p_evt->p_data, p_src_dpu->next_dpu.build_strategy, ADPU2_DataBuffAlloc);
914
915 if (res == SYS_IDB_DATA_READY_ERROR_CODE)
916 {
917 /*a new data is ready*/
918 CB_SetItemReady(_this->cbh.p_cb, _this->cbh.p_producer_data_buff);
919 SYS_ENTER_CRITICAL(cs);
920 _this->cbh.p_producer_data_buff = NULL;
921 SYS_EXIT_CRITICAL(cs);
922 if(_this->notify_data_ready_f)
923 {
924 /* I do not process inline the new data, but I notify the app.
925 * It will be responsibility of the app to call ADPU2_ProcessAndDispatch or manually
926 * do the Process&Dispatch */
927 _this->notify_data_ready_f((IDPU2_t*)_this, _this->p_data_ready_callback_param);
928 }
929 else
930 {
931 res = ADPU2_ProcessAndDispatch(_this);
932 }
933
934 SYS_DEBUGF(SYS_DBG_LEVEL_ALL, ("ADPU2: new data ready\r\n"));
935 }
936 }
937
938 return res;
939}
