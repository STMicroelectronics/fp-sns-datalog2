340static void BLE_SendCommand(char *buf, uint32_t size)
341{
342 if (buf != NULL)
343 {
344 tBleStatus ret;
345 uint32_t j = 0, chunk, tot_len;
346 uint8_t *buffer_out;
347 uint32_t length_wTP;
348
349 if ((size % 19U) == 0U)
350 {
351 length_wTP = (size / 19U) + size;
352 }
353 else
354 {
355 length_wTP = (size / 19U) + 1U + size;
356 }
357
358 buffer_out = BLE_MallocFunction(sizeof(uint8_t) * length_wTP);
359
360 if (buffer_out == NULL)
361 {
362 BLE_MANAGER_PRINTF("Error: Mem calloc error [%ld]: %d@%s\r\n", length_wTP, __LINE__, __FILE__);
363 // TODO: manage error
364 return;
365 }
366 else
367 {
368 tot_len = BLE_Command_TP_Encapsulate(buffer_out, (uint8_t *) buf, size, 20);
369
370 j = 0;
371
372 /* Data are sent as notifications*/
373 while (j < tot_len)
374 {
375 /* TODO: different MTU must be managed with MaxBLECharLen */
376
377 chunk = MIN(20, tot_len - j);
378
379 ret = BLE_PnPLikeUpdate(&buffer_out[j], chunk);
380
381 if (ret == BLE_STATUS_INSUFFICIENT_RESOURCES)
382 {
383 ble_sendMSG_wait();
384 }
385 else if (ret == BLE_STATUS_SUCCESS)
386 {
387 j += chunk;
388 }
389 }
390 }
391
392 BLE_FreeFunction(buffer_out);
393 }
394}

403static uint32_t DebugConsoleParsing(uint8_t *att_data, uint8_t data_length)
404{
405 /* By default Answer with the same message received */
406 uint32_t SendBackData = 1;
407
408 if (SizeOfUpdateBlueFW != 0)
409 {
410 /* Firmware update */
411 int8_t RetValue = UpdateFWBlueMS(&SizeOfUpdateBlueFW, att_data, data_length, 1);
412 if (RetValue != 0)
413 {
414 Term_Update(((uint8_t *)&RetValue), 1);
415 if (RetValue == 1)
416 {
417 /* if OTA checked */
418 PRINT_DBG("%s will restart after the disconnection\r\n", BLE_FW_PACKAGENAME);
419 HAL_Delay(1000);
420 NeedToSwapBanks = 1;
421
422 /* Swap the Flash Banks */
423 SwitchBank();
424 }
425 }
426 SendBackData = 0;
427 }
428 else
429 {
430 /* Received one write from Client on Terminal characteristc */
431 SendBackData = DebugConsoleCommandParsing(att_data, data_length);
432 }
433
434 return SendBackData;
435}

444static uint32_t DebugConsoleCommandParsing(uint8_t *att_data, uint8_t data_length)
445{
446 uint32_t SendBackData = 1;
447
448 /* Help Command */
449 if (!strncmp("help", (char *)(att_data), 4))
450 {
451 /* Print Legend */
452 SendBackData = 0;
453
454 BytesToWrite = sprintf((char *)BufferToWrite,
455 "info\n"
456 "clearDB\n");
457 Term_Update(BufferToWrite, BytesToWrite);
458
459 }
460 else if (!strncmp("versionFw", (char *)(att_data), 9))
461 {
462 BytesToWrite = sprintf((char *)BufferToWrite, "%s_%s_%s.%s.%s\r\n",
463 "U585",
464 BLE_FW_PACKAGENAME,
465 FW_VERSION_MAJOR,
466 FW_VERSION_MINOR,
467 FW_VERSION_PATCH);
468 Term_Update(BufferToWrite, BytesToWrite);
469 SendBackData = 0;
470 }
471 else if (!strncmp("info", (char *)(att_data), 4))
472 {
473 SendBackData = 0;
474
475 BytesToWrite = sprintf((char *)BufferToWrite, "\r\nSTMicroelectronics %s:\n"
476 "\tVersion %s.%s.%s\n"
477 "\tSTM32U585AI-STWIN.box board"
478 "\n",
479 BLE_FW_PACKAGENAME,
480 FW_VERSION_MAJOR, FW_VERSION_MINOR, FW_VERSION_PATCH);
481 Term_Update(BufferToWrite, BytesToWrite);
482
483 BytesToWrite = sprintf((char *)BufferToWrite, "\t(HAL %ld.%ld.%ld_%ld)\n"
484 "\tCompiled %s %s"
485#if defined (__IAR_SYSTEMS_ICC__)
486 " (IAR)\n",
487#elif defined (__CC_ARM)
488 " (KEIL)\n",
489#elif defined (__GNUC__)
490 " (STM32CubeIDE)\n",
491#endif
492 (long)(HAL_GetHalVersion() >> 24),
493 (long)((HAL_GetHalVersion() >> 16) & 0xFF),
494 (long)((HAL_GetHalVersion() >> 8) & 0xFF),
495 (long)(HAL_GetHalVersion() & 0xFF),
496 __DATE__, __TIME__);
497 Term_Update(BufferToWrite, BytesToWrite);
498 BytesToWrite = sprintf((char *)BufferToWrite, "Current Bank =%d\n", CurrentActiveBank);
499 Term_Update(BufferToWrite, BytesToWrite);
500 }
501 else if (!strncmp("upgradeFw", (char *)(att_data), 9))
502 {
503 uint32_t uwCRCValue;
504 uint8_t *PointerByte = (uint8_t *) &SizeOfUpdateBlueFW;
505
506 PointerByte[0] = att_data[ 9];
507 PointerByte[1] = att_data[10];
508 PointerByte[2] = att_data[11];
509 PointerByte[3] = att_data[12];
510
511 /* Check the Maximum Possible OTA size */
512 if (SizeOfUpdateBlueFW > OTA_MAX_PROG_SIZE)
513 {
514 PRINT_DBG("OTA %s SIZE=%ld > %d Max Allowed\r\n", BLE_FW_PACKAGENAME, SizeOfUpdateBlueFW, OTA_MAX_PROG_SIZE);
515 /* Answer with a wrong CRC value for signaling the problem to BlueMS application */
516 BufferToWrite[0] = att_data[13];
517 BufferToWrite[1] = (att_data[14] != 0) ? 0 : 1; /* In order to be sure to have a wrong CRC */
518 BufferToWrite[2] = att_data[15];
519 BufferToWrite[3] = att_data[16];
520 BytesToWrite = 4;
521 Term_Update(BufferToWrite, BytesToWrite);
522 }
523 else
524 {
525 PointerByte = (uint8_t *) &uwCRCValue;
526 PointerByte[0] = att_data[13];
527 PointerByte[1] = att_data[14];
528 PointerByte[2] = att_data[15];
529 PointerByte[3] = att_data[16];
530
531 PRINT_DBG("OTA %s SIZE=%ld uwCRCValue=%lx\r\n", BLE_FW_PACKAGENAME, SizeOfUpdateBlueFW, uwCRCValue);
532
533 /* Reset the Flash */
534 StartUpdateFWBlueMS(SizeOfUpdateBlueFW, uwCRCValue);
535
536 /* Signal that we are ready sending back the CRV value*/
537 BufferToWrite[0] = PointerByte[0];
538 BufferToWrite[1] = PointerByte[1];
539 BufferToWrite[2] = PointerByte[2];
540 BufferToWrite[3] = PointerByte[3];
541 BytesToWrite = 4;
542 Term_Update(BufferToWrite, BytesToWrite);
543 }
544
545 SendBackData = 0;
546 }
547 else if (!strncmp("uid", (char *)(att_data), 3))
548 {
549 /* Write back the STM32 UID */
550 uint8_t *uid = (uint8_t *)STM32_UUID;
551 uint32_t MCU_ID = STM32_MCU_ID[0] & 0xFFF;
552 BytesToWrite = sprintf((char *)BufferToWrite, "%.2X%.2X%.2X%.2X%.2X%.2X%.2X%.2X%.2X%.2X%.2X%.2X_%.3lX\n",
553 uid[ 3], uid[ 2], uid[ 1], uid[ 0],
554 uid[ 7], uid[ 6], uid[ 5], uid[ 4],
555 uid[11], uid[ 10], uid[9], uid[8],
556 (unsigned long)MCU_ID);
557 Term_Update(BufferToWrite, BytesToWrite);
558 SendBackData = 0;
559 }
560 else if (!strncmp("clearDB", (char *)(att_data), 7))
561 {
562 BytesToWrite = sprintf((char *)BufferToWrite, "\nThe Secure database will be cleared\n");
563 Term_Update(BufferToWrite, BytesToWrite);
564 NeedToClearSecureDB = 1;
565 SendBackData = 0;
566 }
567
568 return SendBackData;
569}

606{

608 set_connectable = TRUE;
