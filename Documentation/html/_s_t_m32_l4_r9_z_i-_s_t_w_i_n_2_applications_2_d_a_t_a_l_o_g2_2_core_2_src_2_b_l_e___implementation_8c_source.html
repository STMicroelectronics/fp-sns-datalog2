313static void BLE_SendCommand(char *buf, uint32_t size)
314{
315 if (buf != NULL)
316 {
317 tBleStatus ret;
318 uint32_t j = 0, chunk, tot_len;
319 uint8_t *buffer_out;
320 uint32_t length_wTP;
321
322 if ((size % 19U) == 0U)
323 {
324 length_wTP = (size / 19U) + size;
325 }
326 else
327 {
328 length_wTP = (size / 19U) + 1U + size;
329 }
330
331 buffer_out = BLE_MallocFunction(sizeof(uint8_t) * length_wTP);
332
333 if (buffer_out == NULL)
334 {
335 BLE_MANAGER_PRINTF("Error: Mem calloc error [%ld]: %d@%s\r\n", length_wTP, __LINE__, __FILE__);
336 // TODO: manage error
337 return;
338 }
339 else
340 {
341 tot_len = BLE_Command_TP_Encapsulate(buffer_out, (uint8_t *) buf, size, 20);
342
343 j = 0;
344
345 /* Data are sent as notifications*/
346 while (j < tot_len)
347 {
348 /* TODO: different MTU must be managed with MaxBLECharLen */
349
350 chunk = MIN(20, tot_len - j);
351
352 ret = BLE_PnPLikeUpdate(&buffer_out[j], chunk);
353
354 if (ret == BLE_STATUS_INSUFFICIENT_RESOURCES)
355 {
356 ble_sendMSG_wait();
357 }
358 else if (ret == BLE_STATUS_SUCCESS)
359 {
360 j += chunk;
361 }
362 }
363 }
364
365 BLE_FreeFunction(buffer_out);
366 }
367}

376static uint32_t DebugConsoleParsing(uint8_t *att_data, uint8_t data_length)
377{
378 /* By default Answer with the same message received */
379 uint32_t SendBackData = 1;
380
381 if (SizeOfUpdateBlueFW != 0)
382 {
383 /* Firmware update */
384 int8_t RetValue = UpdateFWBlueMS(&SizeOfUpdateBlueFW, att_data, data_length, 1);
385 if (RetValue != 0)
386 {
387 Term_Update(((uint8_t *)&RetValue), 1);
388 if (RetValue == 1)
389 {
390 /* if OTA checked */
391 PRINT_DBG("%s will restart after the disconnection\r\n", BLE_FW_PACKAGENAME);
392 HAL_Delay(1000);
393 NeedToSwapBanks = 1;
394
395 /* Swap the Flash Banks */
396 SwitchBank();
397 }
398 }
399 SendBackData = 0;
400 }
401 else
402 {
403 /* Received one write from Client on Terminal characteristc */
404 SendBackData = DebugConsoleCommandParsing(att_data, data_length);
405 }
406
407 return SendBackData;
408}

418{
419 uint32_t SendBackData = 1;
420
421 /* Help Command */

442 SendBackData = 0;
443 }
444 else if (!strncmp("info", (char *)(att_data), 4))
445 {
446 SendBackData = 0;
447
448 BytesToWrite = sprintf((char *)BufferToWrite, "\r\nSTMicroelectronics %s:\n"
449 "\tVersion %s.%s.%s\n"
450 "\tSTM32L4R9ZI-STWIN board"
451 "\n",
452 BLE_FW_PACKAGENAME,
453 FW_VERSION_MAJOR, FW_VERSION_MINOR, FW_VERSION_PATCH);
454 Term_Update(BufferToWrite, BytesToWrite);

456 BytesToWrite = sprintf((char *)BufferToWrite, "\t(HAL %ld.%ld.%ld_%ld)\n"
457 "\tCompiled %s %s"

459 " (IAR)\n",
460#elif defined (__CC_ARM)
461 " (KEIL)\n",
462#elif defined (__GNUC__)
463 " (STM32CubeIDE)\n",
464#endif
465 (long)(HAL_GetHalVersion() >> 24),
466 (long)((HAL_GetHalVersion() >> 16) & 0xFF),
467 (long)((HAL_GetHalVersion() >> 8) & 0xFF),
468 (long)(HAL_GetHalVersion() & 0xFF),
469 __DATE__, __TIME__);
470 Term_Update(BufferToWrite, BytesToWrite);
471 BytesToWrite = sprintf((char *)BufferToWrite, "Current Bank =%d\n", CurrentActiveBank);
472 Term_Update(BufferToWrite, BytesToWrite);
473 }
474 else if (!strncmp("upgradeFw", (char *)(att_data), 9))
475 {
476 uint32_t uwCRCValue;
477 uint8_t *PointerByte = (uint8_t *) &SizeOfUpdateBlueFW;
478
479 PointerByte[0] = att_data[ 9];
480 PointerByte[1] = att_data[10];
481 PointerByte[2] = att_data[11];
482 PointerByte[3] = att_data[12];
483
484 /* Check the Maximum Possible OTA size */
485 if (SizeOfUpdateBlueFW > OTA_MAX_PROG_SIZE)
486 {
487 PRINT_DBG("OTA %s SIZE=%ld > %d Max Allowed\r\n", BLE_FW_PACKAGENAME, SizeOfUpdateBlueFW, OTA_MAX_PROG_SIZE);
488 /* Answer with a wrong CRC value for signaling the problem to BlueMS application */
489 BufferToWrite[0] = att_data[13];
490 BufferToWrite[1] = (att_data[14] != 0) ? 0 : 1; /* In order to be sure to have a wrong CRC */
491 BufferToWrite[2] = att_data[15];
492 BufferToWrite[3] = att_data[16];
493 BytesToWrite = 4;
494 Term_Update(BufferToWrite, BytesToWrite);
495 }
496 else
497 {
498 PointerByte = (uint8_t *) &uwCRCValue;
499 PointerByte[0] = att_data[13];
500 PointerByte[1] = att_data[14];
501 PointerByte[2] = att_data[15];
502 PointerByte[3] = att_data[16];
503
504 PRINT_DBG("OTA %s SIZE=%ld uwCRCValue=%lx\r\n", BLE_FW_PACKAGENAME, SizeOfUpdateBlueFW, uwCRCValue);
505
506 /* Reset the Flash */
507 StartUpdateFWBlueMS(SizeOfUpdateBlueFW, uwCRCValue);
508
509 /* Signal that we are ready sending back the CRV value*/
510 BufferToWrite[0] = PointerByte[0];
511 BufferToWrite[1] = PointerByte[1];
512 BufferToWrite[2] = PointerByte[2];
513 BufferToWrite[3] = PointerByte[3];
514 BytesToWrite = 4;
515 Term_Update(BufferToWrite, BytesToWrite);
516 }
517
518 SendBackData = 0;
519 }
520 else if (!strncmp("uid", (char *)(att_data), 3))
521 {
522 /* Write back the STM32 UID */
523 uint8_t *uid = (uint8_t *)STM32_UUID;
524 uint32_t MCU_ID = STM32_MCU_ID[0] & 0xFFF;
525 BytesToWrite = sprintf((char *)BufferToWrite, "%.2X%.2X%.2X%.2X%.2X%.2X%.2X%.2X%.2X%.2X%.2X%.2X_%.3lX\n",
526 uid[ 3], uid[ 2], uid[ 1], uid[ 0],
527 uid[ 7], uid[ 6], uid[ 5], uid[ 4],
528 uid[11], uid[ 10], uid[9], uid[8],
529 (unsigned long)MCU_ID);
530 Term_Update(BufferToWrite, BytesToWrite);
531 SendBackData = 0;
532 }
533 else if (!strncmp("clearDB", (char *)(att_data), 7))
534 {
535 BytesToWrite = sprintf((char *)BufferToWrite, "\nThe Secure database will be cleared\n");
536 Term_Update(BufferToWrite, BytesToWrite);
537 NeedToClearSecureDB = 1;
538 SendBackData = 0;
539 }
540
541 return SendBackData;
542}
