75int8_t UpdateFWBlueMS(uint32_t *SizeOfUpdate, uint8_t *att_data, int32_t data_length, uint8_t WriteMagicNum)
76{
77 int8_t ReturnValue = 0;
78 /* Save the Packed received */
79
80 if(data_length > (*SizeOfUpdate))
81 {
82 /* Too many bytes...Something wrong... necessity to send it again... */
83 OTA_PRINTF("OTA something wrong data_length=%ld RemSizeOfUpdate=%ld....\r\nPlease Try again\r\n",data_length,(*SizeOfUpdate));
84 ReturnValue = -1;
85 /* Reset for Restarting again */
86 *SizeOfUpdate = 0;
87 }
88 else
89 {
90 uint64_t ValueToWrite;
91 int32_t Counter;
92 //OTA_PRINTF("OTA chunck data_length=%ld RemSizeOfUpdate=%ld\r\n",data_length,(*SizeOfUpdate));
93 /* Save the received OTA packed ad save it to flash */
94 /* Unlock the Flash to enable the flash control register access *************/
95 HAL_FLASH_Unlock();
96
97 for(Counter = 0; Counter < data_length; Counter += 8)
98 {
99 memcpy((uint8_t*) &ValueToWrite, att_data + Counter, 8);
100
101 if(HAL_FLASH_Program(FLASH_TYPEPROGRAM_DOUBLEWORD, WritingAddress, ValueToWrite) == HAL_OK)
102 {
103 WritingAddress += 8;
104 }
105 else
106 {
107 /* Error occurred while writing data in Flash memory.
108 User can add here some code to deal with this error
109 FLASH_ErrorTypeDef errorcode = HAL_FLASH_GetError(); */
110 OTA_ERROR_FUNCTION();
111 }
112 }
113 /* Reduce the remaing bytes for OTA completition */
114 *SizeOfUpdate -= data_length;
115
116 if(*SizeOfUpdate == 0)
117 {
118 /* We had received the whole firmware and we have saved it in Flash */
119 OTA_PRINTF("OTA Update saved\r\n");
120
121 if(WriteMagicNum)
122 {
123 uint32_t uwCRCValue = 0;
124
125 if(AspecteduwCRCValue)
126 {
127 /* Init CRC for OTA-integrity check */
128 CrcHandle.Instance = CRC;
129 /* The default polynomial is used */
130 CrcHandle.Init.DefaultPolynomialUse = DEFAULT_POLYNOMIAL_ENABLE;
131
132 /* The default init value is used */
133 CrcHandle.Init.DefaultInitValueUse = DEFAULT_INIT_VALUE_ENABLE;
134
135 /* The input data are not inverted */
136 CrcHandle.Init.InputDataInversionMode = CRC_INPUTDATA_INVERSION_NONE;
137
138 /* The output data are not inverted */
139 CrcHandle.Init.OutputDataInversionMode = CRC_OUTPUTDATA_INVERSION_DISABLE;
140
141 /* 32-bit CRC lengt */
142 CrcHandle.Init.CRCLength = CRC_POLYLENGTH_32B;
143
144 /* The input data are 32-bit long words */
145 CrcHandle.InputDataFormat = CRC_INPUTDATA_FORMAT_WORDS;
146
147 __HAL_RCC_CRC_CLK_ENABLE();
148
149 if(HAL_CRC_GetState(&CrcHandle) != HAL_CRC_STATE_RESET)
150 {
151 HAL_CRC_DeInit(&CrcHandle);
152 }
153
154 if(HAL_CRC_Init(&CrcHandle) != HAL_OK)
155 {
156 /* Initialization Error */
157 OTA_ERROR_FUNCTION();
158 }
159 else
160 {
161 OTA_PRINTF("CRC Initialized\n\r");
162 }
163 /* Compute the CRC */
164 uwCRCValue = HAL_CRC_Calculate(&CrcHandle, (uint32_t*) OTA_ADDRESS_START_BANK, SizeOfUpdateBlueFW >> 2);
165
166 if(uwCRCValue == AspecteduwCRCValue)
167 {
168 ReturnValue = 1;
169 OTA_PRINTF("OTA CRC-checked\r\n");
170 }
171 else
172 {
173 OTA_PRINTF("OTA Error CRC-checking\r\n");
174 }
175 }
176 else
177 {
178 ReturnValue = 1;
179
180 }
181 if(ReturnValue != 1)
182 {
183 ReturnValue = -1;
184 if(AspecteduwCRCValue)
185 {
186 OTA_PRINTF("Wrong CRC! Computed=%lx aspected=%lx ... Try again\r\n",uwCRCValue,AspecteduwCRCValue);
187 }
188 }
189 }
190 }
191
192 /* Lock the Flash to disable the flash control register access (recommended
193 to protect the FLASH memory against possible unwanted operation) *********/
194 HAL_FLASH_Lock();
195 }
196 return ReturnValue;
197}

205void StartUpdateFWBlueMS(uint32_t SizeOfUpdate, uint32_t uwCRCValue)
206{
207 FLASH_EraseInitTypeDef EraseInitStruct;
208 uint32_t SectorError = 0;
209 OTA_PRINTF("Start FLASH Erase\r\n");
210
211 SizeOfUpdateBlueFW = SizeOfUpdate;
212 AspecteduwCRCValue = uwCRCValue;
213
214 WritingAddress = OTA_ADDRESS_START_BANK;
215
216 if(CurrentActiveBank==1) {
217 EraseInitStruct.Banks = FLASH_BANK_2;
218 } else {
219 EraseInitStruct.Banks = FLASH_BANK_1;
220 }
221
222 EraseInitStruct.Page = 0;
223 EraseInitStruct.TypeErase = FLASH_TYPEERASE_PAGES;
224 EraseInitStruct.NbPages = (SizeOfUpdate+16+FLASH_PAGE_SIZE-1)/FLASH_PAGE_SIZE;
225
226 /* Unlock the Flash to enable the flash control register access *************/
227 HAL_FLASH_Unlock();
228
229 /* Clear OPTVERR bit set on virgin samples */
230 __HAL_FLASH_CLEAR_FLAG(FLASH_FLAG_OPTVERR);
231 /* Clear PEMPTY bit set (as the code is executed from Flash which is not empty) */
232 if (__HAL_FLASH_GET_FLAG(FLASH_FLAG_PEMPTY) != 0) {
233 __HAL_FLASH_CLEAR_FLAG(FLASH_FLAG_PEMPTY);
234 }
235
236 if(HAL_FLASHEx_Erase(&EraseInitStruct, &SectorError) != HAL_OK){
237 /* Error occurred while sector erase.
238 User can add here some code to deal with this error.
239 SectorError will contain the faulty sector and then to know the code error on this sector,
240 user can call function 'HAL_FLASH_GetError()'
241 FLASH_ErrorTypeDef errorcode = HAL_FLASH_GetError(); */
242 OTA_ERROR_FUNCTION();
243 } else {
244 OTA_PRINTF("End FLASH Erase %ld Pages of 4KB\r\n",EraseInitStruct.NbPages);
245 }
246
247 /* Lock the Flash to disable the flash control register access (recommended
248 to protect the FLASH memory against possible unwanted operation) *********/
249 HAL_FLASH_Lock();
250
251 /* Delete Fw Id saved on the bank where we will make the FOTA */
252 DeleteOtherFlashBankFwId();
253}

262void ReadFlashBanksFwId(uint16_t *FwId1,uint16_t *FwId2)
263{
264 uint32_t *First;
265 uint32_t *Second;
266
267 //First/Second identification
268 if(CurrentActiveBank==1) {
269 First = (uint32_t *)OTA_FW_ID_BANK1;
270 Second = (uint32_t *)OTA_FW_ID_BANK2;
271 } else {
272 First = (uint32_t *)OTA_FW_ID_BANK2;
273 Second = (uint32_t *)OTA_FW_ID_BANK1;
274 }
275
276 //Read First Bank
277 if(First[0]==OTA_FW_ID_MAGIC_NUM) {
278 *FwId1= First[1]&0xFFFF;
279 } else {
280 *FwId1= OTA_OTA_FW_ID_NOT_VALID;
281 }
282
283 //Read Second Bank
284 if(Second[0]==OTA_FW_ID_MAGIC_NUM) {
285 *FwId2= Second[1]&0xFFFF;
286 } else {
287 *FwId2= OTA_OTA_FW_ID_NOT_VALID;
288 }
289}
