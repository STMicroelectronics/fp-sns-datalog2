137static UINT _TransmitPacket(UX_SLAVE_CLASS_SENSOR_STREAMING *sensor_streaming, uint8_t epVectorIndex)
138{
139 UINT status;

143
144 if(ep_bulk->tx_state == 0U)

146 /* Tx Transfer in progress */
147 ep_bulk->tx_state = 1U;
