127sys_error_code_t SysLoadApplicationContext(ApplicationContext *pAppContext)
128{
129 assert_param(pAppContext != NULL);
130 sys_error_code_t res = SYS_NO_ERROR_CODE;
131
132 /* Allocate the task objects */
133 spHelloWorldObj = HelloWorldTaskAlloc(&MX_GPIO_LEDBlueInitParams, &MX_GPIO_UBInitParams);
134 spUsbCdcObj = UsbCdcTaskAlloc();
135 spI2C2BusObj = I2CBusTaskAlloc(&MX_I2C1InitParams);
136 spISM330DHCXObj = ISM330DHCXTaskAlloc(&MX_GPIO_INT1_DHCXInitParams, NULL, NULL);
137 spIIS2DLPCObj = IIS2DLPCTaskAlloc(&MX_GPIO_INT2_DLPCInitParams, NULL);
138 spIIS2MDCObj = IIS2MDCTaskAlloc(&MX_GPIO_DRDY_MDCInitParams, NULL);
139
140 /* allocate the process task. In this case we use the static allocation. In a embedded application
141 * the memory management is critical. The DProcessTask_t support both static and dynamic allocation. */
142 (void) DProcessTask1StaticAlloc(&sProcessTask);
143
144
145 /* Add the task object to the context. */
146 res = ACAddTask(pAppContext, spHelloWorldObj);
147 res = ACAddTask(pAppContext, (AManagedTask *) spUsbCdcObj);
148 res = ACAddTask(pAppContext, (AManagedTask *) spI2C2BusObj);
149 res = ACAddTask(pAppContext, (AManagedTask *) spISM330DHCXObj);
150 res = ACAddTask(pAppContext, (AManagedTask *) spIIS2DLPCObj);
151 res = ACAddTask(pAppContext, (AManagedTask *) spIIS2MDCObj);
152 res = ACAddTask(pAppContext, (AManagedTask *) &sProcessTask);
153
154 return res;
155}

157sys_error_code_t SysOnStartApplication(ApplicationContext *pAppContext)
158{
159 UNUSED(pAppContext);
160 sys_error_code_t res = SYS_NO_ERROR_CODE;
161
162 /* connect the sensor tasks to the bus. */
163 (void) I2CBusTaskConnectDevice((I2CBusTask *) spI2C2BusObj,
164 (I2CBusIF *)ISM330DHCXTaskGetSensorIF((ISM330DHCXTask *) spISM330DHCXObj));
165 (void) I2CBusTaskConnectDevice((I2CBusTask *) spI2C2BusObj,
166 (I2CBusIF *)IIS2DLPCTaskGetSensorIF((IIS2DLPCTask *) spIIS2DLPCObj));
167 (void) I2CBusTaskConnectDevice((I2CBusTask *) spI2C2BusObj,
168 (I2CBusIF *)IIS2MDCTaskGetSensorIF((IIS2MDCTask *) spIIS2MDCObj));
169
170 /*Select a sensor for the demo*/
171 uint16_t sensor_id = HelloWorldTaskSelectSensor((HelloWorldTask_t *)spHelloWorldObj);
172 /*The sensor is configured by the HelloWorkdTask::ConfigureSensors() before entering the task control loop*/
173
174 /*get the ISourceObservable IF of the sensor*/
175 ISourceObservable *p_data_src = SMGetSensorObserver(sensor_id);
176
177 /*Step 1: allocate and initialize the DPU.*/
178 /*this is a test DPU */
179 ADPU2_t *p_dpu = (ADPU2_t *) Dummy_DPU2Alloc();
180 (void)Dummy_DPU2Init((Dummy_DPU2_t *)p_dpu, 512, 3);
181 (void)ADPU2_SetTag(p_dpu, 0x12); /* optional. A tag is useful to identify a DPU along the process*/
182
183 /*Step 2 (optional): register the DPU with the processing task*/
184 (void) DPT1AddDPU(&sProcessTask, p_dpu);
185
186 /*Step 3: allocate the input data buffer for the DPU. The memory allocation is delegated to the application. */
187 /* how many signal I want to handle in parallel?
188 * If the ODR is not too high or the inference time is not too long, then a buffer for two signals should be enough:
189 * - 1 buffer to acquire a data in the sensor task
190 * - 1 buffer to process a data in the process task
191 */
192 const uint8_t max_signals_in_buffer = 2;
193 size_t buff_size = ADPU2_GetInDataPayloadSize(p_dpu) * max_signals_in_buffer;
194 /*allocate the input buffer in the framework heap*/
195 uint8_t *p_buff = SysAlloc(buff_size);
196 assert_param(p_buff != NULL);
197 res = DPT1SetInDataBuffer(&sProcessTask, p_buff, buff_size);
198 assert_param(!SYS_IS_ERROR_CODE(res));
199 /*allocate the output data buffer for the DPU: it must be big enough to store 1 output data.*/
200 buff_size = ADPU2_GetOutDataPayloadSize(p_dpu);
201 /*allocate the output buffer in the framework heap*/
202 p_buff = SysAlloc(buff_size);
203 assert_param(p_buff != NULL);
204 res = DPT1SetOutDataBuffer(&sProcessTask, p_buff, buff_size);
205 assert_param(!SYS_IS_ERROR_CODE(res));
206
207 /*Step4: connect the DPU of the Processing task with the data source (sensor).
208 * We need a data builder to convert the data from the sensor format (int16_t) to the input data wanted by the DPU (float) */
209 IDataBuilder_t *p_data_builder = Int16ToFloatDB_Alloc();
210 if (p_data_builder != NULL)
211 {
212 res = DPT1AttachToDataSource(&sProcessTask, p_data_src, p_data_builder, E_IDB_NO_DATA_LOSS);
213 assert_param(!SYS_IS_ERROR_CODE(res));
214 /*I want to process the data in the process task execution flow and not inline in the sensor task*/
215 (void) DPT1EnableAsyncDataProcessing(&sProcessTask, true);
216 }
217
218 return SYS_NO_ERROR_CODE;
219}
