461static void ble_send_thread_entry(ULONG thread_input)
462{
463 ble_stream_class_t *obj = (ble_stream_class_t *) thread_input;
464 streamMsg_t received_message = { 0 };
465 uint32_t ctrl_msg;
466
467 while (1)
468 {
469 /* Retrieve a message from the queue. */
470 if (TX_SUCCESS == tx_queue_receive(&obj->ble_app_queue, &received_message, TX_WAIT_FOREVER))
471 {
472 if (sObj.status)
473 {
474 switch (received_message.messageId)
475 {
476 case BLE_ISTREAM_MSG_SEND_CMD:
477 if (ble_stream_SendCommandCallback != NULL)
478 {
479 ble_stream_SendCommandCallback(obj->serialized_cmd, obj->serialized_cmd_size);
480 BLE_FreeFunction(obj->serialized_cmd);
481 }
482 break;
483
484 case BLE_ISTREAM_MSG_SEND_CUSTOM_DATA:
485 if (ble_stream_SendCustomDataCallback != NULL)
486 {
487 ble_stream_SendCustomDataCallback(received_message.streamID);
488 }
489 break;
490
491 case BLE_ISTREAM_MSG_CONNECTED:
492 obj->connected = true;
493 ctrl_msg = BLE_ISTREAM_MSG_STOP_ADV_OB;
494 ISendCtrlMsg(sObj.cmd_parser, &ctrl_msg, 1);
495 break;
496
497 case BLE_ISTREAM_MSG_DISCONNECTED:
498 obj->connected = false;
499 ctrl_msg = BLE_ISTREAM_MSG_START_ADV_OB;
500 ISendCtrlMsg(sObj.cmd_parser, &ctrl_msg, 1);
501 break;
502
503 case BLE_ISTREAM_MSG_UPDATE_ADV:
504 updateAdvData();
505 break;
506
507 default:
508 break;
509 }
510 }
511 }
512 }
513}
