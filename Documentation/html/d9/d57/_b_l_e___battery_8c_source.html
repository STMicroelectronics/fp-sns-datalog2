91tBleStatus BLE_BatteryUpdate(uint32_t BatteryLevel, uint32_t Voltage, uint32_t Current, uint32_t Status)
92{
93 tBleStatus ret;
94
95 uint8_t buff[2 + 2 + 2 + 2 + 1];
96
97 STORE_LE_16(buff, (HAL_GetTick() >> 3));
98 STORE_LE_16(buff + 2, (BatteryLevel * 10U));
99 STORE_LE_16(buff + 4, (Voltage));
100 STORE_LE_16(buff + 6, (Current));
101 buff[8] = (uint8_t)Status;
102
103 ret = ACI_GATT_UPDATE_CHAR_VALUE(&BleCharBattery, 0, 2 + 2 + 2 + 2 + 1, buff);
104
105 if (ret != (tBleStatus)BLE_STATUS_SUCCESS)
106 {
107 if (BLE_StdErr_Service == BLE_SERV_ENABLE)

110 Stderr_Update(BufferToWrite, BytesToWrite);

112 else
113 {
114 BLE_MANAGER_PRINTF("Error Updating Bat Char\r\n");
115 }
116 }
