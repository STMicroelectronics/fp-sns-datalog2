41CircularBuffer *CB_Alloc(uint16_t item_count)
42{
43 CircularBuffer *p_obj = SysAlloc(sizeof(CircularBuffer));
44 if (p_obj != NULL)
45 {
46 p_obj->p_items = SysAlloc(sizeof(CBItem) * item_count);
47
48 if (p_obj->p_items == NULL)
49 {
50 /* release the memory */
51 SysFree(p_obj);
52 p_obj = NULL;
53 }
54 else

280CBItem *CB_PeekNextItem(CircularBuffer *_this, CBItem *p_item)
281{
282 assert_param(_this);
283 assert_param(p_item);
284 CBItem *pNextItem = NULL;
285
286 /* find p_item in the buffer */
287 uint16_t index = 0;
288 uint8_t found = 0;
289 while (!found && (index < _this->item_count))
290 {
291 if (&_this->p_items[index] == p_item)
292 {
293 found = 1;
294 }
295 else
296 {
297 index++;
298 }
299 }
300
301 if (found)
302 {
303 pNextItem = &_this->p_items[CB_INCREMENT_IDX(_this, index)];
304 }
305
306 return pNextItem;
307}
