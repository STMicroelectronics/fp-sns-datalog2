55static int verify_packet(const tHciDataPacket *hciReadPacket)
56{
57 const uint8_t *hci_pckt = hciReadPacket->dataBuff;
58
59 if (hci_pckt[HCI_PCK_TYPE_OFFSET] != HCI_EVENT_PKT)
60 {
61 return 1; /* Incorrect type */
62 }
63
64 if (hci_pckt[EVENT_PARAMETER_TOT_LEN_OFFSET] != hciReadPacket->data_len - (1 + HCI_EVENT_HDR_SIZE))
65 {
66 return 2; /* Wrong length (packet truncated or too long) */
67 }
68
69 return 0;
70}

137void hci_init(void(* UserEvtRx)(void *pData), void *pConf)
138{
139 uint8_t index;
140
141 if (UserEvtRx != NULL)
142 {
143 hciContext.UserEvtRx = UserEvtRx;
144 }
145
146 /* Initialize list heads of ready and free hci data packet queues */
147 list_init_head(&hciReadPktPool);
148 list_init_head(&hciReadPktRxQueue);
149
150 /* Initialize TL BLE layer */
151 hci_tl_lowlevel_init();
152
153 /* Initialize the queue of free hci data packets */
154 for (index = 0; index < HCI_READ_PACKET_NUM_MAX; index++)
155 {
156 list_insert_tail(&hciReadPktPool, (tListNode *)&hciReadPacketBuffer[index]);
157 }
158
159 /* Initialize low level driver */
160 if (hciContext.io.Init) { hciContext.io.Init(NULL); }
161 if (hciContext.io.Reset) { hciContext.io.Reset(); }
162}

174int hci_send_req(struct hci_request *r, BOOL async)
175{
176 uint8_t *ptr;
177 uint16_t opcode = htobs(cmd_opcode_pack(r->ogf, r->ocf));
178 hci_event_pckt *event_pckt;
179 hci_spi_pckt *hci_hdr;
180
181 tHciDataPacket *hciReadPacket = NULL;
182 tListNode hciTempQueue;
183
184 list_init_head(&hciTempQueue);
185
186 free_event_list();
187
188 send_cmd(r->ogf, r->ocf, r->clen, r->cparam);
189
190 if (async)
191 {
192 return 0;
193 }
194
195 while (1)
196 {
197 evt_cmd_complete *cc;
198 evt_cmd_status *cs;
199 evt_le_meta_event *me;
200 uint32_t len;
201
202 uint32_t tickstart = HAL_GetTick();
203
204 while (1)
205 {
206 if ((HAL_GetTick() - tickstart) > HCI_DEFAULT_TIMEOUT_MS)
207 {
208 goto failed;
209 }
210
211 if (!list_is_empty(&hciReadPktRxQueue))
212 {
213 break;
214 }
215 }
216
217 /* Extract packet from HCI event queue. */
218 list_remove_head(&hciReadPktRxQueue, (tListNode **)&hciReadPacket);
219
220 hci_hdr = (void *)hciReadPacket->dataBuff;
221
222 if (hci_hdr->type == HCI_EVENT_PKT)
223 {
224 event_pckt = (void *)(hci_hdr->data);
225
226 ptr = hciReadPacket->dataBuff + (1 + HCI_EVENT_HDR_SIZE);
227 len = hciReadPacket->data_len - (1 + HCI_EVENT_HDR_SIZE);
228
229 switch (event_pckt->evt)
230 {
231 case EVT_CMD_STATUS:
232 cs = (void *) ptr;
233
234 if (cs->opcode != opcode)
235 {
236 goto failed;
237 }
238
239 if (r->event != EVT_CMD_STATUS)
240 {
241 if (cs->status)
242 {
243 goto failed;
244 }
245 break;
246 }
247
248 r->rlen = MIN(len, r->rlen);
249 BLUENRG_memcpy(r->rparam, ptr, r->rlen);
250 goto done;
251
252 case EVT_CMD_COMPLETE:
253 cc = (void *) ptr;
254
255 if (cc->opcode != opcode)
256 {
257 goto failed;
258 }
259
260 ptr += EVT_CMD_COMPLETE_SIZE;
261 len -= EVT_CMD_COMPLETE_SIZE;
262
263 r->rlen = MIN(len, r->rlen);
264 BLUENRG_memcpy(r->rparam, ptr, r->rlen);
265 goto done;
266
267 case EVT_LE_META_EVENT:
268 me = (void *) ptr;
269
270 if (me->subevent != r->event)
271 {
272 break;
273 }
274
275 len -= 1;
276 r->rlen = MIN(len, r->rlen);
277 BLUENRG_memcpy(r->rparam, me->data, r->rlen);
278 goto done;
279
280 case EVT_HARDWARE_ERROR:
281 goto failed;
282
283 default:
284 break;
285 }
286 }
287
288 /* If there are no more packets to be processed, be sure there is at list one
289 packet in the pool to process the expected event.
290 If no free packets are available, discard the processed event and insert it
291 into the pool. */
292 if (list_is_empty(&hciReadPktPool) && list_is_empty(&hciReadPktRxQueue))
293 {
294 list_insert_tail(&hciReadPktPool, (tListNode *)hciReadPacket);
295 hciReadPacket = NULL;
296 }
297 else
298 {
299 /* Insert the packet in a different queue. These packets will be
300 inserted back in the main queue just before exiting from send_req(), so that
301 these events can be processed by the application.
302 */
303 list_insert_tail(&hciTempQueue, (tListNode *)hciReadPacket);
304 hciReadPacket = NULL;
305 }
306 }
307
308failed:
309 if (hciReadPacket != NULL)
310 {
311 list_insert_head(&hciReadPktPool, (tListNode *)hciReadPacket);
312 }
313 move_list(&hciReadPktRxQueue, &hciTempQueue);
314
315 return -1;
316
317done:
318 /* Insert the packet back into the pool.*/
319 list_insert_head(&hciReadPktPool, (tListNode *)hciReadPacket);
320 move_list(&hciReadPktRxQueue, &hciTempQueue);
321
322 return 0;
323}

343int32_t hci_notify_asynch_evt(void *pdata)
344{
345 tHciDataPacket *hciReadPacket = NULL;
346 uint8_t data_len;
347
348 int32_t ret = 0;
349
350 if (list_is_empty(&hciReadPktPool) == FALSE)
351 {
352 /* Queuing a packet to read */
353 list_remove_head(&hciReadPktPool, (tListNode **)&hciReadPacket);
354
355 if (hciContext.io.Receive)
356 {
357 data_len = hciContext.io.Receive(hciReadPacket->dataBuff, HCI_READ_PACKET_SIZE);
358 if (data_len > 0)
359 {
360 hciReadPacket->data_len = data_len;
361 if (verify_packet(hciReadPacket) == 0)
362 {
363 list_insert_tail(&hciReadPktRxQueue, (tListNode *)hciReadPacket);
364 }
365 else
366 list_insert_head(&hciReadPktPool, (tListNode *)hciReadPacket);
367 }
368 else
369 {
370 /* Insert the packet back into the pool*/
371 list_insert_head(&hciReadPktPool, (tListNode *)hciReadPacket);
372 }
373 }
374 }
375 else
376 {
377 ret = 1;
378 }
379 return ret;
