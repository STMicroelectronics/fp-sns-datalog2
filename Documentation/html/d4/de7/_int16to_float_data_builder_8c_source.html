76sys_error_code_t Int16ToFloatDB_vtblOnNewInData(IDataBuilder_t *_this, EMData_t *p_target_data, const EMData_t *p_new_in_data, IDB_BuildStrategy_e build_strategy, DataBuffAllocator_f data_buff_alloc)
77{
78 assert_param(_this != NULL);
79 Int16toFloatDataBuilder_t *p_obj = (Int16toFloatDataBuilder_t*)_this;
80 sys_error_code_t res = SYS_NO_ERROR_CODE;
81 EMData_t reshaped_target_data, reshaped_in_data;
82
83 /*reshape the data as 1D because it is a more convenient format for this builder.*/
84 uint32_t target_elements = EMD_GetElementsCount(p_target_data);
85 uint32_t in_elements = EMD_GetElementsCount(p_new_in_data);
86 EMD_1dInit(&reshaped_in_data, EMD_Data(p_new_in_data), EMD_GetType(p_new_in_data), in_elements);
87 EMD_1dInit(&reshaped_target_data, EMD_Data(p_target_data), E_EM_FLOAT, target_elements);
88
89 register float *p_target_val = (float*)EMD_1dDataAt(&reshaped_target_data, p_obj->index);
90 register int16_t *p_src_val = (int16_t*)EMD_1dDataAt(&reshaped_in_data, 0);
91 /*consume all the new input data*/
92 while (in_elements > 0U)
93 {
94 /*how many elements can I copy in the target data? */
95 uint32_t free_elements = target_elements - p_obj->index;
96 uint32_t element_to_copy = (free_elements < in_elements ? free_elements : in_elements); // MIN(free_elements, in_elements)
97 /*copy the input data elements in the target data payload*/
98 for (uint32_t i=0; i<element_to_copy; ++i)
99 {
100 *p_target_val++ = (float)*p_src_val++;
101 }
102
103 in_elements -= element_to_copy;
104 p_obj->index += element_to_copy;
105 /*check if the target data is ready: did I fill all elements in the target data?*/
106 if (p_obj->index >= target_elements)
107 {
108 if (in_elements > 0U)
109 {
110 /* target data is ready but there are still element to be processed.
111 * What to do depends on the build strategy.*/
112 switch (build_strategy)
113 {
114 case E_IDB_NO_DATA_LOSS:
115 reshaped_target_data.p_payload = data_buff_alloc(_this, p_obj->p_data_build_context);
116 if (reshaped_target_data.p_payload == NULL)
117 {
118 SYS_DEBUGF(SYS_DBG_LEVEL_SEVERE, ("IDB_int16_float: data lost!\r\r"));
119 /*no more buffer => data lost!*/
120 sys_error_handler();
121 }
122 else
123 {
124 /*I have a buffer to build a new data, so I reset the index.*/
125 p_obj->index = 0;
126 p_target_val = (float*)EMD_1dDataAt(&reshaped_target_data, p_obj->index);
127 }
128 break;
129
130 case E_IDB_SKIP_DATA:
131 /*ignore the remain input elements.*/
132 in_elements = 0;
133 res = SYS_IDB_DATA_READY_ERROR_CODE;
134 break;
135
136 default:
137 SYS_DEBUGF(SYS_DBG_LEVEL_WARNING, ("IDB_int16_float: unsupported strategy.\r\n"));
138 res = SYS_IDB_UNSUPPORTED_STRATEGY_ERROR_CODE;
139 break;
140 }
141 }
142 else
143 {
144 /* target data is ready.*/
145 res = SYS_IDB_DATA_READY_ERROR_CODE;
146 }
147 }
148 }
149
150 return res;
151}
