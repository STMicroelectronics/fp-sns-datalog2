77sys_error_code_t T1toT2DB_vtblOnNewInData(IDataBuilder_t *_this, EMData_t *p_target_data, const EMData_t *p_new_in_data, IDB_BuildStrategy_e build_strategy, DataBuffAllocator_f data_buff_alloc)
78{
79 assert_param(_this != NULL);
80 T1toT2DataBuilder_t *p_obj = (T1toT2DataBuilder_t*)_this;
81 sys_error_code_t res = SYS_NO_ERROR_CODE;
82 EMData_t reshaped_target_data, reshaped_in_data;
83
84 /*reshape the data as 1D because it is a more convenient format for this builder.*/
85 uint32_t target_elements = EMD_GetElementsCount(p_target_data);
86 uint32_t in_elements = EMD_GetElementsCount(p_new_in_data);
87 EMD_1dInit(&reshaped_in_data, EMD_Data(p_new_in_data), EMD_GetType(p_new_in_data), in_elements);
88 EMD_1dInit(&reshaped_target_data, EMD_Data(p_target_data), E_EM_FLOAT, target_elements);
89
90 register uint8_t *p_target_val = EMD_1dDataAt(&reshaped_target_data, p_obj->index);
91 register uint8_t *p_src_val = EMD_1dDataAt(&reshaped_in_data, 0);
92 /*consume all the new input data*/
93 while (in_elements > 0U)
94 {
95 /*how many elements can I copy in the target data? */
96 uint32_t free_elements = target_elements - p_obj->index;
97 uint32_t element_to_copy = (free_elements < in_elements ? free_elements : in_elements); // MIN(free_elements, in_elements)
98 /*copy the input data elements in the target data payload*/
99 for (uint32_t i=0; i<element_to_copy; ++i)
100 {
101 p_obj->convert(p_src_val, p_target_val);
102 p_target_val += EMD_GetElementSize(&reshaped_target_data);
103 p_src_val += EMD_GetElementSize(&reshaped_in_data);
104 }
105
106 in_elements -= element_to_copy;
107 p_obj->index += element_to_copy;
108 /*check if the target data is ready: did I fill all elements in the target data?*/
109 if (p_obj->index >= target_elements)
110 {
111 if (in_elements > 0U)
112 {
113 /* target data is ready but there are still element to be processed.
114 * What to do depends on the build strategy.*/
115 switch (build_strategy)
116 {
117 case E_IDB_NO_DATA_LOSS:
118 reshaped_target_data.p_payload = data_buff_alloc(_this, p_obj->p_data_build_context);
119 if (reshaped_target_data.p_payload == NULL)
120 {
121 SYS_DEBUGF(SYS_DBG_LEVEL_SEVERE, ("IDB_t1_t2: data lost!\r\r"));
122 /*no more buffer => data lost!*/
123 sys_error_handler();
124 }
125 else
126 {
127 /*I have a buffer to build a new data, so I reset the index.*/
128 p_obj->index = 0;
129 p_target_val = EMD_1dDataAt(&reshaped_target_data, p_obj->index);
130 }
131 break;
132
133 case E_IDB_SKIP_DATA:
134 /*ignore the remain input elements.*/
135 in_elements = 0;
136 res = SYS_IDB_DATA_READY_ERROR_CODE;
137 break;
138
139 default:
140 SYS_DEBUGF(SYS_DBG_LEVEL_WARNING, ("IDB_t1_t2: unsupported strategy.\r\n"));
141 res = SYS_IDB_UNSUPPORTED_STRATEGY_ERROR_CODE;
142 break;
143 }
144 }
145 else
146 {
147 /* target data is ready.*/
148 res = SYS_IDB_DATA_READY_ERROR_CODE;
149 }
150 }
151 }
152
153 return res;
154}
