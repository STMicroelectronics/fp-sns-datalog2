348void ExtiDefISR(uint16_t exti_pin)
349{
350 EXTIPin2CallbckMap xMap = EXTI_GET_P2F_MAP();
351 for (int i = 0; xMap[i].pfCallback != NULL; i++)
352 {
353 if (__HAL_GPIO_EXTI_GET_IT(xMap[i].nPin) && (xMap[i].nPin == exti_pin))
354 {
355 /* EXTI line interrupt detected */
356 __HAL_GPIO_EXTI_CLEAR_IT(xMap[i].nPin);
357 xMap[i].pfCallback(xMap[i].nPin);
358 }
359 }
