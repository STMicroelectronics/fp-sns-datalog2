62{
63 /* Data structure pointer for BLE service */
64 BleCharTypeDef *BleCharPointer;
65
66 /* Init data structure pointer for Machine Learning Core info service */
67 BleCharPointer = &BleCharFiniteStateMachine;
68 memset(BleCharPointer, 0, sizeof(BleCharTypeDef));
69 BleCharPointer->AttrMod_Request_CB = AttrMod_Request_FiniteStateMachine;
70 BleCharPointer->Read_Request_CB = Read_Request_FiniteStateMachine;
71 COPY_FINITE_STATE_MACHINE_CHAR_UUID((BleCharPointer->uuid));
72 BleCharPointer->Char_UUID_Type = UUID_TYPE_128;
73
74 if (NumReg == BLE_FSM_16_REG)
75 {
76 /* 2 byte timestamp, 16 FSM registers output, 2 FSM output state */

82 BleCharPointer->Char_Value_Length = 2 + 8 + 1;

88 BleCharPointer->Security_Permissions = ATTR_PERMISSION_NONE;
89 BleCharPointer->GATT_Evt_Mask = GATT_NOTIFY_READ_REQ_AND_WAIT_FOR_APPL_RESP;
90 BleCharPointer->Enc_Key_Size = 16;
91 BleCharPointer->Is_Variable = 0;

93 BLE_MANAGER_PRINTF("BLE FSM features ok\r\n");

197tBleStatus BLE_FiniteStateMachineUpdate(uint8_t *fsm_out, uint8_t *fsm_status_a_mainpage,

199{
200 tBleStatus ret;
201
202 /* 2 byte timestamp, 16 FSM registers output, 2 MCL output state */
203 uint8_t buff[2 + 16 + 2];
204
205 /* TimeStamp */
206 STORE_LE_16(buff, (HAL_GetTick() >> 3));
207
208 if (BleNumReg == BLE_FSM_16_REG)
209 {
210 /* FSM outputs registers */
211 memcpy(buff + 2, fsm_out, 16U * sizeof(uint8_t));
212
213 /* Status bits for FSM from 1 to 8 */
214 buff[18] = *fsm_status_a_mainpage;
215 /* Status bit for FSM from 9 to 16 */
216 buff[19] = *fsm_status_b_mainpage;
217
218 ret = ACI_GATT_UPDATE_CHAR_VALUE(&BleCharFiniteStateMachine, 0, 2 + 16 + 2, buff);
219 }
220 else
221 {
222 /* FSM outputs registers */
223 memcpy(buff + 2, fsm_out, 8U * sizeof(uint8_t));
224
225 /* Status bits for FSM from 1 to 8 */
226 buff[10] = *fsm_status_a_mainpage;
227
228 ret = ACI_GATT_UPDATE_CHAR_VALUE(&BleCharFiniteStateMachine, 0, 2 + 8 + 1, buff);
229 }
230
231 if (ret != (tBleStatus)BLE_STATUS_SUCCESS)
