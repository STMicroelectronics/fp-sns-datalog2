142{
143 uint16_t res = SYS_NO_ERROR_CODE;
144
145 if(CB_GetFreeItemFromHead(cb, &_this->p_current_item) != SYS_NO_ERROR_CODE)
146 {
147 res = SYS_CB_NO_READY_ITEM_ERROR_CODE;
148 _this->item_idx = 0;
149 _this->p_current_item = NULL;
150 return res;
151 }
152 else
153 {
154 uint8_t *p_dst;
155 p_dst = (uint8_t*) CB_GetItemData(_this->p_current_item);
156 uint32_t *p_byte_counter_dst;
157 uint8_t header_size;
158
159 /* if required, write ch_number at the beginning of the buffer (first byte) */
160 if(_this->add_channel_to_buffer)
161 {
162 p_dst[0] = ch_number;
163 header_size = DL2_CHANNEL_SIZE + DL2_COUNTER_SIZE;
164 p_byte_counter_dst = (uint32_t*) &p_dst[DL2_CHANNEL_SIZE];
165 }
166 else
167 {
168 header_size = DL2_COUNTER_SIZE;
169 p_byte_counter_dst = (uint32_t*) p_dst;
170 }
171
172 /* increment the byte counter by adding the payload size (item - header) */
173 _this->byte_counter += (cb_item_size - header_size);
174 *p_byte_counter_dst = _this->byte_counter;
175
176 /* move buffer index after header */
177 _this->item_idx = header_size;
178 }
