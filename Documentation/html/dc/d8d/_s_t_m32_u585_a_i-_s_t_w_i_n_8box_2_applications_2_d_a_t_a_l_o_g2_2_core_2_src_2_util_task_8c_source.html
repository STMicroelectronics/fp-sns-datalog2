183sys_error_code_t UtilTask_GetBatteryStatus(uint8_t *batt_percentage, uint8_t *status)
184{
185 sys_error_code_t res = SYS_NO_ERROR_CODE;
186 UtilTask_t *p_obj = (UtilTask_t *) &sTaskObj;
187
188 BCPAcquireState(&p_obj->bc_protocol, &p_obj->BattVolt, &p_obj->STBC02_state);
189 if (p_obj->BattVolt > (uint16_t)RCG_BAT_MAX_VOLTAGE)
190 {
191 p_obj->BattVolt = RCG_BAT_MAX_VOLTAGE;
192 }
193 if (p_obj->BattVolt < (uint16_t)RCG_BAT_MIN_VOLTAGE)
194 {
195 p_obj->BattVolt = RCG_BAT_MIN_VOLTAGE;
196 }
197
198 *batt_percentage = ((p_obj->BattVolt - RCG_BAT_MIN_VOLTAGE) * 100) / (RCG_BAT_MAX_VOLTAGE - RCG_BAT_MIN_VOLTAGE);
199
200 if (*batt_percentage > 100)
201 {
202 *batt_percentage = 100;
203 }
204
205 switch (p_obj->STBC02_state)
206 {
207 case DISCHARGING:
208 *status = UTIL_BATTERY_STATUS_DISCHARGING;
209 break;
210
211 case CHARGING:
212 *status = UTIL_BATTERY_STATUS_CHARGING;
213 break;
214
215 case BATTERY_NOT_CONNECTED:
216 *status = UTIL_BATTERY_STATUS_NOT_CONNECTED;
217 break;
218
219 case PLUGGED_NOT_CHARGING:
220 *status = UTIL_BATTERY_STATUS_FULL;
221 break;
222
223 default:
224 *status = UTIL_BATTERY_STATUS_UNKNOWN;
225 break;
226 }
227
228 return res;
229}

236{

238 sys_error_code_t res = SYS_NO_ERROR_CODE;

247 res = SYS_GET_LAST_LOW_LEVEL_ERROR_CODE();
248 }
249 else
250 {
251 BCTimerDriverParams_t cfg_params =
252 {
253 .p_mx_tim_cfg = (void *) p_obj->p_mx_bc_tim_drv_cfg,
254 .p_mx_gpio_sw_cfg = (void *) p_obj->p_mx_bc_gpio_sw_drv_cfg,

256
257 res = IDrvInit(p_obj->p_bc_timer_driver, &cfg_params);
258 if (SYS_IS_ERROR_CODE(res))
259 {
260 SYS_DEBUGF(SYS_DBG_LEVEL_SEVERE, ("UTIL task: error BC during driver initialization\r\n"));
261 }
262 }
263 }
264
265 if (!SYS_IS_ERROR_CODE(res))
266 {
267 p_obj->p_bc_tim_chg_driver = BCTimChgDriverAlloc();
268 if (p_obj->p_bc_tim_chg_driver == NULL)
269 {
270 SYS_DEBUGF(SYS_DBG_LEVEL_SEVERE, ("UTIL task: unable to alloc BCDriver_t object.\r\n"));
271 res = SYS_GET_LAST_LOW_LEVEL_ERROR_CODE();
272 }
273 else
274 {
275 BCTimChgDriverParams_t cfg_params =
276 {
277 .p_mx_tim_cfg = (void *) p_obj->p_mx_bc_tim_chg_drv_cfg,
278 .p_mx_gpio_chg_cfg = (void *) p_obj->p_mx_bc_gpio_chg_drv_cfg,
279 .p_mx_gpio_cen_cfg = (void *) p_obj->p_mx_bc_gpio_cen_drv_cfg,
280 };
281
282 res = IDrvInit(p_obj->p_bc_tim_chg_driver, &cfg_params);
283 if (SYS_IS_ERROR_CODE(res))
284 {
285 SYS_DEBUGF(SYS_DBG_LEVEL_SEVERE, ("UTIL task: error BC during driver initialization\r\n"));
286 }
287 }
288 }
289
290 if (!SYS_IS_ERROR_CODE(res))
291 {
292 p_obj->p_bc_adc_driver = BCAdcDriverAlloc();
293 if (p_obj->p_bc_adc_driver == NULL)

301 if (SYS_IS_ERROR_CODE(res))
302 {
303 SYS_DEBUGF(SYS_DBG_LEVEL_SEVERE, ("UTIL task: error BC during driver initialization\r\n"));
304 }
305 }
306 }
307
308 if (!SYS_IS_ERROR_CODE(res))
309 {
310 /* Initialize the LED and User Button */
311 // configure User Button 1
312 if (p_obj->p_mx_ub_drv_cfg != NULL)
313 {
314 ((MX_GPIOParams_t *)p_obj->p_mx_ub_drv_cfg)->p_mx_init_f();
315 }
316
317 // configure Led1
318 if (p_obj->p_mx_led1_drv_cfg != NULL)
319 {
320 MX_GPIOParams_t *p_ld1_params = (MX_GPIOParams_t *)p_obj->p_mx_led1_drv_cfg;
321 p_ld1_params->p_mx_init_f();
322 HAL_GPIO_WritePin(p_ld1_params->port, p_ld1_params->pin, GPIO_PIN_RESET);
323 }
324
325 // configure Led2
326 if (p_obj->p_mx_led2_drv_cfg != NULL)
327 {
328 MX_GPIOParams_t *p_ld2_params = (MX_GPIOParams_t *)p_obj->p_mx_led2_drv_cfg;
329 p_ld2_params->p_mx_init_f();

331 }
332
333 }
334
335 // configure the BatteryCharger
336 PB_PWR_Init();
337
338 // Configure random number generated
339 RND_Init();
340
341 return res;

344sys_error_code_t UtilTask_vtblOnCreateTask(AManagedTask *_this, tx_entry_function_t *pTaskCode, CHAR **pName,

364 UTIL_TASK_CFG_IN_QUEUE_ITEM_COUNT * item_size))
365 {
366 res = SYS_UTIL_TASK_INIT_ERROR_CODE;
367 SYS_SET_SERVICE_LEVEL_ERROR_CODE(res);
368 return res;
369 }
370
371 /* Software timer for user led management */
372 if (TX_SUCCESS != tx_timer_create(&p_obj->user_led_timer, "USER_LED_T", UtilTaskSwTimerCallbackUserLed, (ULONG)TX_NULL,
373 AMT_MS_TO_TICKS(UTIL_TASK_CFG_UL_TIMER_PERIOD_MS), AMT_MS_TO_TICKS(UTIL_TASK_CFG_UL_TIMER_PERIOD_MS),
374 TX_AUTO_ACTIVATE))
375 {
376 res = SYS_APP_TASK_INIT_ERROR_CODE;
377 SYS_SET_SERVICE_LEVEL_ERROR_CODE(res);
378 return res;
379 }
380
381 /* initialize the protocol object */
382 res = BCPInit(&p_obj->bc_protocol, p_obj->p_bc_timer_driver, p_obj->p_bc_tim_chg_driver, p_obj->p_bc_adc_driver);
383
384 if (SYS_IS_ERROR_CODE(res))
385 {
386 return res;
387 }
388
389
390 /*****/
391 /* Workaround - disable and enable power supply for external add-on */
392 SYS_DEBUGF(SYS_DBG_LEVEL_VERBOSE, ("External add-on 3.3V off\r\n"));
393
394 /* Switch off the power supply driven from Battery Charger */
395 BCPSendCmd(&p_obj->bc_protocol, E_SW1_OA_OFF);
396 BCPSendCmd(&p_obj->bc_protocol, E_SW1_OB_OFF);
397
398 /* Avoid spurious high signals on the SPI pins */
399 /* Disable SPI pins */
400 HAL_SPI_MspDeInit(&hspi2);
401 GPIO_InitTypeDef GPIO_InitStruct = {0};
402 GPIO_InitStruct.Pin = GPIO_PIN_3;
403 GPIO_InitStruct.Mode = GPIO_MODE_OUTPUT_PP;
404 GPIO_InitStruct.Pull = GPIO_NOPULL;
405 GPIO_InitStruct.Speed = GPIO_SPEED_FREQ_LOW;
406 HAL_GPIO_Init(GPIOD, &GPIO_InitStruct);
407 /* Force low signal on SPI pins */
408 HAL_GPIO_WritePin(GPIOD, GPIO_PIN_3, GPIO_PIN_RESET);
409 HAL_GPIO_WritePin(GPIOI, GPIO_PIN_1 | GPIO_PIN_3, GPIO_PIN_RESET);
410 HAL_Delay(100);
411 /* Re-initialize SPI pins */
412 HAL_SPI_MspInit(&hspi2);
413
414 /* Switch on the power supply newly */
415 BCPSendCmd(&p_obj->bc_protocol, E_SW1_OA_ON);
416 BCPSendCmd(&p_obj->bc_protocol, E_SW1_OB_ON);
417 HAL_Delay(100);
418 SYS_DEBUGF(SYS_DBG_LEVEL_VERBOSE, ("External add-on 3.3V on\r\n"));
419 /* Now the external add-on is ready to be used */
420 /*****/
421
422
423 /* set the (PM_STATE, ExecuteStepFunc) map from the class object. */
424 _this->m_pfPMState2FuncMap = sTheClass.p_pm_state2func_map;
425 p_obj->BattVolt = 0;
426 p_obj->STBC02_state = (ESTBC02_STATE_t) 4;
427
428 *pTaskCode = AMTExRun;
429 *pName = "UTIL";
430 *pvStackStart = NULL; // allocate the task stack in the system memory pool.
431 *pStackDepth = UTIL_TASK_CFG_STACK_DEPTH;
432 *pParams = (ULONG) _this;
433 *pPriority = UTIL_TASK_CFG_PRIORITY;
434 *pPreemptThreshold = UTIL_TASK_CFG_PRIORITY;
435 *pTimeSlice = TX_NO_TIME_SLICE;
436 *pAutoStart = TX_AUTO_START;
437
438 return res;
439}

554{
555 assert_param(_this != NULL);
556 sys_error_code_t res = SYS_NO_ERROR_CODE;
557 UtilTask_t *p_obj = (UtilTask_t *) _this;
558 static uint8_t led_count_state1 = 0;
559
560 struct utilMessage_t msg =
561 {
562 0
563 };
564
565 AMTExSetInactiveState((AManagedTaskEx *) _this, TRUE);
566
567 if (TX_SUCCESS == tx_queue_receive(&p_obj->in_queue, &msg, TX_WAIT_FOREVER))
568 {
569 AMTExSetInactiveState((AManagedTaskEx *) _this, FALSE);
570
571 if (msg.msgId == APP_REPORT_ID_FORCE_STEP)
572 {
573 __NOP();
574 }
575 else if (msg.msgId == APP_MESSAGE_ID_UTIL)
576 {
577 if (msg.nCmdID == UTIL_CMD_ID_BUTTON_EVT)
578 {
579 /* turn-off the battery charger */
580 res = BCPPowerOff(&p_obj->bc_protocol);
581 }
582 else if (msg.nCmdID == UTIL_CMD_ID_DATALOG_LED)
583 {
584 led_count_state1++;
585 if (led_count_state1 == 4)
586 {
587 led_count_state1 = 0;
588 if (p_obj->p_mx_led1_drv_cfg != NULL)
589 {
590 MX_GPIOParams_t *p_ld1_params = (MX_GPIOParams_t *)p_obj->p_mx_led1_drv_cfg;
591 HAL_GPIO_TogglePin(p_ld1_params->port, p_ld1_params->pin);
592 }
593 }
594 }
595 }
596 }
597
598 return res;
599}
