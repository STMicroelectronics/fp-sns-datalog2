110sys_error_code_t SPIMasterDriver_vtblInit(IDriver *_this, void *p_params)
111{
112 assert_param(_this != NULL);
113 assert_param(p_params != NULL);
114 sys_error_code_t res = SYS_NO_ERROR_CODE;
115 UINT nRes = TX_SUCCESS;
116 SPIMasterDriver_t *p_obj = (SPIMasterDriver_t *) _this;
117 p_obj->mx_handle.p_mx_spi_cfg = ((SPIMasterDriverParams_t *) p_params)->p_mx_spi_cfg;
118 SPI_HandleTypeDef *p_spi = p_obj->mx_handle.p_mx_spi_cfg->p_spi_handle;
119
120 p_obj->mx_handle.p_mx_spi_cfg->p_mx_dma_init_f();
121 p_obj->mx_handle.p_mx_spi_cfg->p_mx_init_f();
122
123 /* Register SPI DMA complete Callback */
124 if (HAL_OK != HAL_SPI_RegisterCallback(p_spi, HAL_SPI_RX_COMPLETE_CB_ID, SPIMasterDriverTxRxCpltCallback))
125 {
126 SYS_SET_LOW_LEVEL_ERROR_CODE(SYS_UNDEFINED_ERROR_CODE);
127 res = SYS_UNDEFINED_ERROR_CODE;
128 }
129 else if (HAL_OK != HAL_SPI_RegisterCallback(p_spi, HAL_SPI_TX_COMPLETE_CB_ID, SPIMasterDriverTxRxCpltCallback))
130 {
131 SYS_SET_LOW_LEVEL_ERROR_CODE(SYS_UNDEFINED_ERROR_CODE);
132 res = SYS_UNDEFINED_ERROR_CODE;
133 }
134 else
135 {
136 if (!HWDriverMap_IsInitialized(&sTheClass.ip_drv_map))
137 {
138 (void) HWDriverMap_Init(&sTheClass.ip_drv_map, sTheClass.ip_drv_map_elements, SPIDRV_CFG_HARDWARE_PERIPHERALS_COUNT);
139 }
140
141 /* Add the driver to the map.
142 * Use the peripheral address as unique key for the map. */
143 HWDriverMapElement_t *p_element = NULL;
144 uint32_t key = (uint32_t) p_obj->mx_handle.p_mx_spi_cfg->p_spi_handle->Instance;
145 p_element = HWDriverMap_AddElement(&sTheClass.ip_drv_map, key, _this);
146
147
148 if (p_element == NULL)
149 {
150 SYS_SET_LOW_LEVEL_ERROR_CODE(SYS_INVALID_PARAMETER_ERROR_CODE);
151 res = SYS_INVALID_PARAMETER_ERROR_CODE;
152 }
153 else
154 {
155 nRes = tx_semaphore_create(&p_obj->sync_obj, "SPIDrv", 0);
156 if (nRes != TX_SUCCESS)
157 {
158 SYS_SET_LOW_LEVEL_ERROR_CODE(SYS_OUT_OF_MEMORY_ERROR_CODE);
159 res = SYS_OUT_OF_MEMORY_ERROR_CODE;
160
161 (void) HWDriverMap_RemoveElement(&sTheClass.ip_drv_map, key);
162 }
163 }
164 }
165
166 SYS_DEBUGF(SYS_DBG_LEVEL_VERBOSE, ("SPIMasterDriver: initialization done: %d.\r\n", res));
167
168 return res;
