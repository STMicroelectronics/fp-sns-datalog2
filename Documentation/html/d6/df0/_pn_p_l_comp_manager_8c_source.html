96void PnPLGenerateAcquisitionUUID(char *uuid)
97{
98 uint32_t UUID[4];
99 uint8_t *p8 = (uint8_t *)UUID;
100 uint8_t index = 0;
101
102 UUID[0] = (uint32_t)rand();
103 UUID[1] = (uint32_t)rand();
104 UUID[2] = (uint32_t)rand();
105 UUID[3] = (uint32_t)rand();
106
107 /*
108 * UUID format
109 * xxxxxxxx-xxxx-Mxxx-Nxxx-xxxxxxxxxxxx
110 * M: The four bits of digit M are the UUID version --> 4 is for random number
111 * N: 1 to 3 most significant bits of digit N represent UUID variant --> using Variant 1 (10x)
112 * */
113 p8[5] = 0x40u | (p8[5] & 0xfu);
114 p8[11] = 0x80u | (p8[11] & 0x3fu); /* in variant 1, 3rd bit can be ignored */
115
116 index += (uint8_t)sprintf(&global_uuid[index], "%08lx-", (unsigned long)UUID[0]);
117 index += (uint8_t)sprintf(&global_uuid[index], "%04lx-%04lx-", (unsigned long)((UUID[1] >> 16) & 0xFFFFu),
118 (unsigned long)(UUID[1] & 0xFFFFu));
119 index += (uint8_t)sprintf(&global_uuid[index], "%04lx-%04lx", (unsigned long)((UUID[2] >> 16) & 0xFFFFu),
120 (unsigned long)(UUID[2] & 0xFFFFu));
121 index += (uint8_t)sprintf(&global_uuid[index], "%08lx", (unsigned long)UUID[3]);
122
123 (void)strcpy(uuid, global_uuid);
124}

613uint8_t PnPLParseCommand(char *commandString, PnPLCommand_t *command)
614{
615 uint8_t commandType = 0;
616 char componentName[2 *
617 COMP_KEY_MAX_LENGTH]; /* 2* because this could be a comp or a comm key. If comm_key this is in the form (comp_key*comm_key) */
618 uint8_t ret = extract_PnPL_cmd_data(commandString, &commandType, componentName);
619
620 command->comm_type = commandType;
621 (void)strcpy(command->comp_name, componentName);
622
623 if (ret == PNPL_CMD_NO_ERROR_CODE)
624 {
625 if (commandType == PNPL_CMD_SET)
626 {
627 /* Select right parse/update function */
628 for (uint8_t i = 0; i < PnPLGetNComponents(); i++)
629 {
630 IPnPLComponent_t *p_obj = (IPnPLComponent_t *)(spPnPLObj.Components[i]);
631 if (strcmp(componentName, IPnPLComponentGetKey(p_obj)) == 0)
632 {
633#ifdef PNPL_RESPONSES
634 char *set_response = 0;
635 uint32_t size = 0;
636 (void)IPnPLComponentSetProperty(p_obj, commandString, &set_response, &size, 0);
637 /* SET Response */
638 command->response = (char*)pnpl_malloc(size);
639 if (command->response != NULL)
640 {
641 strcpy(command->response, set_response);
642 pnpl_free(set_response);
643 }
644 /* SET Response*/
645#else
646 (void)IPnPLComponentSetProperty(p_obj, commandString);
647#endif
648 }
649 uint8_t nOfCommands = IPnPLComponentGetNCommands(p_obj); // 0 if none
650 if (nOfCommands > 0u)
651 {
652 for (uint8_t j = 0; j < nOfCommands; j++)
653 {
654 if (strcmp(IPnPLComponentGetCommandKey(p_obj, j), componentName) == 0)
655 {
656#ifdef PNPL_RESPONSES
657 char *cmd_response = 0;
658 uint32_t size = 0;
659 (void)IPnPLCommandExecuteFunction(p_obj, commandString, &cmd_response, &size, 0);
660 /* CMD Response */
661 command->comm_type = PNPL_CMD_COMMAND;
662 command->response = (char*)pnpl_malloc(size);
663 if (command->response != NULL)

811 }
