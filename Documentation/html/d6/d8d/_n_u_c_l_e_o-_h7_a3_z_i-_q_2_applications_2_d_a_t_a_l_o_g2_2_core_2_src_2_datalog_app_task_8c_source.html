482sys_error_code_t DatalogAppTask_OnNewDataReady_vtbl(IEventListener *_this, const DataEvent_t *p_evt)
483{
484 sys_error_code_t res = SYS_NO_ERROR_CODE;
485 DatalogAppTask *p_obj = (DatalogAppTask *) IEventListenerGetOwner(_this);
486
487 uint8_t nPMState = (uint8_t) AMTGetTaskPowerMode((AManagedTask *) p_obj);
488 uint16_t sId = p_evt->tag;
489 uint32_t nBytesPerSample = SMGetnBytesPerSample(sId);
490 uint8_t *data_buf = EMD_Data(p_evt->p_data);
491 uint32_t samplesToSend = EMD_GetPayloadSize(p_evt->p_data) / nBytesPerSample;
492
493 if (nPMState == E_POWER_MODE_SENSORS_ACTIVE)
494 {
495 uint8_t stream_id = p_obj->datalog_model->s_models[sId]->stream_params.stream_id;
496 if (res != 0)
497 {
498 SYS_SET_SERVICE_LEVEL_ERROR_CODE(res);
499 return res;
500 }
501 if (p_obj->sensorContext[sId].old_time_stamp == -1.0f)
502 {
503 p_obj->datalog_model->s_models[sId]->stream_params.ioffset = p_evt->timestamp; /*TODO: can I use PnPL Setter? no*/
504 p_obj->sensorContext[sId].old_time_stamp = p_evt->timestamp;
505 p_obj->sensorContext[sId].n_samples_to_timestamp = p_obj->datalog_model->s_models[sId]->stream_params.spts;
506 }
507 else
508 {
509 while (samplesToSend > 0)
510 {
511 /* n_samples_to_timestamp = 0 if user setup spts = 0 (no timestamp needed) */
512 if (p_obj->sensorContext[sId].n_samples_to_timestamp == 0
513 || samplesToSend < p_obj->sensorContext[sId].n_samples_to_timestamp)
514 {
515 if (IStream_is_enabled((IStream_t *) p_obj->usbx_device))
516 {
517 res = IStream_post_data((IStream_t *) p_obj->usbx_device, stream_id, data_buf, samplesToSend * nBytesPerSample);
518 }
519 if (res != 0)
520 {
521 SYS_SET_SERVICE_LEVEL_ERROR_CODE(res);
522 return res;
523 }
524 if (p_obj->sensorContext[sId].n_samples_to_timestamp != 0)
525 {
526 p_obj->sensorContext[sId].n_samples_to_timestamp -= samplesToSend;
527 }
528 samplesToSend = 0;
529 }
530 else
531 {
532 if (IStream_is_enabled((IStream_t *) p_obj->usbx_device))
533 {
534 res = IStream_post_data((IStream_t *) p_obj->usbx_device, stream_id, data_buf, p_obj->sensorContext[sId].n_samples_to_timestamp * nBytesPerSample);
535 }
536 if (res != 0)
537 {
538 SYS_SET_SERVICE_LEVEL_ERROR_CODE(res);
539 return res;
540 }
541
542 data_buf += p_obj->sensorContext[sId].n_samples_to_timestamp * nBytesPerSample;
543 samplesToSend -= p_obj->sensorContext[sId].n_samples_to_timestamp;
544
545 float measuredODR;
546 double newTS;
547
548 SensorStatus_t sensor_status = SMSensorGetStatus(sId);
549 if (sensor_status.isensor_class == ISENSOR_CLASS_MEMS)
550 {
551 measuredODR = sensor_status.type.mems.measured_odr;
552 }
553 else if (sensor_status.isensor_class == ISENSOR_CLASS_AUDIO)
554 {
555 measuredODR = sensor_status.type.audio.frequency;
556 }
557 else
558 {
559 SYS_SET_SERVICE_LEVEL_ERROR_CODE(res);
560 return res;
561 }
562
563 if (measuredODR != 0.0f)
564 {
565 newTS = p_evt->timestamp - ((1.0 / (double) measuredODR) * samplesToSend);
566 }
567 else
568 {
569 newTS = p_evt->timestamp;
570 }
571
572 if (IStream_is_enabled((IStream_t *) p_obj->usbx_device))
573 {
574 res = IStream_post_data((IStream_t *) p_obj->usbx_device, stream_id, (uint8_t *) &newTS, 8);
575 }
576 if (res != 0)
577 {
578 SYS_SET_SERVICE_LEVEL_ERROR_CODE(res);
579 return res;
580 }
581 p_obj->sensorContext[sId].n_samples_to_timestamp = p_obj->datalog_model->s_models[sId]->stream_params.spts;
582 }
583 }
584 }
585 }
586 return res;
587}
