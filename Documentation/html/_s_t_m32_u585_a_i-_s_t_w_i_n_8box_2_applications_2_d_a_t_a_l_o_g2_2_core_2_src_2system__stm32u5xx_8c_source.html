194void SystemInit(void)
195{
196 /* FPU settings ------------------------------------------------------------*/
197#if (__FPU_PRESENT == 1) && (__FPU_USED == 1)
198 SCB->CPACR |= ((3UL << 20U) | (3UL << 22U)); /* set CP10 and CP11 Full Access */
199#endif
200
201 /* Reset the RCC clock configuration to the default reset state ------------*/
202 /* Set MSION bit */
203 RCC->CR = RCC_CR_MSISON;
204
205 /* Reset CFGR register */
206 RCC->CFGR1 = 0U;
207 RCC->CFGR2 = 0U;
208 RCC->CFGR3 = 0U;
209
210 /* Reset HSEON, CSSON , HSION, PLLxON bits */
211 RCC->CR &= ~(RCC_CR_HSEON | RCC_CR_CSSON | RCC_CR_PLL1ON | RCC_CR_PLL2ON | RCC_CR_PLL3ON);
212
213 /* Reset PLLCFGR register */
214 RCC->PLL1CFGR = 0U;
215
216 /* Reset HSEBYP bit */
217 RCC->CR &= ~(RCC_CR_HSEBYP);
218
219 /* Disable all interrupts */
220 RCC->CIER = 0U;
221
222 /* Configure the Vector Table location add offset address ------------------*/
223#ifdef VECT_TAB_SRAM
224 SCB->VTOR = SRAM1_BASE | VECT_TAB_OFFSET; /* Vector Table Relocation in Internal SRAM */
225#else
226 SCB->VTOR = FLASH_BASE | VECT_TAB_OFFSET; /* Vector Table Relocation in Internal FLASH */
227#endif
228}

272void SystemCoreClockUpdate(void)
273{
274 uint32_t pllr, pllsource, pllm, tmp, pllfracen, msirange;
275 float_t fracn1, pllvco;
276
277 /* Get MSI Range frequency--------------------------------------------------*/
278 if (READ_BIT(RCC->ICSCR1, RCC_ICSCR1_MSIRGSEL) == 0U)
279 {
280 /* MSISRANGE from RCC_CSR applies */
281 msirange = (RCC->CSR & RCC_CSR_MSISSRANGE) >> RCC_CSR_MSISSRANGE_Pos;
282 }
283 else
284 {
285 /* MSIRANGE from RCC_CR applies */
286 msirange = (RCC->ICSCR1 & RCC_ICSCR1_MSISRANGE) >> RCC_ICSCR1_MSISRANGE_Pos;
287 }
288
289 /*MSI frequency range in HZ*/
290 msirange = MSIRangeTable[msirange];
291
292 /* Get SYSCLK source -------------------------------------------------------*/
293 switch (RCC->CFGR1 & RCC_CFGR1_SWS)
294 {
295 case 0x00: /* MSI used as system clock source */
296 SystemCoreClock = msirange;
297 break;
298
299 case 0x04: /* HSI used as system clock source */
300 SystemCoreClock = HSI_VALUE;
301 break;
302
303 case 0x08: /* HSE used as system clock source */
304 SystemCoreClock = HSE_VALUE;
305 break;
306
307 case 0x0C: /* PLL used as system clock source */
308 /* PLL_VCO = (HSE_VALUE or HSI_VALUE or MSI_VALUE/ PLLM) * PLLN
309 SYSCLK = PLL_VCO / PLLR
310 */
311 pllsource = (RCC->PLL1CFGR & RCC_PLL1CFGR_PLL1SRC);
312 pllm = ((RCC->PLL1CFGR & RCC_PLL1CFGR_PLL1M) >> RCC_PLL1CFGR_PLL1M_Pos) + 1U;
313 pllfracen = ((RCC->PLL1CFGR & RCC_PLL1CFGR_PLL1FRACEN) >> RCC_PLL1CFGR_PLL1FRACEN_Pos);
314 fracn1 = (float_t)(uint32_t)(pllfracen * ((RCC->PLL1FRACR & RCC_PLL1FRACR_PLL1FRACN) >> RCC_PLL1FRACR_PLL1FRACN_Pos));
315
316 switch (pllsource)
317 {
318 case 0x00: /* No clock sent to PLL*/
319 pllvco = (float_t)0U;
320 break;
321
322 case 0x02: /* HSI used as PLL clock source */
323 pllvco = ((float_t)HSI_VALUE / (float_t)pllm);
324 break;
325
326 case 0x03: /* HSE used as PLL clock source */
327 pllvco = ((float_t)HSE_VALUE / (float_t)pllm);
328 break;
329
330 default: /* MSI used as PLL clock source */
331 pllvco = ((float_t)msirange / (float_t)pllm);
332 break;
333 }
334
335 pllvco = pllvco * ((float_t)(uint32_t)(RCC->PLL1DIVR & RCC_PLL1DIVR_PLL1N) + (fracn1 / (float_t)0x2000) + (float_t)1U);
336 pllr = (((RCC->PLL1DIVR & RCC_PLL1DIVR_PLL1R) >> RCC_PLL1DIVR_PLL1R_Pos) + 1U);
337 SystemCoreClock = (uint32_t)((uint32_t)pllvco / pllr);
338 break;
339
340 default:
341 SystemCoreClock = msirange;
342 break;
343 }
344 /* Compute HCLK clock frequency --------------------------------------------*/
345 /* Get HCLK prescaler */
346 tmp = AHBPrescTable[((RCC->CFGR2 & RCC_CFGR2_HPRE) >> RCC_CFGR2_HPRE_Pos)];
347 /* HCLK clock frequency */
348 SystemCoreClock >>= tmp;
349}
