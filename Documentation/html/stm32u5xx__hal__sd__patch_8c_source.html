337HAL_StatusTypeDef HAL_SD_Init(SD_HandleTypeDef *hsd)
338{
339 HAL_SD_CardStatusTypeDef CardStatus;
340 uint32_t speedgrade;
341 uint32_t unitsize;
342 uint32_t tickstart;
343
344 /* Check the SD handle allocation */
345 if (hsd == NULL)
346 {
347 return HAL_ERROR;
348 }
349
350 /* Check the parameters */
351 assert_param(IS_SDMMC_ALL_INSTANCE(hsd->Instance));
352 assert_param(IS_SDMMC_CLOCK_EDGE(hsd->Init.ClockEdge));
353 assert_param(IS_SDMMC_CLOCK_POWER_SAVE(hsd->Init.ClockPowerSave));
354 assert_param(IS_SDMMC_BUS_WIDE(hsd->Init.BusWide));
355 assert_param(IS_SDMMC_HARDWARE_FLOW_CONTROL(hsd->Init.HardwareFlowControl));
356 assert_param(IS_SDMMC_CLKDIV(hsd->Init.ClockDiv));
357
358 if (hsd->State == HAL_SD_STATE_RESET)
359 {
360 /* Allocate lock resource and initialize it */
361 hsd->Lock = HAL_UNLOCKED;
362
363#if (USE_SD_TRANSCEIVER != 0U)
364 /* Force SDMMC_TRANSCEIVER_PRESENT for Legacy usage */
365 if (hsd->Init.TranceiverPresent == SDMMC_TRANSCEIVER_UNKNOWN)
366 {
367 hsd->Init.TranceiverPresent = SDMMC_TRANSCEIVER_PRESENT;
368 }
369#endif /*USE_SD_TRANSCEIVER */
370#if defined (USE_HAL_SD_REGISTER_CALLBACKS) && (USE_HAL_SD_REGISTER_CALLBACKS == 1U)
371 /* Reset Callback pointers in HAL_SD_STATE_RESET only */
372 hsd->TxCpltCallback = HAL_SD_TxCpltCallback;
373 hsd->RxCpltCallback = HAL_SD_RxCpltCallback;
374 hsd->ErrorCallback = HAL_SD_ErrorCallback;
375 hsd->AbortCpltCallback = HAL_SD_AbortCallback;
376 hsd->Read_DMALnkLstBufCpltCallback = HAL_SDEx_Read_DMALnkLstBufCpltCallback;
377 hsd->Write_DMALnkLstBufCpltCallback = HAL_SDEx_Write_DMALnkLstBufCpltCallback;
378#if (USE_SD_TRANSCEIVER != 0U)
379 if (hsd->Init.TranceiverPresent == SDMMC_TRANSCEIVER_PRESENT)
380 {
381 hsd->DriveTransceiver_1_8V_Callback = HAL_SD_DriveTransceiver_1_8V_Callback;
382 }
383#endif /* USE_SD_TRANSCEIVER */
384
385 if (hsd->MspInitCallback == NULL)
386 {
387 hsd->MspInitCallback = HAL_SD_MspInit;
388 }
389
390 /* Init the low level hardware */
391 hsd->MspInitCallback(hsd);
392#else
393 /* Init the low level hardware : GPIO, CLOCK, CORTEX...etc */
394 HAL_SD_MspInit(hsd);
395#endif /* USE_HAL_SD_REGISTER_CALLBACKS */
396 }
397
398 hsd->State = HAL_SD_STATE_BUSY;
399
400 /* Initialize the Card parameters */
401 if (HAL_SD_InitCard(hsd) != HAL_OK)
402 {
403 return HAL_ERROR;
404 }
405
406 if (HAL_SD_GetCardStatus(hsd, &CardStatus) != HAL_OK)
407 {
408 return HAL_ERROR;
409 }
410 /* Get Initial Card Speed from Card Status*/
411 speedgrade = CardStatus.UhsSpeedGrade;
412 unitsize = CardStatus.UhsAllocationUnitSize;
413 if ((hsd->SdCard.CardType == CARD_SDHC_SDXC) && ((speedgrade != 0U) || (unitsize != 0U)))
414 {
415 hsd->SdCard.CardSpeed = CARD_ULTRA_HIGH_SPEED;
416 }
417 else
418 {
419 if (hsd->SdCard.CardType == CARD_SDHC_SDXC)
420 {
421 hsd->SdCard.CardSpeed = CARD_HIGH_SPEED;
422 }
423 else
424 {
425 hsd->SdCard.CardSpeed = CARD_NORMAL_SPEED;
426 }
427
428 }
429 /* Configure the bus wide */
430 if (HAL_SD_ConfigWideBusOperation(hsd, hsd->Init.BusWide) != HAL_OK)
431 {
432 return HAL_ERROR;
433 }
434
435 /* Verify that SD card is ready to use after Initialization */
436 tickstart = HAL_GetTick();
437 while ((HAL_SD_GetCardState(hsd) != HAL_SD_CARD_TRANSFER))
438 {
439 if ((HAL_GetTick() - tickstart) >= SDMMC_SWDATATIMEOUT)
440 {
441 hsd->ErrorCode = HAL_SD_ERROR_TIMEOUT;
442 hsd->State = HAL_SD_STATE_READY;
443 return HAL_TIMEOUT;
444 }
445 }
446
447 /* Initialize the error code */
448 hsd->ErrorCode = HAL_SD_ERROR_NONE;
449
450 /* Initialize the SD operation */
451 hsd->Context = SD_CONTEXT_NONE;
452
453 /* Initialize the SD state */
454 hsd->State = HAL_SD_STATE_READY;
455
456 return HAL_OK;
457}

466HAL_StatusTypeDef HAL_SD_InitCard(SD_HandleTypeDef *hsd)
467{
468 uint32_t errorstate;
469 SD_InitTypeDef Init;
470 uint32_t sdmmc_clk;
471
472 /* Default SDMMC peripheral configuration for SD card initialization */
473 Init.ClockEdge = SDMMC_CLOCK_EDGE_RISING;
474 Init.ClockPowerSave = SDMMC_CLOCK_POWER_SAVE_DISABLE;
475 Init.BusWide = SDMMC_BUS_WIDE_1B;
476 Init.HardwareFlowControl = SDMMC_HARDWARE_FLOW_CONTROL_ENABLE;
477
478 /* Init Clock should be less or equal to 400Khz*/
479 sdmmc_clk = HAL_RCCEx_GetPeriphCLKFreq(RCC_PERIPHCLK_SDMMC);
480 if (sdmmc_clk == 0U)
481 {
482 hsd->State = HAL_SD_STATE_READY;
483 hsd->ErrorCode = SDMMC_ERROR_INVALID_PARAMETER;
484 return HAL_ERROR;
485 }
486 Init.ClockDiv = sdmmc_clk / (2U * SD_INIT_FREQ);
487
488#if (USE_SD_TRANSCEIVER != 0U)
489 Init.TranceiverPresent = hsd->Init.TranceiverPresent;
490
491 if (hsd->Init.TranceiverPresent == SDMMC_TRANSCEIVER_PRESENT)
492 {
493 /* Set Transceiver polarity */
494 hsd->Instance->POWER |= SDMMC_POWER_DIRPOL;
495 }
496#elif defined (USE_SD_DIRPOL)
497 /* Set Transceiver polarity */
498 hsd->Instance->POWER |= SDMMC_POWER_DIRPOL;
499#endif /* USE_SD_TRANSCEIVER */
500
501 /* Initialize SDMMC peripheral interface with default configuration */
502 (void)SDMMC_Init(hsd->Instance, Init);
503
504 /* Set Power State to ON */
505 (void)SDMMC_PowerState_ON(hsd->Instance);
506
507 /* wait 74 Cycles: required power up waiting time before starting
508 the SD initialization sequence */
509 if (Init.ClockDiv != 0U)
510 {
511 sdmmc_clk = sdmmc_clk / (2U * Init.ClockDiv);
512 }
513
514 if (sdmmc_clk != 0U)
515 {
516 HAL_Delay(1U + (74U * 1000U / (sdmmc_clk)));
517 }
518
519 /* Identify card operating voltage */
520 errorstate = SD_PowerON(hsd);
521 if (errorstate != HAL_SD_ERROR_NONE)
522 {
523 hsd->State = HAL_SD_STATE_READY;
524 hsd->ErrorCode |= errorstate;
525 return HAL_ERROR;
526 }
527
528 /* Card initialization */
529 errorstate = SD_InitCard(hsd);
530 if (errorstate != HAL_SD_ERROR_NONE)
531 {
532 hsd->State = HAL_SD_STATE_READY;
533 hsd->ErrorCode |= errorstate;
534 return HAL_ERROR;
535 }
536
537 /* Set Block Size for Card */
538 errorstate = SDMMC_CmdBlockLength(hsd->Instance, BLOCKSIZE);
539 if (errorstate != HAL_SD_ERROR_NONE)
540 {
541 /* Clear all the static flags */
542 __HAL_SD_CLEAR_FLAG(hsd, SDMMC_STATIC_FLAGS);
543 hsd->ErrorCode |= errorstate;
544 hsd->State = HAL_SD_STATE_READY;
545 return HAL_ERROR;
546 }
547
548 return HAL_OK;
549}

669HAL_StatusTypeDef HAL_SD_ReadBlocks(SD_HandleTypeDef *hsd, uint8_t *pData, uint32_t BlockAdd, uint32_t NumberOfBlocks,
670 uint32_t Timeout)
671{
672 SDMMC_DataInitTypeDef config;
673 uint32_t errorstate;
674 uint32_t tickstart = HAL_GetTick();
675 uint32_t count;
676 uint32_t data;
677 uint32_t dataremaining;
678 uint32_t add = BlockAdd;
679 uint8_t *tempbuff = pData;
680
681 if (NULL == pData)
682 {
683 hsd->ErrorCode |= HAL_SD_ERROR_PARAM;
684 return HAL_ERROR;
685 }
686
687 if (hsd->State == HAL_SD_STATE_READY)
688 {
689 hsd->ErrorCode = HAL_SD_ERROR_NONE;
690
691 if ((add + NumberOfBlocks) > (hsd->SdCard.LogBlockNbr))
692 {
693 hsd->ErrorCode |= HAL_SD_ERROR_ADDR_OUT_OF_RANGE;
694 return HAL_ERROR;
695 }
696
697 hsd->State = HAL_SD_STATE_BUSY;
698
699 /* Initialize data control register */
700 hsd->Instance->DCTRL = 0U;
701
702 if (hsd->SdCard.CardType != CARD_SDHC_SDXC)
703 {
704 add *= BLOCKSIZE;
705 }
706
707 /* Configure the SD DPSM (Data Path State Machine) */
708 config.DataTimeOut = SDMMC_DATATIMEOUT;
709 config.DataLength = NumberOfBlocks * BLOCKSIZE;
710 config.DataBlockSize = SDMMC_DATABLOCK_SIZE_512B;
711 config.TransferDir = SDMMC_TRANSFER_DIR_TO_SDMMC;
712 config.TransferMode = SDMMC_TRANSFER_MODE_BLOCK;
713 config.DPSM = SDMMC_DPSM_DISABLE;
714 (void)SDMMC_ConfigData(hsd->Instance, &config);
715 __SDMMC_CMDTRANS_ENABLE(hsd->Instance);
716
717 /* Read block(s) in polling mode */
718 if (NumberOfBlocks > 1U)
719 {
720 hsd->Context = SD_CONTEXT_READ_MULTIPLE_BLOCK;
721
722 /* Read Multi Block command */
723 errorstate = SDMMC_CmdReadMultiBlock(hsd->Instance, add);
724 }
725 else
726 {
727 hsd->Context = SD_CONTEXT_READ_SINGLE_BLOCK;
728
729 /* Read Single Block command */
730 errorstate = SDMMC_CmdReadSingleBlock(hsd->Instance, add);
731 }
732 if (errorstate != HAL_SD_ERROR_NONE)
733 {
734 /* Clear all the static flags */
735 __HAL_SD_CLEAR_FLAG(hsd, SDMMC_STATIC_FLAGS);
736 hsd->ErrorCode |= errorstate;
737 hsd->State = HAL_SD_STATE_READY;
738 hsd->Context = SD_CONTEXT_NONE;
739 return HAL_ERROR;
740 }
741
742 /* Poll on SDMMC flags */
743 dataremaining = config.DataLength;
744 while (!__HAL_SD_GET_FLAG(hsd, SDMMC_FLAG_RXOVERR | SDMMC_FLAG_DCRCFAIL | SDMMC_FLAG_DTIMEOUT | SDMMC_FLAG_DATAEND))
745 {
746 if (__HAL_SD_GET_FLAG(hsd, SDMMC_FLAG_RXFIFOHF) && (dataremaining >= SDMMC_FIFO_SIZE))
747 {
748 /* Read data from SDMMC Rx FIFO */
749 for (count = 0U; count < (SDMMC_FIFO_SIZE / 4U); count++)
750 {
751 data = SDMMC_ReadFIFO(hsd->Instance);
752 *tempbuff = (uint8_t)(data & 0xFFU);
753 tempbuff++;
754 *tempbuff = (uint8_t)((data >> 8U) & 0xFFU);
755 tempbuff++;
756 *tempbuff = (uint8_t)((data >> 16U) & 0xFFU);
757 tempbuff++;
758 *tempbuff = (uint8_t)((data >> 24U) & 0xFFU);
759 tempbuff++;
760 }
761 dataremaining -= SDMMC_FIFO_SIZE;
762 }
763
764 if (((HAL_GetTick() - tickstart) >= Timeout) || (Timeout == 0U))
765 {
766 /* Clear all the static flags */
767 __HAL_SD_CLEAR_FLAG(hsd, SDMMC_STATIC_FLAGS);
768 hsd->ErrorCode |= HAL_SD_ERROR_TIMEOUT;
769 hsd->State = HAL_SD_STATE_READY;
770 hsd->Context = SD_CONTEXT_NONE;
771 return HAL_TIMEOUT;
772 }
773 }
774 __SDMMC_CMDTRANS_DISABLE(hsd->Instance);
775
776 /* Send stop transmission command in case of multiblock read */
777 if (__HAL_SD_GET_FLAG(hsd, SDMMC_FLAG_DATAEND) && (NumberOfBlocks > 1U))
778 {
779 if (hsd->SdCard.CardType != CARD_SECURED)
780 {
781 /* Send stop transmission command */
782 errorstate = SDMMC_CmdStopTransfer(hsd->Instance);
783 if (errorstate != HAL_SD_ERROR_NONE)
784 {
785 /* Clear all the static flags */
786 __HAL_SD_CLEAR_FLAG(hsd, SDMMC_STATIC_FLAGS);
787 hsd->ErrorCode |= errorstate;
788 hsd->State = HAL_SD_STATE_READY;
789 hsd->Context = SD_CONTEXT_NONE;
790 return HAL_ERROR;
791 }
792 }
793 }
794
795 /* Get error state */
796 if (__HAL_SD_GET_FLAG(hsd, SDMMC_FLAG_DTIMEOUT))
797 {
798 /* Clear all the static flags */
799 __HAL_SD_CLEAR_FLAG(hsd, SDMMC_STATIC_FLAGS);
800 hsd->ErrorCode |= HAL_SD_ERROR_DATA_TIMEOUT;
801 hsd->State = HAL_SD_STATE_READY;
802 hsd->Context = SD_CONTEXT_NONE;
803 return HAL_ERROR;
804 }
805 else if (__HAL_SD_GET_FLAG(hsd, SDMMC_FLAG_DCRCFAIL))
806 {
807 /* Clear all the static flags */
808 __HAL_SD_CLEAR_FLAG(hsd, SDMMC_STATIC_FLAGS);
809 hsd->ErrorCode |= HAL_SD_ERROR_DATA_CRC_FAIL;
810 hsd->State = HAL_SD_STATE_READY;
811 hsd->Context = SD_CONTEXT_NONE;
812 return HAL_ERROR;
813 }
814 else if (__HAL_SD_GET_FLAG(hsd, SDMMC_FLAG_RXOVERR))
815 {
816 /* Clear all the static flags */
817 __HAL_SD_CLEAR_FLAG(hsd, SDMMC_STATIC_FLAGS);
818 hsd->ErrorCode |= HAL_SD_ERROR_RX_OVERRUN;
819 hsd->State = HAL_SD_STATE_READY;
820 hsd->Context = SD_CONTEXT_NONE;
821 return HAL_ERROR;
822 }
823 else
824 {
825 /* Nothing to do */
826 }
827
828 /* Clear all the static flags */
829 __HAL_SD_CLEAR_FLAG(hsd, SDMMC_STATIC_DATA_FLAGS);
830
831 hsd->State = HAL_SD_STATE_READY;
832
833 return HAL_OK;
834 }
835 else
836 {
837 hsd->ErrorCode |= HAL_SD_ERROR_BUSY;
838 return HAL_ERROR;
839 }
840}

854HAL_StatusTypeDef HAL_SD_WriteBlocks(SD_HandleTypeDef *hsd, const uint8_t *pData, uint32_t BlockAdd,
855 uint32_t NumberOfBlocks, uint32_t Timeout)
856{
857 SDMMC_DataInitTypeDef config;
858 uint32_t errorstate;
859 uint32_t tickstart = HAL_GetTick();
860 uint32_t count;
861 uint32_t data;
862 uint32_t dataremaining;
863 uint32_t add = BlockAdd;
864 const uint8_t *tempbuff = pData;
865
866 if (NULL == pData)
867 {
868 hsd->ErrorCode |= HAL_SD_ERROR_PARAM;
869 return HAL_ERROR;
870 }
871
872 if (hsd->State == HAL_SD_STATE_READY)
873 {
874 hsd->ErrorCode = HAL_SD_ERROR_NONE;
875
876 if ((add + NumberOfBlocks) > (hsd->SdCard.LogBlockNbr))
877 {
878 hsd->ErrorCode |= HAL_SD_ERROR_ADDR_OUT_OF_RANGE;
879 return HAL_ERROR;
880 }
881
882 hsd->State = HAL_SD_STATE_BUSY;
883
884 /* Initialize data control register */
885 hsd->Instance->DCTRL = 0U;
886
887 if (hsd->SdCard.CardType != CARD_SDHC_SDXC)
888 {
889 add *= BLOCKSIZE;
890 }
891
892 /* Configure the SD DPSM (Data Path State Machine) */
893 config.DataTimeOut = SDMMC_DATATIMEOUT;
894 config.DataLength = NumberOfBlocks * BLOCKSIZE;
895 config.DataBlockSize = SDMMC_DATABLOCK_SIZE_512B;
896 config.TransferDir = SDMMC_TRANSFER_DIR_TO_CARD;
897 config.TransferMode = SDMMC_TRANSFER_MODE_BLOCK;
898 config.DPSM = SDMMC_DPSM_DISABLE;
899 (void)SDMMC_ConfigData(hsd->Instance, &config);
900 __SDMMC_CMDTRANS_ENABLE(hsd->Instance);
901
902 /* Write Blocks in Polling mode */
903 if (NumberOfBlocks > 1U)
904 {
905 hsd->Context = SD_CONTEXT_WRITE_MULTIPLE_BLOCK;
906
907 /* Write Multi Block command */
908 errorstate = SDMMC_CmdWriteMultiBlock(hsd->Instance, add);
909 }
910 else
911 {
912 hsd->Context = SD_CONTEXT_WRITE_SINGLE_BLOCK;
913
914 /* Write Single Block command */
915 errorstate = SDMMC_CmdWriteSingleBlock(hsd->Instance, add);
916 }
917 if (errorstate != HAL_SD_ERROR_NONE)
918 {
919 /* Clear all the static flags */
920 __HAL_SD_CLEAR_FLAG(hsd, SDMMC_STATIC_FLAGS);
921 hsd->ErrorCode |= errorstate;
922 hsd->State = HAL_SD_STATE_READY;
923 hsd->Context = SD_CONTEXT_NONE;
924 return HAL_ERROR;
925 }
926
927 /* Write block(s) in polling mode */
928 dataremaining = config.DataLength;
929 while (!__HAL_SD_GET_FLAG(hsd, SDMMC_FLAG_TXUNDERR | SDMMC_FLAG_DCRCFAIL | SDMMC_FLAG_DTIMEOUT |
930 SDMMC_FLAG_DATAEND))
931 {
932 if (__HAL_SD_GET_FLAG(hsd, SDMMC_FLAG_TXFIFOHE) && (dataremaining >= SDMMC_FIFO_SIZE))
933 {
934 /* Write data to SDMMC Tx FIFO */
935 for (count = 0U; count < (SDMMC_FIFO_SIZE / 4U); count++)
936 {
937 data = (uint32_t)(*tempbuff);
938 tempbuff++;
939 data |= ((uint32_t)(*tempbuff) << 8U);
940 tempbuff++;
941 data |= ((uint32_t)(*tempbuff) << 16U);
942 tempbuff++;
943 data |= ((uint32_t)(*tempbuff) << 24U);
944 tempbuff++;
945 (void)SDMMC_WriteFIFO(hsd->Instance, &data);
946 }
947 dataremaining -= SDMMC_FIFO_SIZE;
948 }
949
950 if (((HAL_GetTick() - tickstart) >= Timeout) || (Timeout == 0U))
951 {
952 /* Clear all the static flags */
953 __HAL_SD_CLEAR_FLAG(hsd, SDMMC_STATIC_FLAGS);
954 hsd->ErrorCode |= errorstate;
955 hsd->State = HAL_SD_STATE_READY;
956 hsd->Context = SD_CONTEXT_NONE;
957 return HAL_TIMEOUT;
958 }
959 }
960 __SDMMC_CMDTRANS_DISABLE(hsd->Instance);
961
962 /* Send stop transmission command in case of multiblock write */
963 if (__HAL_SD_GET_FLAG(hsd, SDMMC_FLAG_DATAEND) && (NumberOfBlocks > 1U))
964 {
965 if (hsd->SdCard.CardType != CARD_SECURED)
966 {
967 /* Send stop transmission command */
968 errorstate = SDMMC_CmdStopTransfer(hsd->Instance);
969 if (errorstate != HAL_SD_ERROR_NONE)
970 {
971 /* Clear all the static flags */
972 __HAL_SD_CLEAR_FLAG(hsd, SDMMC_STATIC_FLAGS);
973 hsd->ErrorCode |= errorstate;
974 hsd->State = HAL_SD_STATE_READY;
975 hsd->Context = SD_CONTEXT_NONE;
976 return HAL_ERROR;
977 }
978 }
979 }
980
981 /* Get error state */
982 if (__HAL_SD_GET_FLAG(hsd, SDMMC_FLAG_DTIMEOUT))
983 {
984 /* Clear all the static flags */
985 __HAL_SD_CLEAR_FLAG(hsd, SDMMC_STATIC_FLAGS);
986 hsd->ErrorCode |= HAL_SD_ERROR_DATA_TIMEOUT;
987 hsd->State = HAL_SD_STATE_READY;
988 hsd->Context = SD_CONTEXT_NONE;
989 return HAL_ERROR;
990 }
991 else if (__HAL_SD_GET_FLAG(hsd, SDMMC_FLAG_DCRCFAIL))
992 {
993 /* Clear all the static flags */
994 __HAL_SD_CLEAR_FLAG(hsd, SDMMC_STATIC_FLAGS);
995 hsd->ErrorCode |= HAL_SD_ERROR_DATA_CRC_FAIL;
996 hsd->State = HAL_SD_STATE_READY;
997 hsd->Context = SD_CONTEXT_NONE;
998 return HAL_ERROR;
999 }
1000 else if (__HAL_SD_GET_FLAG(hsd, SDMMC_FLAG_TXUNDERR))

1428HAL_StatusTypeDef HAL_SD_Erase(SD_HandleTypeDef *hsd, uint32_t BlockStartAdd, uint32_t BlockEndAdd)

1435 {

1437

1439 {

1441 return HAL_ERROR;
1442 }
1443

1445 {

1447 return HAL_ERROR;
1448 }

1451

1453 if (((hsd->SdCard.Class) & SDMMC_CCCC_ERASE) == 0U)
1454 {

2840HAL_StatusTypeDef HAL_SD_Abort(SD_HandleTypeDef *hsd)
2841{
2842 uint32_t error_code;
2843 uint32_t tickstart;
2844
2845 if (hsd->State == HAL_SD_STATE_BUSY)
2846 {
2847 /* DIsable All interrupts */
2848 __HAL_SD_DISABLE_IT(hsd, SDMMC_IT_DATAEND | SDMMC_IT_DCRCFAIL | SDMMC_IT_DTIMEOUT | \
2849 SDMMC_IT_TXUNDERR | SDMMC_IT_RXOVERR);
2850 __SDMMC_CMDTRANS_DISABLE(hsd->Instance);
2851
2852 /*we will send the CMD12 in all cases in order to stop the data transfers*/
2853 /*In case the data transfer just finished , the external memory will not respond
2854 and will return HAL_SD_ERROR_CMD_RSP_TIMEOUT*/
2855 /*In case the data transfer aborted , the external memory will respond and will return HAL_SD_ERROR_NONE*/
2856 /*Other scenario will return HAL_ERROR*/
2857
2858 hsd->ErrorCode = SDMMC_CmdStopTransfer(hsd->Instance);
2859 error_code = hsd->ErrorCode;
2860 if ((error_code != HAL_SD_ERROR_NONE) && (error_code != HAL_SD_ERROR_CMD_RSP_TIMEOUT))
2861 {
2862 return HAL_ERROR;
2863 }
2864
2865 tickstart = HAL_GetTick();
2866 if ((hsd->Instance->DCTRL & SDMMC_DCTRL_DTDIR) == SDMMC_TRANSFER_DIR_TO_CARD)
2867 {
2868 if (hsd->ErrorCode == HAL_SD_ERROR_NONE)
2869 {
2870 while (!__HAL_SD_GET_FLAG(hsd, SDMMC_FLAG_DABORT | SDMMC_FLAG_BUSYD0END))
2871 {
2872 if ((HAL_GetTick() - tickstart) >= SDMMC_SWDATATIMEOUT)
2873 {
2874 hsd->ErrorCode = HAL_SD_ERROR_TIMEOUT;
2875 hsd->State = HAL_SD_STATE_READY;
2876 return HAL_TIMEOUT;
2877 }
2878 }
2879 }
2880
2881 if (hsd->ErrorCode == HAL_SD_ERROR_CMD_RSP_TIMEOUT)
2882 {
2883 while (!__HAL_SD_GET_FLAG(hsd, SDMMC_FLAG_DATAEND))
2884 {
2885 if ((HAL_GetTick() - tickstart) >= SDMMC_SWDATATIMEOUT)
2886 {
2887 hsd->ErrorCode = HAL_SD_ERROR_TIMEOUT;
2888 hsd->State = HAL_SD_STATE_READY;
2889 return HAL_TIMEOUT;
2890 }
2891 }
2892 }
2893 }
2894 else if ((hsd->Instance->DCTRL & SDMMC_DCTRL_DTDIR) == SDMMC_TRANSFER_DIR_TO_SDMMC)
2895 {
2896 while (!__HAL_SD_GET_FLAG(hsd, SDMMC_FLAG_DABORT | SDMMC_FLAG_DATAEND))
2897 {
2898 if ((HAL_GetTick() - tickstart) >= SDMMC_SWDATATIMEOUT)
2899 {
2900 hsd->ErrorCode = HAL_SD_ERROR_TIMEOUT;
2901 hsd->State = HAL_SD_STATE_READY;
2902 return HAL_TIMEOUT;
2903 }
2904 }
2905 }
2906 else
2907 {
2908 /* Nothing to do*/
2909 }
2910
2911 /*The reason of all these while conditions previously is that we need to wait the SDMMC and clear
2912 the appropriate flags that will be set depending of the abort/non abort of the memory */
2913 /*Not waiting the SDMMC flags will cause the next SDMMC_DISABLE_IDMA to not get cleared
2914 and will result in next SDMMC read/write operation to fail */
2915
2916 /*SDMMC ready for clear data flags*/
2917 __HAL_SD_CLEAR_FLAG(hsd, SDMMC_FLAG_BUSYD0END);
2918 __HAL_SD_CLEAR_FLAG(hsd, SDMMC_STATIC_DATA_FLAGS);

2920 hsd->Instance->IDMACTRL = SDMMC_DISABLE_IDMA;
2921
2922 hsd->State = HAL_SD_STATE_READY;
2923
2924 /* Initialize the SD operation */
2925 hsd->Context = SD_CONTEXT_NONE;
2926 }

2928}
