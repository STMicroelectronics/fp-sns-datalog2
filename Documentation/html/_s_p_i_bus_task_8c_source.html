424static sys_error_code_t SPIBusTaskExecuteStep(AManagedTask *_this)
425{
426 assert_param(_this);
427 sys_error_code_t res = SYS_NO_ERROR_CODE;
428 SPIBusTask *p_obj = (SPIBusTask *) _this;
429
430 struct spiIOMessage_t xMsg =
431 {
432 0
433 };
434 AMTExSetInactiveState((AManagedTaskEx *) _this, TRUE);
435 if (TX_SUCCESS == tx_queue_receive(&p_obj->in_queue, &xMsg, TX_WAIT_FOREVER))
436 {
437 AMTExSetInactiveState((AManagedTaskEx *) _this, FALSE);
438 switch (xMsg.messageId)
439 {
440 case SM_MESSAGE_ID_FORCE_STEP:
441 __NOP();
442 // do nothing. I need only to resume the task.
443 break;
444
445 case SM_MESSAGE_ID_SPI_BUS_READ:
446 SPIMasterDriverSelectDevice((SPIMasterDriver_t *) p_obj->p_driver, xMsg.pxSensor->p_cs_gpio_port,
447 xMsg.pxSensor->cs_gpio_pin);
448 res = IIODrvRead(p_obj->p_driver, xMsg.pnData, xMsg.nDataSize, xMsg.nRegAddr);
449 SPIMasterDriverDeselectDevice((SPIMasterDriver_t *) p_obj->p_driver, xMsg.pxSensor->p_cs_gpio_port,
450 xMsg.pxSensor->cs_gpio_pin);
451 if (!SYS_IS_ERROR_CODE(res))
452 {
453 res = SPIBusIFNotifyIOComplete(xMsg.pxSensor);
454 }
455 break;
456
457 case SM_MESSAGE_ID_SPI_BUS_WRITE:
458 SPIMasterDriverSelectDevice((SPIMasterDriver_t *) p_obj->p_driver, xMsg.pxSensor->p_cs_gpio_port,
459 xMsg.pxSensor->cs_gpio_pin);
460 res = IIODrvWrite(p_obj->p_driver, xMsg.pnData, xMsg.nDataSize, xMsg.nRegAddr);
461 SPIMasterDriverDeselectDevice((SPIMasterDriver_t *) p_obj->p_driver, xMsg.pxSensor->p_cs_gpio_port,
462 xMsg.pxSensor->cs_gpio_pin);
463 if (!SYS_IS_ERROR_CODE(res))
464 {
465 res = SPIBusIFNotifyIOComplete(xMsg.pxSensor);
466 }
467 break;
468
469 default:
470 //TODO: STF - need to notify the error
471 break;
472 }
473 }
474
475 return res;
476}
