266static sys_error_code_t HelloWorldTaskExecuteStep(AManagedTask *_this)
267{
268 assert_param(_this != NULL);
269 sys_error_code_t res = SYS_NO_ERROR_CODE;
270 HelloWorldTask_t *p_obj = (HelloWorldTask_t *)_this;
271 static bool half_period = true;
272 MX_GPIOParams_t *p_led_params = (MX_GPIOParams_t *)p_obj->pLEDConfigMX;
273 tx_thread_sleep(AMT_MS_TO_TICKS(500));
274
275 if (!half_period)
276 {
277 p_obj->counter++;
278 EPowerMode active_power_mode = SysGetPowerStatus().active_power_mode;
279 if (active_power_mode == E_POWER_MODE_STATE1)
280 {
281// SYS_DEBUGF(SYS_DBG_LEVEL_VERBOSE, ("Hello NUCLEO-U575ZI-Q %d!!\r\n", p_obj->counter));
282 }
283 else if (active_power_mode == E_POWER_MODE_SENSORS_ACTIVE)
284 {
285 if (p_led_params != NULL)
286 {
287 HAL_GPIO_TogglePin(p_led_params->port, p_led_params->pin);
288 }
289 }
290 }
291
292 __NOP();
293 half_period = !half_period;
294
295 return res;
296}
