618sys_error_code_t DatalogAppTask_OnNewDataReady_vtbl(IEventListener *_this, const DataEvent_t *p_evt)
619{
620 sys_error_code_t res = SYS_NO_ERROR_CODE;
621 DatalogAppTask *p_obj = (DatalogAppTask *) IEventListenerGetOwner(_this);
622
623 uint8_t nPMState = (uint8_t) AMTGetTaskPowerMode((AManagedTask *) p_obj);
624 uint16_t sId = p_evt->tag;
625 uint32_t nBytesPerSample = SMGetnBytesPerSample(sId);
626 uint8_t *data_buf = EMD_Data(p_evt->p_data);
627 uint32_t samplesToSend = EMD_GetPayloadSize(p_evt->p_data) / nBytesPerSample;
628
629 if (nPMState == E_POWER_MODE_SENSORS_ACTIVE)
630 {
631 uint8_t stream_id = p_obj->datalog_model->s_models[sId]->stream_params.stream_id;
632 if (res != 0)
633 {
634 SYS_SET_SERVICE_LEVEL_ERROR_CODE(res);
635 return res;
636 }
637
638 if (p_obj->sensorContext[sId].old_time_stamp == -1.0f)
639 {
640 p_obj->datalog_model->s_models[sId]->stream_params.ioffset = p_evt->timestamp;
641 p_obj->sensorContext[sId].old_time_stamp = p_evt->timestamp;
642 p_obj->sensorContext[sId].n_samples_to_timestamp = p_obj->datalog_model->s_models[sId]->stream_params.spts;
643 }
644 else
645 {
646 while (samplesToSend > 0)
647 {
648 /* n_samples_to_timestamp = 0 if user setup spts = 0 (no timestamp needed) */
649 if (p_obj->sensorContext[sId].n_samples_to_timestamp == 0
650 || samplesToSend < p_obj->sensorContext[sId].n_samples_to_timestamp)
651 {
652 if (IStream_is_enabled((IStream_t *) p_obj->usbx_device))
653 {
654 res = IStream_post_data((IStream_t *) p_obj->usbx_device, stream_id, data_buf, samplesToSend * nBytesPerSample);
655 }
656 if (IStream_is_enabled((IStream_t *) p_obj->filex_device))
657 {
658 res = IStream_post_data((IStream_t *) p_obj->filex_device, stream_id, data_buf, samplesToSend * nBytesPerSample);
659 }
660 if (IStream_is_enabled((IStream_t *) p_obj->ble_device))
661 {
662 res = IStream_post_data((IStream_t *) p_obj->ble_device, sId, data_buf, samplesToSend * nBytesPerSample);
663 }
664 if (p_obj->sensorContext[sId].n_samples_to_timestamp != 0)
665 {
666 p_obj->sensorContext[sId].n_samples_to_timestamp -= samplesToSend;
667 }
668 samplesToSend = 0;
669 }
670 else
671 {
672 if (IStream_is_enabled((IStream_t *) p_obj->usbx_device))
673 {
674 res = IStream_post_data((IStream_t *) p_obj->usbx_device, stream_id, data_buf, p_obj->sensorContext[sId].n_samples_to_timestamp * nBytesPerSample);
675 }
676 if (IStream_is_enabled((IStream_t *) p_obj->filex_device))
677 {
678 res = IStream_post_data((IStream_t *) p_obj->filex_device, stream_id, data_buf, p_obj->sensorContext[sId].n_samples_to_timestamp * nBytesPerSample);
679 }
680 if (IStream_is_enabled((IStream_t *) p_obj->ble_device))
681 {
682 res = IStream_post_data((IStream_t *) p_obj->ble_device, sId, data_buf, p_obj->sensorContext[sId].n_samples_to_timestamp * nBytesPerSample);
683 }
684
685 data_buf += p_obj->sensorContext[sId].n_samples_to_timestamp * nBytesPerSample;
686 samplesToSend -= p_obj->sensorContext[sId].n_samples_to_timestamp;
687
688 float measuredODR;
689 double newTS;
690
691 SensorStatus_t sensor_status = SMSensorGetStatus(sId);
692 if (sensor_status.isensor_class == ISENSOR_CLASS_MEMS)
693 {
694 measuredODR = sensor_status.type.mems.measured_odr;
695 }
696 else if (sensor_status.isensor_class == ISENSOR_CLASS_AUDIO)
697 {
698 measuredODR = sensor_status.type.audio.frequency;
699 }
700 else
701 {
702 SYS_SET_SERVICE_LEVEL_ERROR_CODE(res);
703 return res;
704 }
705
706 if (measuredODR != 0.0f)
707 {

1282 if (!(sTaskObj.datalog_model->log_controller_model.status

1299

1301 {
