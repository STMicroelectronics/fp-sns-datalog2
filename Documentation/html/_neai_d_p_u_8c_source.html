72sys_error_code_t NeaiDPU_vtblProcess(IDPU2_t *_this, EMData_t in_data, EMData_t out_data)
73{
74 assert_param(_this != NULL);
75 sys_error_code_t res = SYS_NO_ERROR_CODE;
76 NeaiDPU_t *p_obj = (NeaiDPU_t*)_this;
77 float *p_out = (float*)EMD_Data(&out_data);
78
79 enum neai_state status;
80 float *p_signal = (float*)EMD_Data(&in_data);
81 if (p_obj->proc_mode == E_NEAI_ANOMALY_LEARN && p_obj->proc.anomalyLearn)
82 {
83 status = p_obj->proc.anomalyLearn(p_signal);
84 p_out[0] = (float)status;
85 }
86 else if (p_obj->proc_mode == E_NEAI_ANOMALY_DETECT && p_obj->proc.anomalyDetect)
87 {
88 uint8_t proc_out;
89 status = p_obj->proc.anomalyDetect(p_signal, &proc_out);
90 p_out[0] = (float)status;
91 p_out[1] = (float)proc_out;
92 }
93 else if (p_obj->proc_mode == E_NEAI_CLASSIFICATION && p_obj->proc.classification)
94 {
95 uint16_t id_class;
96 status = p_obj->proc.classification(p_signal, &p_out[2], &id_class);
97 p_out[0] = (float)status;
98 p_out[1] = (float)id_class;
99 }
100 else
101 {
102 SYS_DEBUGF(SYS_DBG_LEVEL_WARNING, ("NEAI: mode (%d) not initialized\r\n", p_obj->proc_mode));
103 res = SYS_UNDEFINED_ERROR_CODE;
104 }
105
106#if defined(SYS_DEBUG)
107 if (!SYS_IS_ERROR_CODE(res))
108 {
109 SYS_DEBUGF(SYS_DBG_LEVEL_ALL, ("NEAI: mode (%d), status (%f)\r\n", p_obj->proc_mode, p_out[0]));
110
111 if (!(status == NEAI_NOT_ENOUGH_CALL_TO_LEARNING) && !(status == NEAI_OK))
112 {
113 if (status == NEAI_INIT_FCT_NOT_CALLED)
114 SYS_DEBUGF(SYS_DBG_LEVEL_WARNING, ("NEAI: Init function not called\r\n"));
115 else if (status == NEAI_NOT_ENOUGH_CALL_TO_LEARNING)
116 SYS_DEBUGF(SYS_DBG_LEVEL_WARNING, ("NEAI: need more learning signals \r\n"));
117 else
118 SYS_DEBUGF(SYS_DBG_LEVEL_WARNING, ("NEAI: (mode %d) status: %d \r\n", p_obj->proc_mode, status));
119 }
120 }
121#endif
122
123 return res;
124}

168sys_error_code_t NeaiDPU_SetProcessingMode(NeaiDPU_t *_this, ENeaiMode_t mode)
169{
170 assert_param(_this != NULL);
171 sys_error_code_t res = SYS_NO_ERROR_CODE;
172 uint8_t in_axes;
173 uint16_t in_signal_size, out_signal_size;
174
175 if ((mode == E_NEAI_ANOMALY_DETECT) || (mode == E_NEAI_ANOMALY_LEARN) || (mode == E_NEAI_CLASSIFICATION))
176 {
177 _this->proc_mode = mode;
178 }
179
180 switch (mode)
181 {
182 case E_NEAI_ANOMALY_LEARN:
183 case E_NEAI_ANOMALY_DETECT:
184 in_axes = AXIS_NUMBER;
185 in_signal_size = DATA_INPUT_USER;
186 out_signal_size = 2;
187 break;
188
189 case E_NEAI_CLASSIFICATION:
190 in_axes = AXIS_NUMBER_NCC;
191 in_signal_size = DATA_INPUT_USER_NCC;
192 out_signal_size = 2 + CLASS_NUMBER_NCC;
193 break;
194
195 default:
196 res = SYS_NOT_IMPLEMENTED_ERROR_CODE;
197 SYS_SET_SERVICE_LEVEL_ERROR_CODE(SYS_NOT_IMPLEMENTED_ERROR_CODE);
198 break;
199 }
200
201 if (!SYS_IS_ERROR_CODE(res))
202 {
203 /* Prepare the EMData_t to initialize the base class.*/
204 EMData_t in_data, out_data;
205 uint8_t *p_in_payload = _this->super.in_data.p_payload;
206 uint8_t *p_out_payload = _this->super.out_data.p_payload;
207 if (in_axes == 1)
208 {
209 res = EMD_Init(&in_data, p_in_payload, E_EM_FLOAT, E_EM_MODE_LINEAR, 1, in_signal_size);
210
211 }
212 else
213 {
214 res = EMD_Init(&in_data, p_in_payload, E_EM_FLOAT, E_EM_MODE_INTERLEAVED, 2, in_signal_size, in_axes);
215 }
216 if (SYS_IS_ERROR_CODE(res))
217 {
218 sys_error_handler();
219 }
220
221 res = EMD_Init(&out_data, p_out_payload, E_EM_FLOAT, E_EM_MODE_LINEAR, 1, out_signal_size);
222 if (SYS_IS_ERROR_CODE(res))
223 {
224 sys_error_handler();
225 }
226
227 _this->super.in_data = in_data;
228 _this->super.out_data = out_data;
229 }
230
231 return res;
232}
