48BleCharTypeDef *BLE_InitGeneralPurposeService(uint8_t Size)
49{
50 /* Data structure pointer for BLE service */
51 BleCharTypeDef *BleCharPointer;
52 uint8_t GP_CharNum = (uint8_t)NumberAllocatedGP;
53
54 /* Some Controls */
55 if ((uint8_t)NumberAllocatedGP >= BLE_GENERAL_PURPOSE_MAX_CHARS_NUM)
56 {
57 BLE_MANAGER_PRINTF("Error GP_CharNum must be < %d\r\n", BLE_GENERAL_PURPOSE_MAX_CHARS_NUM);
58 return NULL;
59 }
60 /* Increment the Number of Allocated General Purpose Features */
61 NumberAllocatedGP++;
62
63 /* Init data structure pointer for General Purpose info service */
64 BleCharPointer = &BleCharGeneralPurpose[GP_CharNum];
65 memset(BleCharPointer, 0, sizeof(BleCharTypeDef));
66 BleCharPointer->AttrMod_Request_CB = AttrMod_Request_GeneralPurpose;
67 COPY_GP_CHAR_UUID((BleCharPointer->uuid));

69 BleCharPointer->Char_UUID_Type = UUID_TYPE_128;
70 BleCharPointer->Char_Value_Length = Size;
71 BleCharPointer->Char_Properties = (uint8_t)CHAR_PROP_NOTIFY;
72 BleCharPointer->Security_Permissions = ATTR_PERMISSION_NONE;
73 BleCharPointer->GATT_Evt_Mask = GATT_NOTIFY_READ_REQ_AND_WAIT_FOR_APPL_RESP;
74 BleCharPointer->Enc_Key_Size = 16;
75 BleCharPointer->Is_Variable = 0;
76
77 BLE_MANAGER_PRINTF("BLE General Purpose feature [%d] ok\r\n", GP_CharNum);
