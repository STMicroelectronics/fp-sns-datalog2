72void ExtiDefISR(uint16_t exti_pin)
73{
74 EXTIPin2CallbckMap xMap = EXTI_GET_P2F_MAP();
75 for (int i = 0; xMap[i].pfCallback != NULL; i++)
76 {
77 if (__HAL_GPIO_EXTI_GET_IT(xMap[i].nPin) && exti_pin == xMap[i].nPin)
78 {
79 /* EXTI line interrupt detected */
80 __HAL_GPIO_EXTI_CLEAR_IT(xMap[i].nPin);
81 xMap[i].pfCallback(xMap[i].nPin);
82 }
83 }
84}
