65sys_error_code_t AI_HAR_DPU_vtblProcess(IDPU2_t *_this, EMData_t in_data, EMData_t out_data)
66{
67 assert_param(_this != NULL);
68 sys_error_code_t res = SYS_NO_ERROR_CODE;
69 AI_HAR_DPU_t *p_obj = (AI_HAR_DPU_t*)_this;
70
71 float scale = p_obj->scale;
72 float *p_in = (float*)EMD_Data(&in_data);
73 float *p_out = (float*)EMD_Data(&out_data);
74 GRAV_input_t gravIn[AI_HAR_DPU_NB_SAMPLE_IN];
75 GRAV_input_t gravOut[AI_HAR_DPU_NB_SAMPLE_IN];
76
77 for (int i=0 ; i < AI_HAR_DPU_NB_SAMPLE_IN ; i++)
78 {
79 gravIn[i].AccX = *p_in++ * scale;
80 gravIn[i].AccY = *p_in++ * scale;
81 gravIn[i].AccZ = *p_in++ * scale;
82 gravOut[i] = gravity_suppress_rotate (&gravIn[i]);
83 }
84
85 /* call Ai library. */
86 p_obj->ai_processing_f(AI_HAR_DPU_NAME, (float *)gravOut, p_out);
87
88 return res;
89}
