112void SystemClock_Restore(void)
113{
114 /* SEQUENCE:
115 * 1. Set PWR regulator to SCALE1_BOOST
116 * 2. PLL ON
117 * 3. Set SYSCLK source to PLL
118 *
119 * NOTE:
120 * Do not change or update the base-clock source (e.g. MSI and LSE)
121 */
122
123 if (HAL_PWREx_ControlVoltageScaling(PWR_REGULATOR_VOLTAGE_SCALE1) != HAL_OK)
124 {
125 sys_error_handler();
126 }
127
128 if (__HAL_RCC_GET_SYSCLK_SOURCE() != RCC_SYSCLKSOURCE_STATUS_PLLCLK)
129 {
130 if (HAL_RCC_OscConfig(&(sSystemClock.osc)) != HAL_OK)
131 {
132 sys_error_handler();
133 }
134 }
135
136 if (HAL_RCC_ClockConfig(&(sSystemClock.clock), sSystemClock.latency) != HAL_OK)
137 {
138 sys_error_handler();
139 }
140}
