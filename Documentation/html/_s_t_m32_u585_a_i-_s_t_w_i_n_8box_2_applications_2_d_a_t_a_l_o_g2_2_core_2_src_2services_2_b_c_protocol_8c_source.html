181sys_error_code_t BCPSendCmd(BCProtocol_t *_this, ESTBC02_SW_CMD_t cmd)
182{
183 assert_param(_this != NULL);
184 sys_error_code_t res = SYS_NO_ERROR_CODE;
185 uint32_t start_tick = HAL_GetTick();
186 bool timeout = false;
187 BCTimerDriver_t *p_driver = NULL;
188 SYS_DECLARE_CS(cs);
189
190 _this->cmd = cmd;
191 _this->sw_state = E_BC_START;
192
193 p_driver = (BCTimerDriver_t *) _this->p_bc_tim_sw_driver;
194 BCDriverSetTimARR(p_driver, BC_STBC02_SW_START_PULSE_US - 1U);
195
196 /* The timer should start immediately after the pin is SET --> Critical Section */
197 SYS_ENTER_CRITICAL(cs);
198
199 /* SW Pin set */
200 BCDriverSetLine1W(p_driver);
201
202 /* Start the driver for Single Wire communication */
203 IDrvStart(_this->p_bc_tim_sw_driver);
204
205 /* Clear the first irq flag which is set immediately */
206 __HAL_TIM_CLEAR_FLAG(p_driver->mx_handle.p_mx_tim_cfg->p_tim, TIM_FLAG_UPDATE);
207
208 SYS_EXIT_CRITICAL(cs);
209
210
211 while ((_this->sw_state != E_BC_IDLE) && (!timeout))
212 {
213 /* 10ms timeout */
214 if (HAL_GetTick() - start_tick > 10U)
215 {
216 timeout = true;
217 res = SYS_BCP_CMD_EXECUTION_ERROR_CODE;
218 SYS_SET_SERVICE_LEVEL_ERROR_CODE(SYS_BCP_CMD_EXECUTION_ERROR_CODE);
219 }
220 }
221
222 /* stop the time base*/
223 IDrvStop(_this->p_bc_tim_sw_driver);
224
225 return res;
226}

255sys_error_code_t BCPAcquireState(BCProtocol_t *_this, uint16_t *voltage, ESTBC02_STATE_t *state)
256{
257 assert_param(_this != NULL);
258 assert_param(voltage != NULL);
259 assert_param(state != NULL);
260
261 sys_error_code_t res = SYS_NO_ERROR_CODE;
262 BCProtocol_t *p_owner = sTimCallbackMap[0].p_owner;
263
264 /* Enable Li-Ion battery measurement */
265 BCPSendCmd(_this, E_BATMS_ON);
266
267 /* Get Battery Voltage */
268 BCPAcquireBatteryVoltage(_this, voltage);
269
270 /* Enable Li-Ion battery measurement */
271 BCPSendCmd(_this, E_BATMS_OFF);
272
273 if (p_owner->chg_irq_counter == p_owner->chg_irq_counter_prev)
274 {
275 /* CHG pin not toggling */
276 *state = DISCHARGING;
277 }
278 else if (*voltage < 100U)
279 {
280 /* Battery not connected */
281 *voltage = 0U;
282 *state = BATTERY_NOT_CONNECTED;
283 }
284 else
285 {
286 /* All other cases */
287 switch (_this->chg_state)
288 {
289 case NOT_VALID_INPUT:
290 *state = DISCHARGING;
291 break;
292 case VALID_INPUT:
293 *state = PLUGGED_NOT_CHARGING;
294 break;
295 case VBAT_LOW:
296 *state = LOW_BATTERY;
297 break;
298 case END_OF_CHARGE:
299 *state = PLUGGED_NOT_CHARGING;
300 break;
301 case CHARGING_PHASE:
302 *state = CHARGING;
303 break;
304 case OVERCHARGE_FAULT:
305 *state = STATE_ERROR;
306 break;
307 case CHARGING_TIMEOUT:
308 *state = STATE_ERROR;
309 HAL_GPIO_WritePin(C_EN_GPIO_Port, C_EN_Pin, GPIO_PIN_RESET);
310 break;
311 case BATTERY_VOLTAGE_BE_LOW_VPRE:
312 *state = STATE_ERROR;
313 HAL_GPIO_WritePin(C_EN_GPIO_Port, C_EN_Pin, GPIO_PIN_RESET);
314 break;
315 case CHARGING_THERMAL_LIMITATION:
316 *state = STATE_ERROR;
317 break;
318 case BATTERY_TEMPERATURE_FAULT:
319 *state = UNKNOWN;
320 break;
321 default:
322 *state = UNKNOWN;
323 break;
324 }
325 p_owner->chg_irq_counter_prev = p_owner->chg_irq_counter;
326 }
327
328 return res;
329}

334static void BCP1WStateMachine(TIM_HandleTypeDef *htim)
335{
336 assert_param(htim != NULL);
337
338 BCProtocol_t *p_owner = sTimCallbackMap[0].p_owner;
339 BCTimerDriver_t *p_driver = (BCTimerDriver_t *) p_owner->p_bc_tim_sw_driver;
340
341 switch (p_owner->sw_state)
342 {
343 case E_BC_IDLE:
344 break;
345 case E_BC_START: /* End of the start pulse */
346 p_owner->pulse = 0;
347 BCDriverResetLine1W(p_driver);
348 BCDriverSetTimARR(p_driver, BC_STBC02_SW_SHORT_PULSE_US - 1U);
349 p_owner->sw_state = E_BC_PULSE_H;
350 break;
351 case E_BC_PULSE_L: /* Beginning of short-low pulse (except first one) */
352 BCDriverSetTimARR(p_driver, BC_STBC02_SW_SHORT_PULSE_US - 1U);
353 BCDriverResetLine1W(p_driver);
354 if (p_owner->pulse < (uint8_t) p_owner->cmd)
355 {
356 p_owner->sw_state = E_BC_PULSE_H;
357 }
358 else
359 {
360 p_owner->sw_state = E_BC_STOP_H;
361 }
362 break;
363 case E_BC_PULSE_H: /* Beginning of short-high pulse */
364 BCDriverSetTimARR(p_driver, BC_STBC02_SW_SHORT_PULSE_US - 1U);
365 BCDriverSetLine1W(p_driver);
366 p_owner->pulse++;
367 p_owner->sw_state = E_BC_PULSE_L;
368 break;
369 case E_BC_STOP_H:
370 BCDriverSetTimARR(p_driver, BC_STBC02_SW_STOP_PULSE_US - 1U);
371 BCDriverSetLine1W(p_driver);
372 p_owner->sw_state = E_BC_STOP_L;
373 break;
374 case E_BC_STOP_L:
375 BCDriverResetLine1W(p_driver);
376 IDrvStop(p_owner->p_bc_tim_sw_driver);
377 p_owner->sw_state = E_BC_IDLE;
378 break;
379 default:
380 break;
381 }
382}

385static void BCPComputeFrequency(TIM_HandleTypeDef *htim)
386{
387 assert_param(htim != NULL);
388
389 BCProtocol_t *p_owner = sTimCallbackMap[0].p_owner;
390 float uwFrequency = 0; /* Frequency Value [Hz] */
391
392 p_owner->chg_irq_counter++;
393 if (p_owner->ic_flag == 0)
394 {
395 /* Get the 1st Input Capture value */
396 p_owner->ic_value1 = HAL_TIM_ReadCapturedValue(htim, TIM_CHANNEL_1);
397 p_owner->ic_flag = 1;
398 }
399 else if (p_owner->ic_flag == 1)
400 {
401 uint32_t uwDiffCapture;
402
403 /* Get the 2nd Input Capture value */
404 p_owner->ic_value2 = HAL_TIM_ReadCapturedValue(htim, TIM_CHANNEL_1);
405
406 /* Capture computation */
407 if (p_owner->ic_value2 > p_owner->ic_value1)
408 {
409 uwDiffCapture = (p_owner->ic_value2 - p_owner->ic_value1);
410 }
411 else if (p_owner->ic_value2 < p_owner->ic_value1)
412 {
413 /* 0xFFFF is max TIM1_CCRx value */
414 uwDiffCapture = ((0xFFFFU - p_owner->ic_value1) + p_owner->ic_value2) + 1U;
415 }
416 else
417 {
418 /* IC values are the same: it should never happen */
419 uwDiffCapture = 0xFFFFFFFFU;
420 }
421
422 /* Frequency computation */
423 uwFrequency = (float) BC_CHG_TIM_COUNTING_FREQ / (float) uwDiffCapture;
424 p_owner->ic_flag = 0;
425
426 BCPFreq2Status(uwFrequency);
427 }
428}

435static void BCPFreq2Status(float freq)
436{
437 BCProtocol_t *p_owner = sTimCallbackMap[0].p_owner;
438 ESTBC02_CHG_STATE_t ChgState;
439 float half_step;
440
441 if (freq > (float) 0)
442 {
443 for (ChgState = END_OF_CHARGE; ChgState < BATTERY_TEMPERATURE_FAULT; ChgState++)
444 {
445 half_step = (sSTBC02_ChgStateInfo[ChgState].freq + sSTBC02_ChgStateInfo[(uint8_t) ChgState + 1U].freq) / (float) 2;
446 if (freq < half_step)
447 {
448 p_owner->chg_state = ChgState;
449 break;
450 }
451 }
452 half_step = (sSTBC02_ChgStateInfo[ChgState - 1U].freq + sSTBC02_ChgStateInfo[(uint8_t) ChgState].freq) / (float) 2;
453 if (freq > half_step)
454 {
455 p_owner->chg_state = ChgState;
456 }
457 }
