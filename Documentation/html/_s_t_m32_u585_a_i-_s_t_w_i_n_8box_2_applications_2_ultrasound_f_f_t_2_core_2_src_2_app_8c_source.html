107{

109 sys_error_code_t xRes = SYS_NO_ERROR_CODE;
110
111 /* Allocate the task objects */
112 sFFTAudioObj = FDM_ACOTaskAlloc();
113 sDatalogAppObj = DatalogAppTaskAlloc();

115 sIMP23ABSUObj = IMP23ABSUTaskAlloc(&MX_MDF1InitParams, &MX_ADC1InitParams);
116
117 /* Add the task object to the context. */
118 xRes = ACAddTask(pAppContext, (AManagedTask *) sFFTAudioObj);
119 xRes = ACAddTask(pAppContext, (AManagedTask *) sDatalogAppObj);
120 xRes = ACAddTask(pAppContext, (AManagedTask *) sUtilObj);
121 xRes = ACAddTask(pAppContext, (AManagedTask *) sIMP23ABSUObj);
122
123 pIMP23ABSU_MIC_PnPLObj = Imp23absu_Mic_PnPLAlloc();
124 pLogControllerPnPLObj = Log_Controller_PnPLAlloc();

129

131
132 return xRes;
