435static sys_error_code_t I2CBusTaskExecuteStep(AManagedTask *_this)
436{
437 assert_param(_this != NULL);
438 sys_error_code_t res = SYS_NO_ERROR_CODE;
439 I2CBusTask *p_obj = (I2CBusTask *) _this;
440
441 struct i2cIOMessage_t msg =
442 {
443 0
444 };
445 AMTExSetInactiveState((AManagedTaskEx *) _this, TRUE);
446 if (TX_SUCCESS == tx_queue_receive(&p_obj->in_queue, &msg, TX_WAIT_FOREVER))
447 {
448 AMTExSetInactiveState((AManagedTaskEx *) _this, FALSE);
449 switch (msg.messageId)
450 {
451 case SM_MESSAGE_ID_FORCE_STEP:
452 __NOP();
453 /* do nothing. I need only to resume the task. */
454 break;
455
456 case SM_MESSAGE_ID_I2C_BUS_READ:
457
458 I2CMasterDriverSetDeviceAddr((I2CMasterDriver_t *) p_obj->p_driver, msg.pxSensor->address);
459 I2CMasterDriverSetAddrSize((I2CMasterDriver_t *) p_obj->p_driver, msg.pxSensor->address_size);
460 res = IIODrvRead(p_obj->p_driver, msg.pnData, msg.nDataSize, msg.nRegAddr);
461 if (!SYS_IS_ERROR_CODE(res))
462 {
463 res = I2CBusIFNotifyIOComplete(msg.pxSensor);
464 }
465 break;
466
467 case SM_MESSAGE_ID_I2C_BUS_WRITE:
468 I2CMasterDriverSetDeviceAddr((I2CMasterDriver_t *) p_obj->p_driver, msg.pxSensor->address);
469 I2CMasterDriverSetAddrSize((I2CMasterDriver_t *) p_obj->p_driver, msg.pxSensor->address_size);
470 res = IIODrvWrite(p_obj->p_driver, msg.pnData, msg.nDataSize, msg.nRegAddr);
471 if (!SYS_IS_ERROR_CODE(res))
472 {
473 res = I2CBusIFNotifyIOComplete(msg.pxSensor);
474 }
475 break;
476
477 default:
478 SYS_DEBUGF(SYS_DBG_LEVEL_WARNING, ("I2C: unsupported message id:%d\r\n", msg.messageId));
479 res = SYS_I2CBUS_TASK_UNSUPPORTED_CMD_ERROR_CODE;
480 SYS_SET_SERVICE_LEVEL_ERROR_CODE(SYS_I2CBUS_TASK_UNSUPPORTED_CMD_ERROR_CODE);
481 break;
482 }
483 }
484
485 return res;
486}
