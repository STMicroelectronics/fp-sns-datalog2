219static FFT_error_t FFT_compute_avg_output( FFT_DPU2_t * const pObj)
220{
221 assert_param(pObj != NULL);
222
223 uint32_t fft_out_data_len;
224 FFT_error_t fft_retVal;
225
227 fft_out_data_len = pObj->super.out_data.shapes[0];
228
230 fft_retVal = FFT_Process(&pObj->fft_instance, pObj->fft_temp_avg_out);
231
233 arm_add_f32(pObj->fft_temp_avg_out, pObj->fft_avg_out, pObj->fft_avg_out, fft_out_data_len);
234
236 pObj->avg_counter++;
237
239 if( pObj->avg_counter >= pObj->n_average )
240 {
242 float fft_avg_scale_factor = pObj->n_average;//FFT_COMPUTE_AVG_SCALE_FACTOR(pObj->avg_counter, fft_out_data_len);
243
245 arm_scale_f32(pObj->fft_avg_out, fft_avg_scale_factor, (float*)pObj->super.out_data.p_payload, fft_out_data_len);
246
248 pObj->avg_counter = 0u;
249
251 memset(pObj->fft_avg_out, 0, sizeof(float) * fft_out_data_len);
252 }
253
254 return fft_retVal;
