799sys_error_code_t ISM330ISTask_vtblDoEnterPowerMode(AManagedTask *_this, const EPowerMode ActivePowerMode,
800 const EPowerMode NewPowerMode)
801{
802 assert_param(_this != NULL);
803 sys_error_code_t res = SYS_NO_ERROR_CODE;
804 ISM330ISTask *p_obj = (ISM330ISTask *) _this;
805 stmdev_ctx_t *p_sensor_drv = (stmdev_ctx_t *) &p_obj->p_sensor_bus_if->m_xConnector;
806
807 if (NewPowerMode == E_POWER_MODE_SENSORS_ACTIVE)
808 {
809 if (ISM330ISTaskSensorIsActive(p_obj))
810 {
811 SMMessage report =
812 {
813 .sensorMessage.messageId = SM_MESSAGE_ID_SENSOR_CMD,
814 .sensorMessage.nCmdID = SENSOR_CMD_ID_INIT
815 };
816
817 if (tx_queue_send(&p_obj->in_queue, &report, AMT_MS_TO_TICKS(100)) != TX_SUCCESS)
818 {
819 res = SYS_SENSOR_TASK_MSG_LOST_ERROR_CODE;
820 SYS_SET_SERVICE_LEVEL_ERROR_CODE(SYS_SENSOR_TASK_MSG_LOST_ERROR_CODE);
821 }
822
823 // reset the variables for the actual odr computation.
824 p_obj->prev_timestamp = 0.0f;
825 }
826
827 SYS_DEBUGF(SYS_DBG_LEVEL_VERBOSE, ("ISM330IS: -> SENSORS_ACTIVE\r\n"));
828 }
829 else if (NewPowerMode == E_POWER_MODE_STATE1)
830 {
831 if (ActivePowerMode == E_POWER_MODE_SENSORS_ACTIVE)
832 {
833 /* Deactivate the sensor */
834 ism330is_xl_data_rate_set(p_sensor_drv, ISM330IS_XL_ODR_OFF);
835 ism330is_gy_data_rate_set(p_sensor_drv, ISM330IS_GY_ODR_OFF);
836
837 /* Empty the task queue and disable INT or timer */
838 tx_queue_flush(&p_obj->in_queue);
839 if (p_obj->p_irq_config == NULL)
840 {
841 tx_timer_deactivate(&p_obj->read_timer);
842 }
843 else
844 {
845 ISM330ISTaskConfigureIrqPin(p_obj, TRUE);
846 }
847 if (p_obj->p_ispu_config == NULL)
848 {
849 tx_timer_deactivate(&p_obj->ispu_timer);
850 }
851 else
852 {
853 ISM330ISTaskConfigureISPUPin(p_obj, TRUE);
854 }
855 }
856 SYS_DEBUGF(SYS_DBG_LEVEL_VERBOSE, ("ISM330IS: -> STATE1\r\n"));
857 }
858 else if (NewPowerMode == E_POWER_MODE_SLEEP_1)
859 {
860 /* the MCU is going in stop so I put the sensor in low power
861 from the INIT task */
862 res = ISM330ISTaskEnterLowPowerMode(p_obj);
863 if (SYS_IS_ERROR_CODE(res))
864 {
865 SYS_DEBUGF(SYS_DBG_LEVEL_WARNING, ("ISM330IS - Enter Low Power Mode failed.\r\n"));
866 }
867 if (p_obj->p_irq_config != NULL)
868 {
869 ISM330ISTaskConfigureIrqPin(p_obj, TRUE);
870 }
871 if (p_obj->p_ispu_config != NULL)
872 {
873 ISM330ISTaskConfigureISPUPin(p_obj, TRUE);
874 }
875 /* notify the bus */
876 if (p_obj->p_sensor_bus_if->m_pfBusCtrl != NULL)
877 {
878 p_obj->p_sensor_bus_if->m_pfBusCtrl(p_obj->p_sensor_bus_if, E_BUS_CTRL_DEV_NOTIFY_POWER_MODE, 0);
879 }
880 if (p_obj->p_irq_config == NULL)
881 {
882 tx_timer_deactivate(&p_obj->read_timer);
883 }
884 if (p_obj->p_ispu_config == NULL)
885 {
886 tx_timer_deactivate(&p_obj->ispu_timer);
887 }
888
889 SYS_DEBUGF(SYS_DBG_LEVEL_VERBOSE, ("ISM330IS: -> SLEEP_1\r\n"));
890 }
891
892 return res;
893}
