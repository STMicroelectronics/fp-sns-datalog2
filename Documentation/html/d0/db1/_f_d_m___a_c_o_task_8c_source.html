487static sys_error_code_t FDM_ACOTaskExecuteStepDatalog(AManagedTask *_this)
488{
489
490 sys_error_code_t xRes = SYS_NO_ERROR_CODE;
491 FDM_ACOTask *p_obj = (FDM_ACOTask *) _this;
492
493 APPReport xReport =
494 {
495 0
496 };
497
498 AMTExSetInactiveState((AManagedTaskEx *) _this, TRUE);
499
500 if (TX_SUCCESS == tx_queue_receive(&p_obj->in_queue, &xReport, TX_WAIT_FOREVER))
501 {
502 AMTExSetInactiveState((AManagedTaskEx *) _this, FALSE);
503
504 switch (xReport.msgId)
505 {
506 case APP_REPORT_ID_FORCE_STEP:
507 /* do nothing. I need only to resume. */
508 __NOP();
509 break;
510 case APP_REPORT_ID_ACOUSTIC_CMD:
511 if (xReport.acousticDomainReport.nCmdID == ACO_CMD_ID_START_ACQ)
512 {
513 __NOP();
514 }
515 else if (xReport.acousticDomainReport.nCmdID == ACO_CMD_ID_SET_DRY)
516 {
517 (void) ADPU2_ProcessAndDispatch((ADPU2_t *) p_obj->p_dpu);
518 }
519
520 break;
521
522 default:
523 xRes = SYS_APP_TASK_UNKNOWN_REPORT_ERROR_CODE;
524 SYS_SET_SERVICE_LEVEL_ERROR_CODE(SYS_APP_TASK_UNKNOWN_REPORT_ERROR_CODE)
525 ;
526
527 SYS_DEBUGF(SYS_DBG_LEVEL_WARNING, ("FDM_ACOTask: unexpected report in Datalog: %i\r\n", xReport.msgId));
528 break;
529 }
530 }
531
532 return xRes;
533
