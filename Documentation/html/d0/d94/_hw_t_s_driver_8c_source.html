200uint64_t HwTSDriver_vtblGetTimestamp(ITSDriver_t *_this)
201{
202 assert_param(_this != NULL);
203 HwTSDriver_t *pxObj = (HwTSDriver_t *)_this;
204 uint32_t nPeriod;
205 uint64_t nCounter;
206 uint64_t nTimestamp;
207 TIM_HandleTypeDef *pxTim;
208 UINT nPosture;
209
210 /* Enter a critical section*/
211 nPosture = tx_interrupt_control(TX_INT_DISABLE);
212
213 pxTim = pxObj->m_xHwHandle.pxTimParams->pxTim;
214 nCounter = LL_TIM_GetCounter(pxTim->Instance);
215 nPeriod = LL_TIM_GetAutoReload(pxTim->Instance) + 1U;
216 if (__HAL_TIM_GET_FLAG(pxTim, TIM_FLAG_UPDATE))
217 {
218 /* Update Event happened while already in critical section */
219 /* Evaluate if the timer was read before or after the "UPDATE" event */
220 if (nCounter < (nPeriod / (uint64_t)2))
221 {
222 /* After*/
223 nTimestamp = s_xHwResources.m_nTimestampTick + nCounter + nPeriod;
224 }
225 else
226 {
227 /* Before*/
228 nTimestamp = s_xHwResources.m_nTimestampTick + nCounter;
229 }
230 }
231 else
232 {
233 // No Update Event, just sum the timer value to the global TimeStamp
234 nTimestamp = s_xHwResources.m_nTimestampTick + nCounter;
235 }
236
237 /* Exit the critical section*/
238 (void)tx_interrupt_control(nPosture);
239
240 return nTimestamp;
241}
