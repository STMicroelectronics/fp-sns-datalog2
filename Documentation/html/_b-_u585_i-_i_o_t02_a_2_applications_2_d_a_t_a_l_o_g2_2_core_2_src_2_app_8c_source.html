162sys_error_code_t SysLoadApplicationContext(ApplicationContext *pAppContext)
163{
164 assert_param(pAppContext);
165 sys_error_code_t res = SYS_NO_ERROR_CODE;
166
167 /* set the allocator functions for parson */
168 json_set_allocation_functions(SysAlloc, SysFree);
169
170 /* Allocate the task objects */
171 spHelloWorldObj = HelloWorldTaskAlloc(&MX_GPIO_PH7InitParams, &MX_GPIO_PC13InitParams);
172 spDatalogAppObj = DatalogAppTaskAlloc();
173 spI2CBusObj = I2CBusTaskAlloc(&MX_I2C2InitParams);
174 spISM330DHCXObj = ISM330DHCXTaskAlloc(&MX_GPIO_PE11InitParams, NULL, NULL);
175 spIIS2MDCObj = IIS2MDCTaskAlloc(&MX_GPIO_PD10InitParams, NULL);
176 spLPS22HHObj = LPS22HHTaskAlloc(&MX_GPIO_PG2InitParams, NULL);
177 spMP23DB01HPObj = MP23DB01HPTaskAlloc(&MX_ADF1InitParams);
178
179
180 /* Add the task object to the context. */
181 res = ACAddTask(pAppContext, spHelloWorldObj);
182 res = ACAddTask(pAppContext, (AManagedTask *) spDatalogAppObj);
183 res = ACAddTask(pAppContext, (AManagedTask *)spI2CBusObj);
184 res = ACAddTask(pAppContext, (AManagedTask *)spISM330DHCXObj);
185 res = ACAddTask(pAppContext, (AManagedTask *)spIIS2MDCObj);
186 res = ACAddTask(pAppContext, (AManagedTask *)spLPS22HHObj);
187 res = ACAddTask(pAppContext, (AManagedTask *)spMP23DB01HPObj);
188
189 /* --- PnPL Components Allocation */
190 pAutomode_PnPLObj = Automode_PnPLAlloc();
191 pLog_Controller_PnPLObj = Log_Controller_PnPLAlloc();
192 pTags_Info_PnPLObj = Tags_Info_PnPLAlloc();
193 pAcquisition_Info_PnPLObj = Acquisition_Info_PnPLAlloc();
194 pFirmware_Info_PnPLObj = Firmware_Info_PnPLAlloc();
195 pDeviceinformation_PnPLObj = Deviceinformation_PnPLAlloc();
196 pIsm330dhcx_Gyro_PnPLObj = Ism330dhcx_Gyro_PnPLAlloc();
197 pIsm330dhcx_Acc_PnPLObj = Ism330dhcx_Acc_PnPLAlloc();
198 pIis2mdc_Mag_PnPLObj = Iis2mdc_Mag_PnPLAlloc();
199 pIsm330dhcx_Mlc_PnPLObj = Ism330dhcx_Mlc_PnPLAlloc();
200 pLps22hh_Press_PnPLObj = Lps22hh_Press_PnPLAlloc();
